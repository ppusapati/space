## 1. Executive Summary

The Authentication & Authorization system provides comprehensive identity and access management:

- **Multi-Tenant Identity:** Complete tenant isolation with shared infrastructure
- **OAuth2/OIDC Compliance:** Industry-standard authentication protocols
- **Fine-Grained Authorization:** RBAC + ABAC hybrid policy model
- **Zero-Trust Architecture:** Verify explicitly, least privilege, assume breach
- **Enterprise SSO:** SAML 2.0, OIDC federation, Active Directory integration
- **API Security:** Key management, rate limiting, scope-based access

## 3. Authentication Flows

### 3.1 Login Flow

```
USER LOGIN FLOW
===============

1. User submits credentials (email + password)
2. Rate limit check
3. User lookup by email
4. Check user status (active, locked, etc.)
5. Verify password hash (bcrypt)
6. Check MFA requirement
   - If MFA enabled: return challenge
   - User submits MFA code
   - Verify TOTP/SMS code
7. Create session
8. Generate tokens (access, refresh, ID)
9. Update last login timestamp
10. Audit log success
11. Return tokens to client
```

### 3.2 Authentication Service

```go
// AuthenticationService handles user authentication
type AuthenticationService struct {
    userStore      UserStore
    sessionService *SessionService
    tokenService   *TokenService
    mfaService     *MFAService
    auditLogger    AuditLogger
    rateLimiter    RateLimiter
    config         *AuthnConfig
}

// Login authenticates a user
func (as *AuthenticationService) Login(ctx context.Context, req *LoginRequest) (*LoginResponse, error) {
    response := &LoginResponse{}
    
    // 1. Rate limiting check
    if err := as.rateLimiter.Check(ctx, fmt.Sprintf("login:%s", req.Email)); err != nil {
        response.Error = "too many login attempts"
        response.ErrorCode = "RATE_LIMITED"
        return response, nil
    }
    
    // 2. Get user by email
    user, err := as.userStore.GetByEmail(ctx, req.Email)
    if err != nil {
        response.Error = "invalid credentials"
        response.ErrorCode = "INVALID_CREDENTIALS"
        return response, nil
    }
    
    // 3. Check user status
    if user.Status != UserStatusActive {
        response.Error = fmt.Sprintf("account %s", user.Status)
        response.ErrorCode = "ACCOUNT_INACTIVE"
        return response, nil
    }
    
    // 4. Check lockout
    if user.LockedUntil != nil && user.LockedUntil.After(time.Now()) {
        response.Error = "account temporarily locked"
        response.ErrorCode = "ACCOUNT_LOCKED"
        return response, nil
    }
    
    // 5. Verify password
    if err := bcrypt.CompareHashAndPassword([]byte(user.PasswordHash), []byte(req.Password)); err != nil {
        as.handleFailedLogin(ctx, user, req)
        response.Error = "invalid credentials"
        response.ErrorCode = "INVALID_CREDENTIALS"
        return response, nil
    }
    
    // 6. Check MFA requirement
    if user.MFAEnabled {
        if req.MFACode == "" {
            challenge, _ := as.mfaService.CreateChallenge(ctx, user)
            response.RequiresMFA = true
            response.MFAChallenge = challenge
            return response, nil
        }
        
        if !as.mfaService.VerifyCode(ctx, user, req.MFACode) {
            response.Error = "invalid MFA code"
            response.ErrorCode = "INVALID_MFA"
            return response, nil
        }
    }
    
    // 7. Create session
    session, _ := as.sessionService.Create(ctx, &CreateSessionRequest{
        UserID:    user.ID,
        TenantID:  user.TenantID,
        IPAddress: req.IPAddress,
        UserAgent: req.UserAgent,
    })
    
    // 8. Generate tokens
    tokens, _ := as.tokenService.GenerateTokens(ctx, &TokenRequest{
        UserID:    user.ID,
        TenantID:  user.TenantID,
        SessionID: session.ID,
        Roles:     user.TenantRoles[user.TenantID],
    })
    
    // 9. Update last login
    as.userStore.UpdateLastLogin(ctx, user.ID)
    
    // 10. Clear failed attempts & audit
    as.userStore.ClearFailedAttempts(ctx, user.ID)
    as.auditLogger.Log(ctx, &AuditEvent{
        Type:      "login_success",
        UserID:    user.ID,
        TenantID:  user.TenantID,
        SessionID: session.ID,
    })
    
    response.Success = true
    response.User = user
    response.AccessToken = tokens.AccessToken
    response.RefreshToken = tokens.RefreshToken
    response.IDToken = tokens.IDToken
    response.SessionID = session.ID
    response.ExpiresIn = int64(tokens.ExpiresIn.Seconds())
    
    return response, nil
}
```

## 4. OAuth2 & OIDC Implementation

### 4.1 OAuth2 Server

```go
// OAuth2Server implements OAuth2 authorization server
type OAuth2Server struct {
    clientStore     ClientStore
    tokenService    *TokenService
    authzCodeStore  AuthzCodeStore
    config          *OAuth2Config
}

// OAuth2Config configures OAuth2 server
type OAuth2Config struct {
    Issuer                 string
    AuthorizationEndpoint  string
    TokenEndpoint          string
    SupportedGrantTypes    []string  // authorization_code, refresh_token, client_credentials
    AuthCodeTTL            time.Duration
    RequirePKCE            bool
}

// Authorize handles authorization endpoint
func (os *OAuth2Server) Authorize(ctx context.Context, req *AuthorizationRequest, userID string) (*AuthorizationResponse, error) {
    // 1. Validate client
    client, err := os.clientStore.Get(ctx, req.ClientID)
    if err != nil {
        return nil, &OAuth2Error{Error: "invalid_client"}
    }
    
    // 2. Validate redirect URI
    if !os.isValidRedirectURI(client, req.RedirectURI) {
        return nil, &OAuth2Error{Error: "invalid_request", Description: "invalid redirect_uri"}
    }
    
    // 3. Validate PKCE (required for public clients)
    if os.config.RequirePKCE && req.CodeChallenge == "" {
        return nil, &OAuth2Error{Error: "invalid_request", Description: "code_challenge required"}
    }
    
    // 4. Generate authorization code
    code := &AuthorizationCode{
        Code:                generateSecureCode(32),
        ClientID:            client.ID,
        UserID:              userID,
        RedirectURI:         req.RedirectURI,
        Scope:               req.Scope,
        CodeChallenge:       req.CodeChallenge,
        CodeChallengeMethod: req.CodeChallengeMethod,
        ExpiresAt:           time.Now().Add(os.config.AuthCodeTTL),
    }
    
    os.authzCodeStore.Store(ctx, code)
    
    // 5. Build redirect URL
    redirectURL := fmt.Sprintf("%s?code=%s&state=%s", req.RedirectURI, code.Code, req.State)
    
    return &AuthorizationResponse{RedirectURI: redirectURL, Code: code.Code}, nil
}

// Token handles token endpoint
func (os *OAuth2Server) Token(ctx context.Context, req *TokenRequest) (*TokenResponse, error) {
    switch req.GrantType {
    case "authorization_code":
        return os.handleAuthorizationCode(ctx, req)
    case "refresh_token":
        return os.handleRefreshToken(ctx, req)
    case "client_credentials":
        return os.handleClientCredentials(ctx, req)
    default:
        return nil, &OAuth2Error{Error: "unsupported_grant_type"}
    }
}

// handleAuthorizationCode handles authorization code grant
func (os *OAuth2Server) handleAuthorizationCode(ctx context.Context, req *TokenRequest) (*TokenResponse, error) {
    // 1. Validate client
    client, err := os.authenticateClient(ctx, req.ClientID, req.ClientSecret)
    if err != nil {
        return nil, &OAuth2Error{Error: "invalid_client"}
    }
    
    // 2. Get authorization code
    code, err := os.authzCodeStore.Get(ctx, req.Code)
    if err != nil {
        return nil, &OAuth2Error{Error: "invalid_grant"}
    }
    
    // 3. Validate code
    if code.Used || time.Now().After(code.ExpiresAt) || code.ClientID != client.ID {
        return nil, &OAuth2Error{Error: "invalid_grant"}
    }
    
    // 4. Validate PKCE
    if code.CodeChallenge != "" {
        if !os.verifyCodeVerifier(code.CodeChallenge, code.CodeChallengeMethod, req.CodeVerifier) {
            return nil, &OAuth2Error{Error: "invalid_grant", Description: "invalid code_verifier"}
        }
    }
    
    // 5. Mark code as used
    os.authzCodeStore.MarkUsed(ctx, req.Code)
    
    // 6. Generate tokens
    tokens, _ := os.tokenService.GenerateOAuth2Tokens(ctx, &OAuth2TokenRequest{
        UserID:   code.UserID,
        ClientID: client.ID,
        Scopes:   strings.Split(code.Scope, " "),
    })
    
    return tokens, nil
}

// verifyCodeVerifier verifies PKCE code verifier (S256)
func (os *OAuth2Server) verifyCodeVerifier(challenge, method, verifier string) bool {
    if method == "S256" || method == "" {
        h := sha256.Sum256([]byte(verifier))
        computed := base64.RawURLEncoding.EncodeToString(h[:])
        return computed == challenge
    }
    return verifier == challenge // plain method
}
```

### 4.2 OIDC Provider

```go
// OIDCProvider implements OpenID Connect
type OIDCProvider struct {
    oauth2Server *OAuth2Server
    privateKey   *rsa.PrivateKey
    config       *OIDCConfig
}

// GetDiscovery returns OIDC discovery document
func (op *OIDCProvider) GetDiscovery() *OIDCDiscovery {
    return &OIDCDiscovery{
        Issuer:                           op.config.Issuer,
        AuthorizationEndpoint:            op.config.Issuer + "/oauth2/authorize",
        TokenEndpoint:                    op.config.Issuer + "/oauth2/token",
        UserinfoEndpoint:                 op.config.Issuer + "/oauth2/userinfo",
        JwksURI:                          op.config.Issuer + "/.well-known/jwks.json",
        ScopesSupported:                  []string{"openid", "profile", "email", "offline_access"},
        ResponseTypesSupported:           []string{"code", "token", "id_token"},
        GrantTypesSupported:              []string{"authorization_code", "refresh_token"},
        SubjectTypesSupported:            []string{"public"},
        IDTokenSigningAlgValuesSupported: []string{"RS256"},
        CodeChallengeMethodsSupported:    []string{"S256"},
    }
}

// GenerateIDToken generates an ID token
func (op *OIDCProvider) GenerateIDToken(ctx context.Context, user *User, clientID, nonce string, scopes []string) (string, error) {
    claims := &IDTokenClaims{
        RegisteredClaims: jwt.RegisteredClaims{
            Issuer:    op.config.Issuer,
            Subject:   user.ID,
            Audience:  jwt.ClaimStrings{clientID},
            IssuedAt:  jwt.NewNumericDate(time.Now()),
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(op.config.IDTokenTTL)),
        },
        Nonce:    nonce,
        TenantID: user.TenantID,
    }
    
    // Add profile claims if scope includes "profile"
    if containsScope(scopes, "profile") {
        claims.Name = user.DisplayName
        claims.GivenName = user.FirstName
        claims.FamilyName = user.LastName
    }
    
    // Add email claims if scope includes "email"
    if containsScope(scopes, "email") {
        claims.Email = user.Email
        claims.EmailVerified = user.EmailVerified
    }
    
    token := jwt.NewWithClaims(jwt.SigningMethodRS256, claims)
    return token.SignedString(op.privateKey)
}
```

## 5. JWT Token Management

### 5.1 Token Service

```go
// TokenService manages JWT tokens
type TokenService struct {
    privateKey   *rsa.PrivateKey
    publicKey    *rsa.PublicKey
    tokenStore   TokenStore
    config       *TokenConfig
}

// TokenConfig configures token service
type TokenConfig struct {
    Issuer           string
    Audience         []string
    AccessTokenTTL   time.Duration  // 15 minutes
    RefreshTokenTTL  time.Duration  // 7 days
    RotateRefresh    bool
}

// AccessTokenClaims represents access token claims
type AccessTokenClaims struct {
    jwt.RegisteredClaims
    UserID      string   `json:"uid"`
    TenantID    string   `json:"tid"`
    SessionID   string   `json:"sid,omitempty"`
    Roles       []string `json:"roles,omitempty"`
    Permissions []string `json:"permissions,omitempty"`
    Type        string   `json:"type"`
}

// GenerateTokens generates a token set
func (ts *TokenService) GenerateTokens(ctx context.Context, req *TokenRequest) (*TokenSet, error) {
    now := time.Now()
    
    // Generate access token
    accessClaims := &AccessTokenClaims{
        RegisteredClaims: jwt.RegisteredClaims{
            Issuer:    ts.config.Issuer,
            Subject:   req.UserID,
            Audience:  jwt.ClaimStrings(ts.config.Audience),
            IssuedAt:  jwt.NewNumericDate(now),
            ExpiresAt: jwt.NewNumericDate(now.Add(ts.config.AccessTokenTTL)),
            ID:        generateID(),
        },
        UserID:      req.UserID,
        TenantID:    req.TenantID,
        SessionID:   req.SessionID,
        Roles:       req.Roles,
        Permissions: ts.getPermissionsForRoles(ctx, req.TenantID, req.Roles),
        Type:        "access",
    }
    
    accessToken := jwt.NewWithClaims(jwt.SigningMethodRS256, accessClaims)
    accessTokenString, _ := accessToken.SignedString(ts.privateKey)
    
    // Generate refresh token
    refreshToken := &RefreshToken{
        ID:        generateID(),
        Token:     generateSecureCode(64),
        UserID:    req.UserID,
        TenantID:  req.TenantID,
        SessionID: req.SessionID,
        ExpiresAt: now.Add(ts.config.RefreshTokenTTL),
    }
    ts.tokenStore.StoreRefreshToken(ctx, refreshToken)
    
    return &TokenSet{
        AccessToken:  accessTokenString,
        RefreshToken: refreshToken.Token,
        TokenType:    "Bearer",
        ExpiresIn:    ts.config.AccessTokenTTL,
    }, nil
}

// ValidateAccessToken validates an access token
func (ts *TokenService) ValidateAccessToken(ctx context.Context, tokenString string) (*AccessTokenClaims, error) {
    token, err := jwt.ParseWithClaims(tokenString, &AccessTokenClaims{}, func(token *jwt.Token) (interface{}, error) {
        if _, ok := token.Method.(*jwt.SigningMethodRSA); !ok {
            return nil, fmt.Errorf("unexpected signing method")
        }
        return ts.publicKey, nil
    })
    
    if err != nil {
        return nil, fmt.Errorf("invalid token: %w", err)
    }
    
    claims, ok := token.Claims.(*AccessTokenClaims)
    if !ok || !token.Valid {
        return nil, fmt.Errorf("invalid token claims")
    }
    
    // Check if token is revoked
    if ts.tokenStore.IsRevoked(ctx, claims.ID) {
        return nil, fmt.Errorf("token revoked")
    }
    
    return claims, nil
}

// RefreshAccessToken refreshes tokens using refresh token
func (ts *TokenService) RefreshAccessToken(ctx context.Context, refreshTokenStr string) (*TokenSet, error) {
    refreshToken, err := ts.tokenStore.GetRefreshToken(ctx, refreshTokenStr)
    if err != nil {
        return nil, fmt.Errorf("invalid refresh token")
    }
    
    if time.Now().After(refreshToken.ExpiresAt) {
        return nil, fmt.Errorf("refresh token expired")
    }
    
    if refreshToken.Revoked {
        // Potential token theft - revoke all tokens
        ts.tokenStore.RevokeByUser(ctx, refreshToken.UserID)
        return nil, fmt.Errorf("refresh token revoked")
    }
    
    // Generate new tokens
    newTokens, _ := ts.GenerateTokens(ctx, &TokenRequest{
        UserID:    refreshToken.UserID,
        TenantID:  refreshToken.TenantID,
        SessionID: refreshToken.SessionID,
    })
    
    // Rotate refresh token
    if ts.config.RotateRefresh {
        ts.tokenStore.RevokeRefreshToken(ctx, refreshTokenStr)
    }
    
    return newTokens, nil
}
```

## 6. Session Management

### 6.1 Session Service

```go
// SessionService manages user sessions
type SessionService struct {
    store  SessionStore
    config *SessionConfig
}

// SessionConfig configures session management
type SessionConfig struct {
    SessionTTL         time.Duration  // 24 hours
    IdleTimeout        time.Duration  // 1 hour
    MaxSessionsPerUser int            // 5
}

// Session represents a user session
type Session struct {
    ID             string
    UserID         string
    TenantID       string
    DeviceInfo     *DeviceInfo
    IPAddress      string
    UserAgent      string
    CreatedAt      time.Time
    ExpiresAt      time.Time
    LastActivityAt time.Time
    Revoked        bool
}

// Create creates a new session
func (ss *SessionService) Create(ctx context.Context, req *CreateSessionRequest) (*Session, error) {
    // Check session limit
    existingSessions, _ := ss.store.GetByUser(ctx, req.UserID)
    
    if len(existingSessions) >= ss.config.MaxSessionsPerUser {
        // Revoke oldest session
        oldest := existingSessions[0]
        for _, s := range existingSessions {
            if s.CreatedAt.Before(oldest.CreatedAt) {
                oldest = s
            }
        }
        ss.Revoke(ctx, oldest.ID)
    }
    
    now := time.Now()
    session := &Session{
        ID:             generateSecureCode(32),
        UserID:         req.UserID,
        TenantID:       req.TenantID,
        DeviceInfo:     req.DeviceInfo,
        IPAddress:      req.IPAddress,
        UserAgent:      req.UserAgent,
        CreatedAt:      now,
        ExpiresAt:      now.Add(ss.config.SessionTTL),
        LastActivityAt: now,
    }
    
    ss.store.Create(ctx, session)
    return session, nil
}

// Validate validates a session and updates activity
func (ss *SessionService) Validate(ctx context.Context, sessionID string) (*Session, error) {
    session, err := ss.store.Get(ctx, sessionID)
    if err != nil {
        return nil, err
    }
    
    // Check expiration
    if time.Now().After(session.ExpiresAt) {
        return nil, fmt.Errorf("session expired")
    }
    
    // Check revocation
    if session.Revoked {
        return nil, fmt.Errorf("session revoked")
    }
    
    // Check idle timeout
    if time.Since(session.LastActivityAt) > ss.config.IdleTimeout {
        ss.Revoke(ctx, sessionID)
        return nil, fmt.Errorf("session idle timeout")
    }
    
    // Update last activity
    session.LastActivityAt = time.Now()
    ss.store.UpdateActivity(ctx, sessionID, session.LastActivityAt)
    
    return session, nil
}

// Revoke revokes a session
func (ss *SessionService) Revoke(ctx context.Context, sessionID string) error {
    return ss.store.Revoke(ctx, sessionID)
}

// RevokeAll revokes all sessions for a user
func (ss *SessionService) RevokeAll(ctx context.Context, userID string) error {
    sessions, _ := ss.store.GetByUser(ctx, userID)
    for _, session := range sessions {
        ss.Revoke(ctx, session.ID)
    }
    return nil
}
```

```go
        TenantID: tenantID,
        UserID:   userID,
        Roles:    roles,
        JoinedAt: time.Now(),
    })
}
```

## 8. Authorization Framework

### 8.1 Authorization Service

```go
// AuthorizationService handles authorization
type AuthorizationService struct {
    rbacEngine  *RBACEngine
    abacEngine  *ABACEngine
    auditLogger AuditLogger
    config      *AuthzConfig
}

// AuthzConfig configures authorization
type AuthzConfig struct {
    DefaultPolicy PolicyEffect  // deny
    EnableRBAC    bool
    EnableABAC    bool
}

// AuthorizationRequest represents an authorization request
type AuthorizationRequest struct {
    Principal *Principal
    Action    string
    Resource  string
    Context   map[string]interface{}
}

// AuthorizationResult represents authorization result
type AuthorizationResult struct {
    Allowed       bool
    Effect        PolicyEffect
    Reason        string
    MatchedPolicy string
}

// Authorize evaluates an authorization request
func (as *AuthorizationService) Authorize(ctx context.Context, req *AuthorizationRequest) (*AuthorizationResult, error) {
    result := &AuthorizationResult{Effect: as.config.DefaultPolicy}
    
    // 1. Check RBAC
    if as.config.EnableRBAC {
        rbacResult := as.rbacEngine.Evaluate(ctx, req)
        if rbacResult.Matched {
            result.Allowed = rbacResult.Effect == EffectAllow
            result.Effect = rbacResult.Effect
            result.Reason = rbacResult.Reason
            result.MatchedPolicy = rbacResult.MatchedRole
        }
    }
    
    // 2. Check ABAC (can override RBAC)
    if as.config.EnableABAC {
        abacResult := as.abacEngine.Evaluate(ctx, req)
        if abacResult.Matched {
            // ABAC deny always takes precedence
            if abacResult.Effect == EffectDeny {
                result.Allowed = false
                result.Effect = EffectDeny
                result.Reason = abacResult.Reason
            } else if !result.Allowed && abacResult.Effect == EffectAllow {
                result.Allowed = true
                result.Effect = EffectAllow
            }
        }
    }
    
    // 3. Audit log
    as.auditLogger.Log(ctx, &AuditEvent{
        Type:     "authorization",
        UserID:   req.Principal.UserID,
        Action:   req.Action,
        Resource: req.Resource,
        Allowed:  result.Allowed,
    })
    
    return result, nil
}

// CheckPermission checks a single permission
func (as *AuthorizationService) CheckPermission(ctx context.Context, principal *Principal, permission string) bool {
    // Check direct permissions
    for _, p := range principal.Permissions {
        if matchPermission(p, permission) {
            return true
        }
    }
    
    // Check role permissions
    for _, role := range principal.Roles {
        perms := as.rbacEngine.GetRolePermissions(ctx, principal.TenantID, role)
        for _, p := range perms {
            if matchPermission(p, permission) {
                return true
            }
        }
    }
    
    return false
}

// matchPermission matches permission with wildcard support
func matchPermission(pattern, permission string) bool {
    if pattern == "*" || pattern == permission {
        return true
    }
    // Wildcard: "spacecraft:*" matches "spacecraft:read"
    if strings.HasSuffix(pattern, ":*") {
        prefix := strings.TrimSuffix(pattern, "*")
        return strings.HasPrefix(permission, prefix)
    }
    return false
}
```

```go
// Evaluate evaluates RBAC for a request
func (re *RBACEngine) Evaluate(ctx context.Context, req *AuthorizationRequest) *RBACResult {
    result := &RBACResult{}
    
    requiredPerm := req.Action
    
    // Check each role
    for _, roleName := range req.Principal.Roles {
        if re.roleHasPermission(ctx, req.Principal.TenantID, roleName, requiredPerm, 0) {
            result.Matched = true
            result.Effect = EffectAllow
            result.Reason = fmt.Sprintf("role '%s' grants permission", roleName)
            result.MatchedRole = roleName
            return result
        }
    }
    
    // Check direct permissions
    for _, perm := range req.Principal.Permissions {
        if matchPermission(perm, requiredPerm) {
            result.Matched = true
            result.Effect = EffectAllow
            result.Reason = fmt.Sprintf("direct permission '%s'", perm)
            return result
        }
    }
    
    return result
}

// roleHasPermission checks if role has permission (with inheritance)
func (re *RBACEngine) roleHasPermission(ctx context.Context, tenantID, roleName, permission string, depth int) bool {
    if depth > re.config.MaxInheritDepth {
        return false
    }
    
    role, _ := re.roleStore.Get(ctx, tenantID, roleName)
    
    // Check direct permissions
    for _, perm := range role.Permissions {
        if matchPermission(perm, permission) {
            return true
        }
    }
    
    // Check parent roles (inheritance)
    if re.config.EnableInheritance {
        for _, parentName := range role.ParentRoles {
            if re.roleHasPermission(ctx, tenantID, parentName, permission, depth+1) {
                return true
            }
        }
    }
    
    return false
}

// GetRolePermissions gets all permissions for a role
func (re *RBACEngine) GetRolePermissions(ctx context.Context, tenantID, roleName string) []string {
    visited := make(map[string]bool)
    return re.getRolePermissionsRecursive(ctx, tenantID, roleName, 0, visited)
}

func (re *RBACEngine) getRolePermissionsRecursive(ctx context.Context, tenantID, roleName string, depth int, visited map[string]bool) []string {
    if depth > re.config.MaxInheritDepth || visited[roleName] {
        return nil
    }
    visited[roleName] = true
    
    role, _ := re.roleStore.Get(ctx, tenantID, roleName)
    permissions := append([]string{}, role.Permissions...)
    
    if re.config.EnableInheritance {
        for _, parentName := range role.ParentRoles {
            parentPerms := re.getRolePermissionsRecursive(ctx, tenantID, parentName, depth+1, visited)
            permissions = append(permissions, parentPerms...)
        }
    }
    
    return permissions
}
```

### 10.1 ABAC Engine

```go
// ABACEngine implements Attribute-Based Access Control
type ABACEngine struct {
    policyStore PolicyStore
    config      *ABACConfig
}

// Policy represents an ABAC policy
type Policy struct {
    ID         string
    TenantID   string
    Name       string
    Priority   int
    Effect     PolicyEffect  // allow, deny
    Subjects   *SubjectCondition
    Resources  *ResourceCondition
    Actions    []string
    Conditions []*Condition
    Enabled    bool
}

// Condition represents a policy condition
type Condition struct {
    Key      string             // e.g., "context.time_of_day"
    Operator ConditionOperator  // equals, gt, lt, in, etc.
    Value    interface{}
}

// ConditionOperator defines operators
type ConditionOperator string

const (
    OpEquals    ConditionOperator = "equals"
    OpNotEquals ConditionOperator = "not_equals"
    OpContains  ConditionOperator = "contains"
    OpIn        ConditionOperator = "in"
    OpNotIn     ConditionOperator = "not_in"
    OpGT        ConditionOperator = "gt"
    OpLT        ConditionOperator = "lt"
)

// Evaluate evaluates ABAC policies
func (ae *ABACEngine) Evaluate(ctx context.Context, req *AuthorizationRequest) *ABACResult {
    result := &ABACResult{}
    
    policies, _ := ae.policyStore.GetPolicies(ctx, req.Principal.TenantID)
    
    // Sort by priority
    sort.Slice(policies, func(i, j int) bool {
        return policies[i].Priority < policies[j].Priority
    })
    
    for _, policy := range policies {
        if !policy.Enabled {
            continue
        }
        
        if ae.policyMatches(ctx, policy, req) {
            result.Matched = true
            result.Effect = policy.Effect
            result.Reason = fmt.Sprintf("policy '%s' matched", policy.Name)
            result.MatchedPolicy = policy.ID
            
            // First deny wins
            if policy.Effect == EffectDeny {
                return result
            }
        }
    }
    
    return result
}

// policyMatches checks if policy matches request
func (ae *ABACEngine) policyMatches(ctx context.Context, policy *Policy, req *AuthorizationRequest) bool {
    // Check action
    if !ae.matchAction(policy.Actions, req.Action) {
        return false
    }
    
    // Check subject
    if !ae.matchSubject(policy.Subjects, req.Principal) {
        return false
    }
    
    // Check resource
    if !ae.matchResource(policy.Resources, req.Resource, req.Context) {
        return false
    }
    
    // Evaluate conditions
    if !ae.evaluateConditions(policy.Conditions, req.Context) {
        return false
    }
    
    return true
}

// evaluateConditions evaluates policy conditions
func (ae *ABACEngine) evaluateConditions(conditions []*Condition, context map[string]interface{}) bool {
    for _, cond := range conditions {
        value := ae.getNestedValue(context, cond.Key)
        
        switch cond.Operator {
        case OpEquals:
            if fmt.Sprintf("%v", value) != fmt.Sprintf("%v", cond.Value) {
                return false
            }
        case OpNotEquals:
            if fmt.Sprintf("%v", value) == fmt.Sprintf("%v", cond.Value) {
                return false
            }
        case OpIn:
            list, _ := cond.Value.([]interface{})
            found := false
            for _, item := range list {
                if fmt.Sprintf("%v", value) == fmt.Sprintf("%v", item) {
                    found = true
                    break
                }
            }
            if !found {
                return false
            }
        case OpGT:
            if compareValues(value, cond.Value) <= 0 {
                return false
            }
        case OpLT:
            if compareValues(value, cond.Value) >= 0 {
                return false
            }
        }
    }
    return true
}
```

```go
// Create creates a new API key
func (aks *APIKeyService) Create(ctx context.Context, req *CreateAPIKeyRequest) (*CreateAPIKeyResponse, error) {
    // Check limits
    userKeyCount, _ := aks.store.CountByUser(ctx, req.UserID)
    if userKeyCount >= aks.config.MaxKeysPerUser {
        return nil, fmt.Errorf("max API keys per user reached")
    }
    
    // Generate key
    keyBytes := make([]byte, aks.config.KeyLength)
    rand.Read(keyBytes)
    
    plainKey := fmt.Sprintf("%s_%s", aks.config.KeyPrefix, hex.EncodeToString(keyBytes))
    
    // Hash for storage
    hash := sha256.Sum256([]byte(plainKey))
    keyHash := hex.EncodeToString(hash[:])
    
    // Calculate expiry
    var expiresAt *time.Time
    if req.ExpiresIn > 0 {
        exp := time.Now().Add(req.ExpiresIn)
        expiresAt = &exp
    }
    
    apiKey := &APIKey{
        ID:          generateID(),
        KeyHash:     keyHash,
        KeyPrefix:   plainKey[:len(aks.config.KeyPrefix)+8],
        Name:        req.Name,
        TenantID:    req.TenantID,
        UserID:      req.UserID,
        Scopes:      req.Scopes,
        Permissions: req.Permissions,
        ExpiresAt:   expiresAt,
        CreatedAt:   time.Now(),
    }
    
    aks.store.Create(ctx, apiKey)
    
    return &CreateAPIKeyResponse{
        APIKey:   apiKey,
        PlainKey: plainKey, // Only returned once
    }, nil
}

// Validate validates an API key
func (aks *APIKeyService) Validate(ctx context.Context, plainKey string) (*APIKey, error) {
    hash := sha256.Sum256([]byte(plainKey))
    keyHash := hex.EncodeToString(hash[:])
    
    apiKey, err := aks.store.GetByHash(ctx, keyHash)
    if err != nil {
        return nil, fmt.Errorf("invalid API key")
    }
    
    if apiKey.Revoked {
        return nil, fmt.Errorf("API key revoked")
    }
    
    if apiKey.ExpiresAt != nil && time.Now().After(*apiKey.ExpiresAt) {
        return nil, fmt.Errorf("API key expired")
    }
    
    // Update last used
    now := time.Now()
    apiKey.LastUsedAt = &now
    aks.store.UpdateLastUsed(ctx, apiKey.ID, now)
    
    return apiKey, nil
}

// Revoke revokes an API key
func (aks *APIKeyService) Revoke(ctx context.Context, keyID, revokedBy string) error {
    return aks.store.Revoke(ctx, keyID, revokedBy)
}

// Rotate creates a new key and revokes the old one
func (aks *APIKeyService) Rotate(ctx context.Context, keyID string) (*CreateAPIKeyResponse, error) {
    oldKey, _ := aks.store.Get(ctx, keyID)
    
    newKey, err := aks.Create(ctx, &CreateAPIKeyRequest{
        Name:        oldKey.Name + " (rotated)",
        TenantID:    oldKey.TenantID,
        UserID:      oldKey.UserID,
        Scopes:      oldKey.Scopes,
        Permissions: oldKey.Permissions,
    })
    if err != nil {
        return nil, err
    }
    
    aks.Revoke(ctx, keyID, oldKey.UserID)
    
    return newKey, nil
}
```

```go
// RateLimiter implements rate limiting
type RateLimiter struct {
    windows map[string]*SlidingWindow
    config  *RateLimitConfig
    mu      sync.RWMutex
}

// RateLimitConfig configures rate limits
type RateLimitConfig struct {
    LoginAttemptsPerMinute int  // 10
    APIRequestsPerSecond   int  // 100
    PasswordResetsPerHour  int  // 3
}

// Check checks rate limit
func (rl *RateLimiter) Check(ctx context.Context, key string) error {
    rl.mu.Lock()
    defer rl.mu.Unlock()
    
    window, exists := rl.windows[key]
    if !exists {
        window = &SlidingWindow{
            Requests: make([]time.Time, 0),
            Limit:    rl.config.LoginAttemptsPerMinute,
            Window:   time.Minute,
        }
        rl.windows[key] = window
    }
    
    now := time.Now()
    
    // Clean old requests
    cutoff := now.Add(-window.Window)
    newRequests := make([]time.Time, 0)
    for _, t := range window.Requests {
        if t.After(cutoff) {
            newRequests = append(newRequests, t)
        }
    }
    window.Requests = newRequests
    
    if len(window.Requests) >= window.Limit {
        return fmt.Errorf("rate limit exceeded")
    }
    
    window.Requests = append(window.Requests, now)
    return nil
}
```

```go
// BruteForceProtection protects against brute force attacks
type BruteForceProtection struct {
    attempts map[string]*AttemptTracker
    config   *BruteForceConfig
    mu       sync.RWMutex
}

// BruteForceConfig configures brute force protection
type BruteForceConfig struct {
    MaxAttempts     int           // 5
    LockoutDuration time.Duration // 30 minutes
    ResetAfter      time.Duration // 1 hour
}

// RecordAttempt records a login attempt
func (bf *BruteForceProtection) RecordAttempt(key string, success bool) {
    bf.mu.Lock()
    defer bf.mu.Unlock()
    
    tracker, exists := bf.attempts[key]
    if !exists {
        tracker = &AttemptTracker{FirstTry: time.Now()}
        bf.attempts[key] = tracker
    }
    
    tracker.LastTry = time.Now()
    
    if success {
        delete(bf.attempts, key)
        return
    }
    
    tracker.Count++
    
    if tracker.Count >= bf.config.MaxAttempts {
        lockUntil := time.Now().Add(bf.config.LockoutDuration)
        tracker.LockedUntil = &lockUntil
    }
}

// IsLocked checks if account/IP is locked
func (bf *BruteForceProtection) IsLocked(key string) bool {
    bf.mu.RLock()
    defer bf.mu.RUnlock()
    
    tracker, exists := bf.attempts[key]
    if !exists {
        return false
    }
    
    if tracker.LockedUntil != nil && time.Now().Before(*tracker.LockedUntil) {
        return true
    }
    
    return false
}
```
