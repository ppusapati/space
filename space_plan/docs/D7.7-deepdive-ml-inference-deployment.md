## 1. Executive Summary

### 1.1 Purpose

The ML Inference Deployment system provides a scalable, production-grade infrastructure for serving machine learning models on Earth Observation data:

- **Multi-Model Serving:** Support for TensorFlow, PyTorch, ONNX, and TensorRT
- **Scalable Infrastructure:** Auto-scaling from single GPU to distributed clusters
- **Low Latency:** Sub-second inference for real-time applications
- **High Throughput:** Process millions of tiles per day in batch mode
- **Model Management:** Versioning, A/B testing, and gradual rollouts

### 1.2 Key Metrics

| Metric | Target | Description |
|--------|--------|-------------|
| **Inference Latency** | < 100ms | P95 for single tile (256×256) |
| **Batch Throughput** | 10K tiles/min | Per GPU node |
| **Model Load Time** | < 30s | Cold start to ready |

```
   Check Version ──▶ Select Server ──▶ Route to GPU/CPU

   Decode Output ──▶ Apply Threshold ──▶ Merge Tiles ──▶ Format Response
```

```go
// triton_client.go

package mlserving

import (
    "context"
    triton "github.com/triton-inference-server/client/src/go/grpc_generated"
    "google.golang.org/grpc"
)

// TritonClient wraps Triton Inference Server
type TritonClient struct {
    conn      *grpc.ClientConn
    client    triton.GRPCInferenceServiceClient
    modelName string
    modelVer  string
    timeout   time.Duration
}

// Infer performs inference on input tensors
func (c *TritonClient) Infer(ctx context.Context, inputs []*Tensor) ([]*Tensor, error) {
    ctx, cancel := context.WithTimeout(ctx, c.timeout)
    defer cancel()
    
    request := &triton.ModelInferRequest{
        ModelName:    c.modelName,
        ModelVersion: c.modelVer,
    }
    
    // Convert inputs to Triton format
    for _, input := range inputs {
        request.Inputs = append(request.Inputs, &triton.ModelInferRequest_InferInputTensor{
            Name:     input.Name,
            Datatype: input.DataType,
            Shape:    input.Shape,
            Contents: &triton.InferTensorContents{Fp32Contents: input.Float32Data},
        })
    }
    
    request.Outputs = []*triton.ModelInferRequest_InferRequestedOutputTensor{{Name: "output"}}
    
    response, err := c.client.ModelInfer(ctx, request)
    if err != nil {
        return nil, err
    }
    
    // Parse outputs
    outputs := make([]*Tensor, len(response.Outputs))
    for i, out := range response.Outputs {
        outputs[i] = &Tensor{
            Name:        out.Name,
            Shape:       out.Shape,
            DataType:    out.Datatype,
            Float32Data: response.RawOutputContents[i],
        }
    }
    
    return outputs, nil
}

// Tensor represents inference tensor
type Tensor struct {
    Name        string
    Shape       []int64
    DataType    string
    Float32Data []float32
}
```

### 5.1 Pipeline Implementation

```go
// pipeline.go

package mlserving

// InferencePipeline orchestrates the inference workflow
type InferencePipeline struct {
    preprocessor  *Preprocessor
    modelRouter   *ModelRouter
    postprocessor *Postprocessor
    resultStore   *ResultStore
    config        *PipelineConfig
}

// PipelineConfig configures the pipeline
type PipelineConfig struct {
    TileSize       int       // 256, 512
    TileOverlap    int       // pixels
    BatchSize      int       // max batch size
    MaxConcurrency int       // parallel processing
    NormalizeMean  []float64
    NormalizeStd   []float64
}

// InferenceRequest represents an inference request
type InferenceRequest struct {
    ID           string            `json:"id"`
    ImageSource  *ImageSource      `json:"image_source"`
    ModelID      string            `json:"model_id"`
    ModelVersion string            `json:"model_version,omitempty"`
    AOI          *GeoJSON          `json:"aoi,omitempty"`
    Options      *InferenceOptions `json:"options,omitempty"`
}

// InferenceResult represents inference output
type InferenceResult struct {
    ID             string                 `json:"id"`
    RequestID      string                 `json:"request_id"`
    ModelID        string                 `json:"model_id"`
    ModelVersion   string                 `json:"model_version"`
    Status         string                 `json:"status"`
    Mask           *RasterOutput          `json:"mask,omitempty"`
    Vectors        *VectorOutput          `json:"vectors,omitempty"`
    Detections     []Detection            `json:"detections,omitempty"`
    ProcessingTime float64                `json:"processing_time_ms"`
}

// Execute runs the inference pipeline
func (p *InferencePipeline) Execute(ctx context.Context, req *InferenceRequest) (*InferenceResult, error) {
    startTime := time.Now()
    
    result := &InferenceResult{
        ID:        generateID(),
        RequestID: req.ID,
        ModelID:   req.ModelID,
        Status:    "processing",
    }
    
    // 1. Load and preprocess image
    tiles, metadata, err := p.preprocessor.Process(ctx, req.ImageSource, req.AOI)
    if err != nil {
        return nil, fmt.Errorf("preprocessing failed: %w", err)
    }
    
    // 2. Get model client
    client, version, err := p.modelRouter.GetClient(ctx, req.ModelID, req.ModelVersion)
    if err != nil {
        return nil, fmt.Errorf("model routing failed: %w", err)
    }
    result.ModelVersion = version
    
    // 3. Run inference on tiles (batched)
    outputs, err := p.runInference(ctx, client, tiles)
    if err != nil {
        return nil, fmt.Errorf("inference failed: %w", err)
    }
    
    // 4. Postprocess results
    err = p.postprocessor.Process(ctx, outputs, metadata, req.Options, result)
    if err != nil {
        return nil, fmt.Errorf("postprocessing failed: %w", err)
    }
    
    result.Status = "completed"
    result.ProcessingTime = time.Since(startTime).Seconds() * 1000
    
    return result, nil
}

// runInference executes batched inference
func (p *InferencePipeline) runInference(ctx context.Context, client InferenceClient, tiles []*Tile) ([]*TileOutput, error) {
    outputs := make([]*TileOutput, len(tiles))
    
    // Process in batches
    for start := 0; start < len(tiles); start += p.config.BatchSize {
        end := start + p.config.BatchSize
        if end > len(tiles) {
            end = len(tiles)
        }
        
        batch := tiles[start:end]
        tensors := make([]*Tensor, len(batch))
        for i, tile := range batch {
            tensors[i] = tile.ToTensor()
        }
        
        // Run inference
        outputTensors, err := client.Infer(ctx, tensors)
        if err != nil {
            return nil, err
        }
        
        // Store outputs
        for i, tile := range batch {
            outputs[start+i] = &TileOutput{Tile: tile, Output: outputTensors[i]}
        }
    }
    
    return outputs, nil
}
```

### 5.2 Preprocessor

```go
// preprocessor.go

// Preprocessor handles image loading and preprocessing
type Preprocessor struct {
    config    *PipelineConfig
    cogReader *COGReader
}

// Process loads and tiles the input image
func (p *Preprocessor) Process(ctx context.Context, source *ImageSource, aoi *GeoJSON) ([]*Tile, *ImageMetadata, error) {
    // Load image data
    img, metadata, err := p.loadImage(ctx, source)
    if err != nil {
        return nil, nil, err
    }
    
    // Apply AOI mask if provided
    if aoi != nil {
        img, metadata = p.applyAOI(img, metadata, aoi)
    }
    
    // Create overlapping tiles
    tiles := p.createTiles(img, metadata)
    
    // Normalize tiles
    for _, tile := range tiles {
        p.normalizeTile(tile)
    }
    
    return tiles, metadata, nil
}

// createTiles divides image into overlapping tiles
func (p *Preprocessor) createTiles(data []float32, metadata *ImageMetadata) []*Tile {
    tileSize := p.config.TileSize
    overlap := p.config.TileOverlap
    stride := tileSize - overlap
    
    width := metadata.Width
    height := metadata.Height
    
    numTilesX := int(math.Ceil(float64(width-overlap) / float64(stride)))
    numTilesY := int(math.Ceil(float64(height-overlap) / float64(stride)))
    
    tiles := make([]*Tile, 0, numTilesX*numTilesY)
    
    for ty := 0; ty < numTilesY; ty++ {
        for tx := 0; tx < numTilesX; tx++ {
            x := tx * stride
            y := ty * stride
            
            // Clip to bounds
            if x + tileSize > width {
                x = width - tileSize
            }
            if y + tileSize > height {
                y = height - tileSize
            }
            
            // Extract tile data
            tileData := p.extractTileData(data, x, y, tileSize, metadata)
            geoTransform := p.tileGeoTransform(metadata.GeoTransform, x, y)
            
            tiles = append(tiles, &Tile{
                X: x, Y: y,
                Width: tileSize, Height: tileSize,
                Data: tileData,
                GeoTransform: geoTransform,
            })
        }
    }
    
    return tiles
}

// normalizeTile applies ImageNet-style normalization
func (p *Preprocessor) normalizeTile(tile *Tile) {
    mean := p.config.NormalizeMean // e.g., [0.485, 0.456, 0.406]
    std := p.config.NormalizeStd   // e.g., [0.229, 0.224, 0.225]
    
    pixelsPerBand := tile.Width * tile.Height
    numBands := len(tile.Data) / pixelsPerBand
    
    for b := 0; b < numBands; b++ {
        offset := b * pixelsPerBand
        for i := 0; i < pixelsPerBand; i++ {
            tile.Data[offset+i] = (tile.Data[offset+i]/255.0 - float32(mean[b])) / float32(std[b])
        }
    }
}
```

### 5.3 Postprocessor

```go
// postprocessor.go

// Postprocessor handles inference output processing
type Postprocessor struct {
    config *PipelineConfig
}

// Process postprocesses inference outputs
func (p *Postprocessor) Process(ctx context.Context, outputs []*TileOutput,
    metadata *ImageMetadata, options *InferenceOptions, result *InferenceResult) error {
    
    outputShape := outputs[0].Output.Shape
    
    // Segmentation: [C, H, W]
    if len(outputShape) == 3 {
        return p.processSegmentation(ctx, outputs, metadata, options, result)
    }
    
    // Detection: [N, 6]
    if len(outputShape) == 2 && outputShape[1] == 6 {
        return p.processDetection(ctx, outputs, metadata, options, result)
    }
    
    return fmt.Errorf("unknown output format")
}

// processSegmentation merges tile outputs with overlap blending
func (p *Postprocessor) processSegmentation(ctx context.Context, outputs []*TileOutput,
    metadata *ImageMetadata, options *InferenceOptions, result *InferenceResult) error {
    
    numClasses := int(outputs[0].Output.Shape[0])
    fullWidth := metadata.Width
    fullHeight := metadata.Height
    
    // Accumulate probabilities with counts for overlap blending
    probs := make([][]float32, numClasses)
    counts := make([]float32, fullWidth*fullHeight)
    for c := 0; c < numClasses; c++ {
        probs[c] = make([]float32, fullWidth*fullHeight)
    }
    
    // Merge tiles
    for _, out := range outputs {
        tile := out.Tile
        data := out.Output.Float32Data
        pixelsPerClass := tile.Width * tile.Height
        
        for row := 0; row < tile.Height; row++ {
            for col := 0; col < tile.Width; col++ {
                gx := tile.X + col
                gy := tile.Y + row
                if gx >= fullWidth || gy >= fullHeight {
                    continue
                }
                
                globalIdx := gy*fullWidth + gx
                tileIdx := row*tile.Width + col
                
                for c := 0; c < numClasses; c++ {
                    probs[c][globalIdx] += data[c*pixelsPerClass+tileIdx]
                }
                counts[globalIdx]++
            }
        }
    }
    
    // Average and get argmax
    mask := make([]int, fullWidth*fullHeight)
    for i := 0; i < fullWidth*fullHeight; i++ {
        if counts[i] == 0 {
            continue
        }
        
        maxProb := float32(-1)
        maxClass := 0
        for c := 0; c < numClasses; c++ {
            avg := probs[c][i] / counts[i]
            if avg > maxProb {
                maxProb = avg
                maxClass = c
            }
        }
        mask[i] = maxClass
    }
    
    result.Mask = &RasterOutput{
        Width: fullWidth, Height: fullHeight,
        Data: mask, NumClasses: numClasses,
    }
    
    return nil
}

// processDetection applies NMS and converts to geo coordinates
func (p *Postprocessor) processDetection(ctx context.Context, outputs []*TileOutput,
    metadata *ImageMetadata, options *InferenceOptions, result *InferenceResult) error {
    
    threshold := float32(0.5)
    if options != nil && options.Threshold > 0 {
        threshold = float32(options.Threshold)
    }
    
    var allDetections []Detection
    
    for _, out := range outputs {
        tile := out.Tile
        data := out.Output.Float32Data
        numDetections := int(out.Output.Shape[0])
        
        for i := 0; i < numDetections; i++ {
            offset := i * 6
            confidence := data[offset+4]
            if confidence < threshold {
                continue
            }
            
            // Convert to geo coordinates
            x1 := float64(tile.X) + float64(data[offset+0])
            y1 := float64(tile.Y) + float64(data[offset+1])
            x2 := float64(tile.X) + float64(data[offset+2])
            y2 := float64(tile.Y) + float64(data[offset+3])
            
            geoX1, geoY1 := p.pixelToGeo(x1, y1, metadata.GeoTransform)
            geoX2, geoY2 := p.pixelToGeo(x2, y2, metadata.GeoTransform)
            
            allDetections = append(allDetections, Detection{
                BBox:       []float64{geoX1, geoY1, geoX2, geoY2},
                Confidence: float64(confidence),
                ClassID:    int(data[offset+5]),
            })
        }
    }
    
    // Apply NMS
    result.Detections = p.applyNMS(allDetections, 0.5)
    return nil
}
```

```go
    id       int
    pipeline *InferencePipeline
    jobChan  chan *BatchJob
}

// processJob processes a single batch job
func (w *BatchWorker) processJob(ctx context.Context, job *BatchJob) {
    job.Status = JobStatusRunning
    
    // Get input items
    items, _ := w.resolveInputItems(ctx, job)
    job.Progress.TotalItems = len(items)
    
    results := make([]*InferenceResult, 0)
    
    for i, item := range items {
        result, err := w.processItem(ctx, job, item)
        if err != nil {
            job.Progress.FailedItems++
        } else {
            results = append(results, result)
        }
        
        job.Progress.ProcessedItems = i + 1
        
        // Estimate remaining time
        elapsed := time.Since(*job.StartedAt).Seconds()
        avgTime := elapsed / float64(i+1)
        job.Progress.EstimatedTime = float64(len(items)-i-1) * avgTime
    }
    
    // Write outputs
    w.writeOutputs(ctx, job, results)
    
    job.Status = JobStatusCompleted
}
```

```go
// realtime_service.go

// RealtimeInferenceService handles synchronous inference
type RealtimeInferenceService struct {
    registry       *ModelRegistry
    router         *ModelRouter
    pipeline       *InferencePipeline
    rateLimiter    *RateLimiter
    circuitBreaker *CircuitBreaker
    sem            *semaphore.Weighted
    config         *RealtimeConfig
}

// Infer performs synchronous inference
func (s *RealtimeInferenceService) Infer(ctx context.Context, req *InferenceRequest) (*InferenceResult, error) {
    // Rate limiting
    if !s.rateLimiter.Allow() {
        return nil, ErrRateLimited
    }
    
    // Circuit breaker check
    if !s.circuitBreaker.Allow() {
        return nil, ErrCircuitOpen
    }
    
    // Concurrency control
    ctx, cancel := context.WithTimeout(ctx, s.config.RequestTimeout)
    defer cancel()
    
    if err := s.sem.Acquire(ctx, 1); err != nil {
        return nil, ErrTimeout
    }
    defer s.sem.Release(1)
    
    // Select model version (traffic splitting)
    modelVersion, err := s.selectModelVersion(ctx, req.ModelID, req.ModelVersion)
    if err != nil {
        return nil, err
    }
    req.ModelVersion = modelVersion
    
    // Execute inference
    result, err := s.pipeline.Execute(ctx, req)
    
    if err != nil {
        s.circuitBreaker.RecordFailure()
        return nil, err
    }
    s.circuitBreaker.RecordSuccess()
    
    return result, nil
}

// selectModelVersion selects based on traffic split
func (s *RealtimeInferenceService) selectModelVersion(ctx context.Context, modelID, requestedVersion string) (string, error) {
    if requestedVersion != "" {
        return requestedVersion, nil
    }
    
    versions, _ := s.registry.GetVersionsForTrafficSplit(ctx, modelID)
    if len(versions) == 0 {
        return "", ErrNoActiveVersion
    }
    
    if len(versions) == 1 {
        return versions[0].Version, nil
    }
    
    // Weighted random selection
    r := rand.Float64()
    cumulative := 0.0
    for _, v := range versions {
        cumulative += v.TrafficWeight
        if r <= cumulative {
            return v.Version, nil
        }
    }
    
    return versions[0].Version, nil
}

// CircuitBreaker implements circuit breaker pattern
type CircuitBreaker struct {
    failures   int
    threshold  int
    resetTime  time.Duration
    state      string // closed, open, half-open
    lastFail   time.Time
    mu         sync.RWMutex
}

func (cb *CircuitBreaker) Allow() bool {
    cb.mu.RLock()
    defer cb.mu.RUnlock()
    
    if cb.state == "closed" {
        return true
    }
    if cb.state == "open" && time.Since(cb.lastFail) > cb.resetTime {
        cb.state = "half-open"
        return true
    }
    return cb.state == "half-open"
}
```

```go
// GPUDevice represents a GPU

    UUID        string
    Name        string
    Memory      uint64
    FreeMemory  uint64
    Utilization int
    Temperature int
    Models      []string
}

// AllocateGPU allocates a GPU for a model
func (gm *GPUManager) AllocateGPU(modelID string, memoryRequired uint64) (int, error) {
    // Find GPU with lowest utilization and enough memory
    var bestDevice *GPUDevice
    bestScore := float64(-1)
    
    for _, device := range gm.devices {
        if device.FreeMemory < memoryRequired {
            continue
        }
        
        memoryScore := float64(device.FreeMemory) / float64(device.Memory)
        utilizationScore := 1.0 - float64(device.Utilization)/100.0
        score := memoryScore*0.6 + utilizationScore*0.4
        
        if score > bestScore {
            bestScore = score
            bestDevice = device
        }
    }
    
    if bestDevice == nil {
        return -1, fmt.Errorf("no GPU available")
    }
    
    bestDevice.FreeMemory -= memoryRequired
    bestDevice.Models = append(bestDevice.Models, modelID)
    
    return bestDevice.Index, nil
}

// GPUMonitor collects GPU metrics
func (m *GPUMonitor) collectMetrics() {
    for i := range m.devices {
        handle, _ := nvml.DeviceGetHandleByIndex(i)
        
        memory, _ := handle.GetMemoryInfo()

        temp, _ := handle.GetTemperature(nvml.TEMPERATURE_GPU)
```

```go
    outputPath := filepath.Join(o.workDir, "model.plan")

        args = append(args, "--fp16")
    } else if o.config.TargetPrecision == "INT8" {
        args = append(args, "--int8")
    }
    
    cmd := exec.CommandContext(ctx, "trtexec", args...)
```

```go
// deployment_manager.go

type DeploymentManager struct {
    registry       *ModelRegistry
    servingCluster *ServingCluster
    validator      *ModelValidator
}

type DeploymentRequest struct {
    ModelID        string
    Version        string
    Strategy       DeploymentStrategy  // recreate, canary, blue_green
    InitialWeight  float64
    RampUpInterval time.Duration
    RampUpStep     float64
    AutoRollback   bool
    ErrorThreshold float64
}

// Deploy deploys a model version
func (dm *DeploymentManager) Deploy(ctx context.Context, req *DeploymentRequest) (*Deployment, error) {
    // Validate model version
    version, err := dm.registry.GetVersion(ctx, req.ModelID, req.Version)
    if err != nil {
        return nil, err
    }
    
    // Pre-deployment validation
    if err := dm.validator.Validate(ctx, version); err != nil {
        return nil, fmt.Errorf("validation failed: %w", err)
    }
    
    deployment := &Deployment{
        ID: generateID(),
        ModelID: req.ModelID,
        Version: req.Version,
        Strategy: req.Strategy,
        Status: "in_progress",
    }
    
    switch req.Strategy {
    case StrategyCanary:
        return dm.deployCanary(ctx, deployment, req)
    case StrategyBlueGreen:
        return dm.deployBlueGreen(ctx, deployment, req)
    default:
        return dm.deployRecreate(ctx, deployment, req)
    }
}

// deployCanary implements canary deployment with gradual ramp-up
func (dm *DeploymentManager) deployCanary(ctx context.Context, deployment *Deployment, req *DeploymentRequest) (*Deployment, error) {
    // Start with initial weight
    err := dm.registry.UpdateTrafficSplit(ctx, req.ModelID, map[string]float64{
        "production": 1 - req.InitialWeight,
        req.Version:  req.InitialWeight,
    })
    if err != nil {
        return nil, err
    }
    
    // Gradual ramp-up
    currentWeight := req.InitialWeight
    ticker := time.NewTicker(req.RampUpInterval)
    defer ticker.Stop()
    
    for currentWeight < 1.0 {
        select {
        case <-ctx.Done():
            return deployment, ctx.Err()
        case <-ticker.C:
            // Check metrics
            if req.AutoRollback {
                canaryMetrics := dm.getCanaryMetrics(ctx, req.ModelID, req.Version)
                if canaryMetrics.ErrorRate > req.ErrorThreshold {
                    return dm.rollback(ctx, deployment)
                }
            }
            
            // Increase weight
            currentWeight += req.RampUpStep
            if currentWeight > 1.0 {
                currentWeight = 1.0
            }
            
            dm.registry.UpdateTrafficSplit(ctx, req.ModelID, map[string]float64{
                "production": 1 - currentWeight,
                req.Version:  currentWeight,
            })
        }
    }
    
    // Promote to production
    dm.registry.PromoteVersion(ctx, req.Version, VersionStatusProduction)
    deployment.Status = "completed"
    
    return deployment, nil
}
```

```go
type AutoScaler struct {

type AutoScaleConfig struct {
    MinReplicas     int
    MaxReplicas     int
    TargetGPUUtil   float64  // 80%
    TargetLatencyMs float64  // 100ms
    ScaleUpCooldown time.Duration
    ScaleDownCooldown time.Duration
}

// Evaluate evaluates scaling needs
func (as *AutoScaler) Evaluate(ctx context.Context, modelID string) (*ScaleDecision, error) {
    currentReplicas := as.cluster.GetReplicaCount(modelID)
    
    // Get metrics
    gpuUtil := as.metrics.GetGPUUtilization(modelID)
    latencyP95 := as.metrics.GetLatencyP95(modelID)
    queueDepth := as.metrics.GetRequestQueueDepth(modelID)
    
    decision := &ScaleDecision{
        ModelID:  modelID,
        Current:  currentReplicas,
        Desired:  currentReplicas,
    }
    
    // Scale up conditions
    if gpuUtil > as.config.TargetGPUUtil*1.2 || 
       latencyP95 > as.config.TargetLatencyMs*1.5 ||
       queueDepth > 100 {
        decision.Desired = min(currentReplicas+1, as.config.MaxReplicas)
        decision.Reason = "high_load"
    }
    
    // Scale down conditions
    if gpuUtil < as.config.TargetGPUUtil*0.5 &&
       latencyP95 < as.config.TargetLatencyMs*0.5 &&
       queueDepth < 10 {
        decision.Desired = max(currentReplicas-1, as.config.MinReplicas)
        decision.Reason = "low_load"
    }
    
    return decision, nil
}
```

```sql
-- Update timestamp trigger
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_models_updated
    BEFORE UPDATE ON ml_models
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER trg_versions_updated
    BEFORE UPDATE ON ml_model_versions
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

-- Version history tracking
CREATE OR REPLACE FUNCTION log_version_change()
RETURNS TRIGGER AS $$
BEGIN
    IF OLD.status != NEW.status OR OLD.traffic_weight != NEW.traffic_weight THEN
        INSERT INTO ml_version_history (
            version_id, old_status, new_status, 
            old_weight, new_weight, changed_at
        ) VALUES (
            NEW.id, OLD.status, NEW.status,
            OLD.traffic_weight, NEW.traffic_weight, NOW()
        );
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_version_history
    AFTER UPDATE ON ml_model_versions
    FOR EACH ROW EXECUTE FUNCTION log_version_change();

-- Update job progress
CREATE OR REPLACE FUNCTION update_job_progress()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE ml_batch_jobs 
    SET 
        processed_items = (
            SELECT COUNT(*) FROM ml_job_items 
            WHERE job_id = NEW.job_id AND status = 'completed'
        ),
        failed_items = (
            SELECT COUNT(*) FROM ml_job_items 
            WHERE job_id = NEW.job_id AND status = 'failed'
        )
    WHERE id = NEW.job_id;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_job_progress
    AFTER UPDATE OF status ON ml_job_items
    FOR EACH ROW EXECUTE FUNCTION update_job_progress();

-- ============================================
-- VIEWS
-- ============================================

-- Active models with production version
CREATE VIEW ml_active_models AS
SELECT 
    m.id,
    m.name,
    m.type,
    m.framework,
    v.version as production_version,
    v.deployed_at,
    m.tenant_id
FROM ml_models m
JOIN ml_model_versions v ON m.id = v.model_id AND v.status = 'production';

-- Model performance summary
CREATE VIEW ml_model_performance AS
SELECT 
    model_id,
    model_version,
    SUM(request_count) as total_requests,
    SUM(success_count) as total_success,
    SUM(error_count) as total_errors,
    AVG(latency_p95_ms) as avg_p95_latency,
    MAX(latency_p99_ms) as max_p99_latency,
    SUM(tiles_processed) as total_tiles
FROM ml_inference_metrics
WHERE time > NOW() - INTERVAL '24 hours'
GROUP BY model_id, model_version;
```

## Appendix F: Model Validation Framework

### F.1 Validation Implementation

```go
// validator.go

package mlserving

import (
    "context"
    "fmt"
)

// ModelValidator validates models before deployment
type ModelValidator struct {
    registry     *ModelRegistry
    testDataPath string
    thresholds   *ValidationThresholds
}

// ValidationThresholds defines validation criteria
type ValidationThresholds struct {
    MinAccuracy       float64
    MaxLatencyMs      float64
    MinThroughput     float64
    MaxMemoryMB       int64
    MaxModelSizeMB    int64
}

// ValidationResult contains validation results
type ValidationResult struct {
    Valid           bool                   `json:"valid"`
    ModelID         string                 `json:"model_id"`
    Version         string                 `json:"version"`
    Checks          []ValidationCheck      `json:"checks"`
    Metrics         map[string]float64     `json:"metrics"`
    Recommendations []string               `json:"recommendations"`
    Duration        float64                `json:"duration_seconds"`
}

// ValidationCheck represents a single check
type ValidationCheck struct {
    Name     string  `json:"name"`
    Passed   bool    `json:"passed"`
    Expected string  `json:"expected"`
    Actual   string  `json:"actual"`
    Message  string  `json:"message"`
}

// Validate performs comprehensive model validation
func (v *ModelValidator) Validate(ctx context.Context, version *ModelVersion) (*ValidationResult, error) {
    result := &ValidationResult{
        Valid:   true,
        ModelID: version.ModelID,
        Version: version.Version,
        Checks:  make([]ValidationCheck, 0),
        Metrics: make(map[string]float64),
    }
    
    // 1. Artifact validation
    artifactCheck := v.validateArtifact(ctx, version)
    result.Checks = append(result.Checks, artifactCheck)
    if !artifactCheck.Passed {
        result.Valid = false
    }
    
    // 2. Schema validation
    schemaCheck := v.validateSchema(ctx, version)
    result.Checks = append(result.Checks, schemaCheck)
    if !schemaCheck.Passed {
        result.Valid = false
    }
    
    // 3. Inference test
    inferenceCheck, metrics := v.validateInference(ctx, version)
    result.Checks = append(result.Checks, inferenceCheck)
    if !inferenceCheck.Passed {
        result.Valid = false
    }
    for k, val := range metrics {
        result.Metrics[k] = val
    }
    
    // 4. Performance validation
    perfCheck := v.validatePerformance(ctx, version, metrics)
    result.Checks = append(result.Checks, perfCheck)
    if !perfCheck.Passed {
        result.Valid = false
    }
    
    // 5. Memory validation
    memCheck := v.validateMemory(ctx, version)
    result.Checks = append(result.Checks, memCheck)
    if !memCheck.Passed {
        result.Valid = false
    }
    
    // 6. Accuracy validation (if test data available)
    if v.testDataPath != "" {
        accCheck := v.validateAccuracy(ctx, version)
        result.Checks = append(result.Checks, accCheck)
        if !accCheck.Passed {
            result.Valid = false
        }
    }
    
    // Generate recommendations
    result.Recommendations = v.generateRecommendations(result)
    
    return result, nil
}

// validateArtifact checks model artifact integrity
func (v *ModelValidator) validateArtifact(ctx context.Context, version *ModelVersion) ValidationCheck {
    check := ValidationCheck{
        Name: "artifact_integrity",
    }
    
    // Download and verify hash
    artifact, err := v.downloadArtifact(ctx, version.ArtifactPath)
    if err != nil {
        check.Passed = false
        check.Message = fmt.Sprintf("Failed to download artifact: %v", err)
        return check
    }
    
    hash := computeSHA256(artifact)
    if hash != version.ArtifactHash {
        check.Passed = false
        check.Expected = version.ArtifactHash
        check.Actual = hash
        check.Message = "Artifact hash mismatch"
        return check
    }
    
    check.Passed = true
    check.Message = "Artifact integrity verified"
    return check
}

// validateSchema checks input/output schema
func (v *ModelValidator) validateSchema(ctx context.Context, version *ModelVersion) ValidationCheck {
    check := ValidationCheck{
        Name: "schema_validation",
    }
    
    model, _ := v.registry.GetModel(ctx, version.ModelID)
    
    // Load model and check input/output shapes
    client, err := v.createTestClient(ctx, version)
    if err != nil {
        check.Passed = false
        check.Message = fmt.Sprintf("Failed to load model: %v", err)
        return check
    }
    defer client.Close()
    
    // Create dummy input matching expected schema
    dummyInput := v.createDummyInput(model.InputSpec)
    
    // Run inference
    outputs, err := client.Infer(ctx, []*Tensor{dummyInput})
    if err != nil {
        check.Passed = false
        check.Message = fmt.Sprintf("Inference failed: %v", err)
        return check
    }
    
    // Validate output shape
    if !v.validateOutputShape(outputs[0], model.OutputSpec) {
        check.Passed = false
        check.Expected = fmt.Sprintf("%v", model.OutputSpec.Shape)
        check.Actual = fmt.Sprintf("%v", outputs[0].Shape)
        check.Message = "Output shape mismatch"
        return check
    }
    
    check.Passed = true
    check.Message = "Schema validation passed"
    return check
}

// validateInference runs inference tests
func (v *ModelValidator) validateInference(ctx context.Context, version *ModelVersion) (ValidationCheck, map[string]float64) {
    check := ValidationCheck{
        Name: "inference_test",
    }
    metrics := make(map[string]float64)
    
    client, _ := v.createTestClient(ctx, version)
    defer client.Close()
    
    model, _ := v.registry.GetModel(ctx, version.ModelID)
    
    // Run multiple inference iterations
    iterations := 100
    var totalLatency float64
    var minLatency, maxLatency float64 = 1e9, 0
    
    for i := 0; i < iterations; i++ {
        input := v.createDummyInput(model.InputSpec)
        
        start := time.Now()
        _, err := client.Infer(ctx, []*Tensor{input})
        latency := time.Since(start).Seconds() * 1000
        
        if err != nil {
            check.Passed = false
            check.Message = fmt.Sprintf("Inference failed at iteration %d: %v", i, err)
            return check, metrics
        }
        
        totalLatency += latency
        if latency < minLatency {
            minLatency = latency
        }
        if latency > maxLatency {
            maxLatency = latency
        }
    }
    
    avgLatency := totalLatency / float64(iterations)
    
    metrics["latency_avg_ms"] = avgLatency
    metrics["latency_min_ms"] = minLatency
    metrics["latency_max_ms"] = maxLatency
    metrics["throughput_per_sec"] = 1000.0 / avgLatency
    
    check.Passed = true
    check.Message = fmt.Sprintf("Inference test passed (avg latency: %.2fms)", avgLatency)
    return check, metrics
}

// validatePerformance checks against performance thresholds
func (v *ModelValidator) validatePerformance(ctx context.Context, version *ModelVersion, metrics map[string]float64) ValidationCheck {
    check := ValidationCheck{
        Name: "performance_validation",
    }
    
    avgLatency := metrics["latency_avg_ms"]
    
    if avgLatency > v.thresholds.MaxLatencyMs {
        check.Passed = false
        check.Expected = fmt.Sprintf("<= %.2fms", v.thresholds.MaxLatencyMs)
        check.Actual = fmt.Sprintf("%.2fms", avgLatency)
        check.Message = "Latency exceeds threshold"
        return check
    }
    
    throughput := metrics["throughput_per_sec"]
    if throughput < v.thresholds.MinThroughput {
        check.Passed = false
        check.Expected = fmt.Sprintf(">= %.2f/sec", v.thresholds.MinThroughput)
        check.Actual = fmt.Sprintf("%.2f/sec", throughput)
        check.Message = "Throughput below threshold"
        return check
    }
    
    check.Passed = true
    check.Message = "Performance within thresholds"
    return check
}

// validateMemory checks GPU memory usage
func (v *ModelValidator) validateMemory(ctx context.Context, version *ModelVersion) ValidationCheck {
    check := ValidationCheck{
        Name: "memory_validation",
    }
    
    // Get memory usage from loaded model
    memUsed := v.getModelMemoryUsage(ctx, version)
    memMB := memUsed / (1024 * 1024)
    
    if memMB > v.thresholds.MaxMemoryMB {
        check.Passed = false
        check.Expected = fmt.Sprintf("<= %d MB", v.thresholds.MaxMemoryMB)
        check.Actual = fmt.Sprintf("%d MB", memMB)
        check.Message = "Memory usage exceeds threshold"
        return check
    }
    
    check.Passed = true
    check.Message = fmt.Sprintf("Memory usage: %d MB", memMB)
    return check
}

// validateAccuracy runs accuracy tests
func (v *ModelValidator) validateAccuracy(ctx context.Context, version *ModelVersion) ValidationCheck {
    check := ValidationCheck{
        Name: "accuracy_validation",
    }
    
    // Load test dataset
    testData, labels, _ := v.loadTestData(ctx)
    
    client, _ := v.createTestClient(ctx, version)
    defer client.Close()
    
    correct := 0
    total := len(testData)
    
    for i, input := range testData {
        outputs, _ := client.Infer(ctx, []*Tensor{input})
        prediction := v.getPrediction(outputs[0])
        
        if prediction == labels[i] {
            correct++
        }
    }
    
    accuracy := float64(correct) / float64(total)
    
    if accuracy < v.thresholds.MinAccuracy {
        check.Passed = false
        check.Expected = fmt.Sprintf(">= %.2f%%", v.thresholds.MinAccuracy*100)
        check.Actual = fmt.Sprintf("%.2f%%", accuracy*100)
        check.Message = "Accuracy below threshold"
        return check
    }
    
    check.Passed = true
    check.Message = fmt.Sprintf("Accuracy: %.2f%%", accuracy*100)
    return check
}

// generateRecommendations generates optimization recommendations
func (v *ModelValidator) generateRecommendations(result *ValidationResult) []string {
    recommendations := make([]string, 0)
    
    if latency, ok := result.Metrics["latency_avg_ms"]; ok && latency > 50 {
        recommendations = append(recommendations, 
            "Consider TensorRT optimization for lower latency")
    }
    
    if throughput, ok := result.Metrics["throughput_per_sec"]; ok && throughput < 20 {
        recommendations = append(recommendations,
            "Consider increasing batch size for better throughput")
    }
    
    return recommendations
}
```
