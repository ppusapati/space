## 1. Executive Summary

### 1.1 Purpose

The Command Execution & Verification system provides secure, reliable satellite commanding with full traceability:

- **CCSDS Compliance:** TC Space Data Link and Packet protocols
- **Multi-Level Verification:** Pre-uplink, uplink confirmation, execution verification
- **Secure Commanding:** Encryption, authentication, and authorization
- **Intelligent Retry:** Automatic retry with configurable policies
- **Real-Time Tracking:** Live command status and telemetry correlation
- **Full Auditability:** Complete command history and compliance logging

### 1.2 Key Metrics

| Metric | Target | Description |
|--------|--------|-------------|
| **Command Latency** | < 500ms | Ground processing time |
| **Uplink Success Rate** | > 99.5% | First-attempt delivery |
| **Verification Time** | < 30s | From uplink to verification |
| **Queue Throughput** | 1000 cmd/min | Peak processing capacity |
| **Retry Success** | > 95% | After automatic retry |
| **Availability** | 99.99% | System uptime |

### 1.3 Command Categories

| Category | Priority | Examples | Verification |
|----------|----------|----------|--------------|
| **Critical** | P0 | Emergency safe mode, collision avoidance | Immediate telemetry |
| **Time-Critical** | P1 | Maneuver execution, imaging tasking | Execution confirmation |
| **Routine** | P2 | Payload config, housekeeping | State verification |
| **Deferred** | P3 | Software updates, parameter changes | Scheduled verification |

### 3.2 State Machine Implementation

```go
// state_machine.go

package command

import (
    "context"
    "fmt"
    "sync"
    "time"
)

// CommandState represents command states
type CommandState string

const (
    StateCreated       CommandState = "created"
    StateValidating    CommandState = "validating"
    StateRejected      CommandState = "rejected"
    StateUnauthorized  CommandState = "unauthorized"
    StateQueued        CommandState = "queued"
    StateScheduled     CommandState = "scheduled"
    StateConflicted    CommandState = "conflicted"
    StateCancelled     CommandState = "cancelled"
    StateUplinking     CommandState = "uplinking"
    StateSent          CommandState = "sent"
    StateUplinkTimeout CommandState = "uplink_timeout"
    StateUplinkFailed  CommandState = "uplink_failed"
    StateAcked         CommandState = "acked"
    StateVerifying     CommandState = "verifying"
    StateCompleted     CommandState = "completed"
    StateExecFailed    CommandState = "exec_failed"
    StateVerifyTimeout CommandState = "verify_timeout"
    StateRetrying      CommandState = "retrying"
)

// CommandEvent represents state transition events
type CommandEvent string

const (
    EventSubmit       CommandEvent = "submit"
    EventValid        CommandEvent = "valid"
    EventInvalid      CommandEvent = "invalid"
    EventUnauthorized CommandEvent = "unauthorized"
    EventSchedule     CommandEvent = "schedule"
    EventConflict     CommandEvent = "conflict"
    EventCancel       CommandEvent = "cancel"
    EventUplink       CommandEvent = "uplink"
    EventConfirm      CommandEvent = "confirm"
    EventTimeout      CommandEvent = "timeout"
    EventFail         CommandEvent = "fail"
    EventAck          CommandEvent = "ack"
    EventVerify       CommandEvent = "verify"
    EventSuccess      CommandEvent = "success"
    EventRetry        CommandEvent = "retry"
)

// Transition defines a state transition
type Transition struct {
    From   CommandState
    Event  CommandEvent
    To     CommandState
    Action func(*Command) error
}

// CommandStateMachine manages command state transitions
type CommandStateMachine struct {
    transitions map[CommandState]map[CommandEvent]*Transition
    mu          sync.RWMutex
    eventBus    EventBus
    store       CommandStore
}

// NewCommandStateMachine creates a new state machine
func NewCommandStateMachine(eventBus EventBus, store CommandStore) *CommandStateMachine {
    sm := &CommandStateMachine{
        transitions: make(map[CommandState]map[CommandEvent]*Transition),
        eventBus:    eventBus,
        store:       store,
    }
    sm.registerTransitions()
    return sm
}

// registerTransitions registers all valid transitions
func (sm *CommandStateMachine) registerTransitions() {
    transitions := []Transition{
        // Created -> Validating
        {StateCreated, EventSubmit, StateValidating, nil},
        
        // Validating -> Queued/Rejected/Unauthorized
        {StateValidating, EventValid, StateQueued, nil},
        {StateValidating, EventInvalid, StateRejected, nil},
        {StateValidating, EventUnauthorized, StateUnauthorized, nil},
        
        // Queued -> Scheduled/Cancelled
        {StateQueued, EventSchedule, StateScheduled, nil},
        {StateQueued, EventCancel, StateCancelled, nil},
        
        // Scheduled -> Uplinking/Conflicted/Cancelled
        {StateScheduled, EventUplink, StateUplinking, nil},
        {StateScheduled, EventConflict, StateConflicted, nil},
        {StateScheduled, EventCancel, StateCancelled, nil},
        
        // Uplinking -> Sent/UplinkTimeout/UplinkFailed
        {StateUplinking, EventConfirm, StateSent, nil},
        {StateUplinking, EventTimeout, StateUplinkTimeout, nil},
        {StateUplinking, EventFail, StateUplinkFailed, nil},
        
        // Sent -> Acked
        {StateSent, EventAck, StateAcked, nil},
        {StateSent, EventTimeout, StateUplinkTimeout, nil},
        
        // Acked -> Verifying
        {StateAcked, EventVerify, StateVerifying, nil},
        
        // Verifying -> Completed/ExecFailed/VerifyTimeout
        {StateVerifying, EventSuccess, StateCompleted, nil},
        {StateVerifying, EventFail, StateExecFailed, nil},
        {StateVerifying, EventTimeout, StateVerifyTimeout, nil},
        
        // Retry transitions
        {StateUplinkTimeout, EventRetry, StateRetrying, nil},
        {StateUplinkFailed, EventRetry, StateRetrying, nil},
        {StateVerifyTimeout, EventRetry, StateRetrying, nil},
        {StateExecFailed, EventRetry, StateRetrying, nil},
        {StateRetrying, EventSchedule, StateQueued, nil},
    }
    
    for _, t := range transitions {
        if sm.transitions[t.From] == nil {
            sm.transitions[t.From] = make(map[CommandEvent]*Transition)
        }
        trans := t
        sm.transitions[t.From][t.Event] = &trans
    }
}

// ProcessEvent processes a state transition event
func (sm *CommandStateMachine) ProcessEvent(ctx context.Context, cmd *Command, event CommandEvent) error {
    sm.mu.Lock()
    defer sm.mu.Unlock()
    
    currentState := cmd.State
    
    // Find valid transition
    stateTransitions, ok := sm.transitions[currentState]
    if !ok {
        return fmt.Errorf("no transitions from state %s", currentState)
    }
    
    transition, ok := stateTransitions[event]
    if !ok {
        return fmt.Errorf("invalid event %s for state %s", event, currentState)
    }
    
    // Execute action if defined
    if transition.Action != nil {
        if err := transition.Action(cmd); err != nil {
            return fmt.Errorf("transition action failed: %w", err)
        }
    }
    
    // Update state
    previousState := cmd.State
    cmd.State = transition.To
    cmd.UpdatedAt = time.Now()
    
    // Record state history
    cmd.StateHistory = append(cmd.StateHistory, StateHistoryEntry{
        FromState:  previousState,
        ToState:    transition.To,
        Event:      event,
        Timestamp:  time.Now(),
        Reason:     cmd.LastReason,
    })
    
    // Persist state change
    if err := sm.store.UpdateState(ctx, cmd); err != nil {
        return fmt.Errorf("failed to persist state: %w", err)
    }
    
    // Emit state change event
    sm.eventBus.Publish(ctx, &CommandStateChangedEvent{
        CommandID:     cmd.ID,
        PreviousState: previousState,
        CurrentState:  cmd.State,
        Event:         event,
        Timestamp:     time.Now(),
    })
    
    return nil
}

// CanTransition checks if a transition is valid
func (sm *CommandStateMachine) CanTransition(state CommandState, event CommandEvent) bool {
    sm.mu.RLock()
    defer sm.mu.RUnlock()
    
    stateTransitions, ok := sm.transitions[state]
    if !ok {
        return false
    }
    
    _, ok = stateTransitions[event]
    return ok
}

// GetValidEvents returns valid events for a state
func (sm *CommandStateMachine) GetValidEvents(state CommandState) []CommandEvent {
    sm.mu.RLock()
    defer sm.mu.RUnlock()
    
    events := make([]CommandEvent, 0)
    if stateTransitions, ok := sm.transitions[state]; ok {
        for event := range stateTransitions {
            events = append(events, event)
        }
    }
    return events
}

// StateHistoryEntry records a state transition
type StateHistoryEntry struct {
    FromState  CommandState
    ToState    CommandState
    Event      CommandEvent
    Timestamp  time.Time
    Reason     string
    ActorID    string
}

// CommandStateChangedEvent is emitted on state change
type CommandStateChangedEvent struct {
    CommandID     string
    PreviousState CommandState
    CurrentState  CommandState
    Event         CommandEvent
    Timestamp     time.Time
}
```

```go
// CommandType defines command types

    CmdTypeTimeTagged   CommandType = "time_tagged"
    CmdTypeSequence     CommandType = "sequence"

// NewCommandManager creates a new command manager
func NewCommandManager(config *CommandConfig, deps *CommandDependencies) *CommandManager {
    return &CommandManager{
        stateMachine:  deps.StateMachine,
        validator:     deps.Validator,
        encoder:       deps.Encoder,
        scheduler:     deps.Scheduler,
        uplink:        deps.Uplink,
        verification:  deps.Verification,
        store:         deps.Store,
        eventBus:      deps.EventBus,
        config:        config,
    }
}

// SubmitCommand submits a new command
func (cm *CommandManager) SubmitCommand(ctx context.Context, req *SubmitCommandRequest) (*Command, error) {
    // Create command
    cmd := &Command{
        ID:           generateCommandID(),
        SpacecraftID: req.SpacecraftID,
        Type:         req.Type,
        Priority:     req.Priority,
        State:        StateCreated,
        Parameters:   req.Parameters,
        MaxRetries:   cm.config.MaxRetries,
        CreatedBy:    req.UserID,
        Tags:         req.Tags,
        Notes:        req.Notes,
        CreatedAt:    time.Now(),
        UpdatedAt:    time.Now(),
        TenantID:     req.TenantID,
    }
    
    if cmd.Priority == 0 {
        cmd.Priority = cm.config.DefaultPriority
    }
    
    // Set expected state for verification
    if req.ExpectedState != nil {
        cmd.ExpectedState = req.ExpectedState
        cmd.VerificationMethod = req.VerificationMethod
    }
    
    // Store command
    if err := cm.store.Create(ctx, cmd); err != nil {
        return nil, fmt.Errorf("failed to store command: %w", err)
    }
    
    // Transition to validating
    if err := cm.stateMachine.ProcessEvent(ctx, cmd, EventSubmit); err != nil {
        return nil, fmt.Errorf("failed to submit: %w", err)
    }
    
    // Start validation asynchronously
    go cm.validateCommand(context.Background(), cmd)
    
    return cmd, nil
}

// validateCommand validates the command
func (cm *CommandManager) validateCommand(ctx context.Context, cmd *Command) {
    // Validate command
    result, err := cm.validator.Validate(ctx, cmd)
    if err != nil {
        cmd.LastError = err.Error()
        cm.stateMachine.ProcessEvent(ctx, cmd, EventInvalid)
        return
    }
    
    if !result.Valid {
        cmd.LastReason = result.Reason
        if result.Unauthorized {
            cm.stateMachine.ProcessEvent(ctx, cmd, EventUnauthorized)
        } else {
            cm.stateMachine.ProcessEvent(ctx, cmd, EventInvalid)
        }
        return
    }
    
    // Encode command
    encoded, err := cm.encoder.Encode(ctx, cmd)
    if err != nil {
        cmd.LastError = err.Error()
        cm.stateMachine.ProcessEvent(ctx, cmd, EventInvalid)
        return
    }
    cmd.EncodedData = encoded
    
    // Create frame
    frame, err := cm.encoder.Frame(ctx, cmd)
    if err != nil {
        cmd.LastError = err.Error()
        cm.stateMachine.ProcessEvent(ctx, cmd, EventInvalid)
        return
    }
    cmd.FrameData = frame
    
    // Transition to queued
    cm.stateMachine.ProcessEvent(ctx, cmd, EventValid)
    
    // Schedule if realtime
    if cmd.Type == CmdTypeRealtime {
        cm.scheduleCommand(ctx, cmd)
    }
}

// scheduleCommand schedules the command for uplink
func (cm *CommandManager) scheduleCommand(ctx context.Context, cmd *Command) {
    // Check for conflicts
    conflicts, err := cm.scheduler.CheckConflicts(ctx, cmd)
    if err != nil || len(conflicts) > 0 {
        cmd.LastReason = "scheduling conflict detected"
        cm.stateMachine.ProcessEvent(ctx, cmd, EventConflict)
        return
    }
    
    // Find next pass
    pass, err := cm.scheduler.FindNextPass(ctx, cmd.SpacecraftID)
    if err != nil {
        cmd.LastError = err.Error()
        return
    }
    
    cmd.PassID = pass.ID
    cmd.SequenceNumber = cm.scheduler.GetNextSequenceNumber(cmd.SpacecraftID)
    
    // Schedule
    if err := cm.scheduler.Schedule(ctx, cmd, pass); err != nil {
        cmd.LastError = err.Error()
        return
    }
    
    cm.stateMachine.ProcessEvent(ctx, cmd, EventSchedule)
}

// UplinkCommand uplinks the command
func (cm *CommandManager) UplinkCommand(ctx context.Context, cmdID string) error {
    cmd, err := cm.store.Get(ctx, cmdID)
    if err != nil {
        return err
    }
    
    if cmd.State != StateScheduled {
        return fmt.Errorf("command not in scheduled state: %s", cmd.State)
    }
    
    // Transition to uplinking
    if err := cm.stateMachine.ProcessEvent(ctx, cmd, EventUplink); err != nil {
        return err
    }
    
    // Start uplink with timeout
    uplinkCtx, cancel := context.WithTimeout(ctx, cm.config.UplinkTimeout)
    defer cancel()
    
    result, err := cm.uplink.Uplink(uplinkCtx, cmd)
    if err != nil {
        cmd.LastError = err.Error()
        cm.stateMachine.ProcessEvent(ctx, cmd, EventFail)
        return err
    }
    
    if result.Confirmed {
        now := time.Now()
        cmd.UplinkTime = &now
        cm.stateMachine.ProcessEvent(ctx, cmd, EventConfirm)
        
        // Wait for ACK
        go cm.waitForAck(context.Background(), cmd)
    } else if result.Timeout {
        cm.stateMachine.ProcessEvent(ctx, cmd, EventTimeout)
        cm.handleRetry(ctx, cmd)
    } else {
        cmd.LastError = result.Error
        cm.stateMachine.ProcessEvent(ctx, cmd, EventFail)
        cm.handleRetry(ctx, cmd)
    }
    
    return nil
}

// waitForAck waits for command acknowledgment
func (cm *CommandManager) waitForAck(ctx context.Context, cmd *Command) {
    // Subscribe to ACK events
    ackChan := cm.eventBus.Subscribe(fmt.Sprintf("cmd.ack.%s", cmd.ID))
    defer cm.eventBus.Unsubscribe(ackChan)
    
    timeout := time.After(cm.config.UplinkTimeout)
    
    select {
    case <-ackChan:
        now := time.Now()
        cmd.AckTime = &now
        cm.stateMachine.ProcessEvent(ctx, cmd, EventAck)
        
        // Start verification
        go cm.verifyCommand(context.Background(), cmd)
        
    case <-timeout:
        cm.stateMachine.ProcessEvent(ctx, cmd, EventTimeout)
        cm.handleRetry(ctx, cmd)
        
    case <-ctx.Done():
        return
    }
}

// verifyCommand verifies command execution
func (cm *CommandManager) verifyCommand(ctx context.Context, cmd *Command) {
    // Transition to verifying
    if err := cm.stateMachine.ProcessEvent(ctx, cmd, EventVerify); err != nil {
        return
    }
    
    verifyCtx, cancel := context.WithTimeout(ctx, cm.config.VerificationTimeout)
    defer cancel()
    
    result, err := cm.verification.Verify(verifyCtx, cmd)
    if err != nil {
        cmd.LastError = err.Error()
        cm.stateMachine.ProcessEvent(ctx, cmd, EventTimeout)
        cm.handleRetry(ctx, cmd)
        return
    }
    
    if result.Verified {
        now := time.Now()
        cmd.VerifyTime = &now
        cmd.CompletedTime = &now
        cmd.ActualState = result.ActualState
        cm.stateMachine.ProcessEvent(ctx, cmd, EventSuccess)
    } else {
        cmd.LastReason = result.Reason
        cmd.ActualState = result.ActualState
        cm.stateMachine.ProcessEvent(ctx, cmd, EventFail)
        cm.handleRetry(ctx, cmd)
    }
}

// handleRetry handles command retry
func (cm *CommandManager) handleRetry(ctx context.Context, cmd *Command) {
    if cmd.RetryCount >= cmd.MaxRetries {
        return // Max retries reached
    }
    
    cmd.RetryCount++
    cm.stateMachine.ProcessEvent(ctx, cmd, EventRetry)
    
    // Delay before retry
    time.AfterFunc(cm.config.RetryDelay, func() {
        cm.stateMachine.ProcessEvent(ctx, cmd, EventSchedule)
        cm.scheduleCommand(context.Background(), cmd)
    })
}

// CancelCommand cancels a command
func (cm *CommandManager) CancelCommand(ctx context.Context, cmdID, reason string) error {
    cmd, err := cm.store.Get(ctx, cmdID)
    if err != nil {
        return err
    }
    
    if !cm.stateMachine.CanTransition(cmd.State, EventCancel) {
        return fmt.Errorf("cannot cancel command in state %s", cmd.State)
    }
    
    cmd.LastReason = reason
    return cm.stateMachine.ProcessEvent(ctx, cmd, EventCancel)
}

    ExpectedState      map[string]interface{}
```

```go
// validator.go

package command

import (
    "context"
    "fmt"
    "strings"
)

// CommandValidator validates commands
type CommandValidator struct {
    catalog      *CommandCatalog
    ruleEngine   *RuleEngine
    authz        AuthorizationService
    stateChecker *StateChecker
}

// ValidationResult contains validation results
type ValidationResult struct {
    Valid        bool
    Unauthorized bool
    Errors       []ValidationError
    Warnings     []ValidationWarning
    Reason       string
}

// ValidationError represents a validation error
type ValidationError struct {
    Code    string
    Field   string
    Message string
}

// ValidationWarning represents a validation warning
type ValidationWarning struct {
    Code    string
    Field   string
    Message string
}

// NewCommandValidator creates a new validator
func NewCommandValidator(catalog *CommandCatalog, authz AuthorizationService) *CommandValidator {
    return &CommandValidator{
        catalog:      catalog,
        ruleEngine:   NewRuleEngine(),
        authz:        authz,
        stateChecker: NewStateChecker(),
    }
}

// Validate validates a command
func (cv *CommandValidator) Validate(ctx context.Context, cmd *Command) (*ValidationResult, error) {
    result := &ValidationResult{Valid: true}
    
    // 1. Get command definition
    def, ok := cv.catalog.Get(cmd.Type)
    if !ok {
        result.Valid = false
        result.Errors = append(result.Errors, ValidationError{
            Code:    "UNKNOWN_COMMAND",
            Message: fmt.Sprintf("unknown command type: %s", cmd.Type),
        })
        return result, nil
    }
    
    // 2. Validate parameters
    paramErrors := cv.validateParameters(cmd, def)
    result.Errors = append(result.Errors, paramErrors...)
    
    // 3. Check authorization
    authResult, err := cv.checkAuthorization(ctx, cmd, def)
    if err != nil {
        return nil, err
    }
    if !authResult.Authorized {
        result.Valid = false
        result.Unauthorized = true
        result.Reason = authResult.Reason
        return result, nil
    }
    
    // 4. Check constraints
    constraintErrors := cv.checkConstraints(ctx, cmd, def)
    result.Errors = append(result.Errors, constraintErrors...)
    
    // 5. Check spacecraft state
    stateErrors := cv.checkSpacecraftState(ctx, cmd, def)
    result.Errors = append(result.Errors, stateErrors...)
    
    // 6. Run rule engine
    ruleResult := cv.ruleEngine.Evaluate(ctx, cmd, def)
    result.Errors = append(result.Errors, ruleResult.Errors...)
    result.Warnings = append(result.Warnings, ruleResult.Warnings...)
    
    // Set final validity
    result.Valid = len(result.Errors) == 0
    if !result.Valid {
        result.Reason = cv.summarizeErrors(result.Errors)
    }
    
    return result, nil
}

// validateParameters validates command parameters
func (cv *CommandValidator) validateParameters(cmd *Command, def *CommandDefinition) []ValidationError {
    var errors []ValidationError
    
    for _, paramDef := range def.Parameters {
        value, exists := cmd.Parameters[paramDef.Name]
        
        // Check required
        if paramDef.Required && !exists {
            errors = append(errors, ValidationError{
                Code:    "REQUIRED_PARAMETER",
                Field:   paramDef.Name,
                Message: fmt.Sprintf("required parameter '%s' is missing", paramDef.Name),
            })
            continue
        }
        
        if !exists {
            continue
        }
        
        // Validate type
        if err := cv.validateParameterType(value, paramDef); err != nil {
            errors = append(errors, ValidationError{
                Code:    "INVALID_TYPE",
                Field:   paramDef.Name,
                Message: err.Error(),
            })
        }
        
        // Validate range
        if paramDef.Range != nil {
            if err := cv.validateRange(value, paramDef); err != nil {
                errors = append(errors, ValidationError{
                    Code:    "OUT_OF_RANGE",
                    Field:   paramDef.Name,
                    Message: err.Error(),
                })
            }
        }
        
        // Validate enumeration
        if paramDef.Type == ParamTypeEnum {
            if err := cv.validateEnumeration(value, paramDef); err != nil {
                errors = append(errors, ValidationError{
                    Code:    "INVALID_ENUM",
                    Field:   paramDef.Name,
                    Message: err.Error(),
                })
            }
        }
    }
    
    return errors
}

// checkAuthorization checks command authorization
func (cv *CommandValidator) checkAuthorization(ctx context.Context, cmd *Command, def *CommandDefinition) (*AuthorizationResult, error) {
    // Build authorization request
    req := &AuthorizationRequest{
        UserID:       cmd.CreatedBy,
        Action:       "command:execute",
        Resource:     fmt.Sprintf("spacecraft:%s:command:%s", cmd.SpacecraftID, def.Mnemonic),
        Context: map[string]interface{}{
            "criticality":  def.Criticality,
            "command_type": cmd.Type,
            "spacecraft":   cmd.SpacecraftID,
        },
    }
    
    // Check authorization
    return cv.authz.Authorize(ctx, req)
}

// checkConstraints evaluates command constraints
func (cv *CommandValidator) checkConstraints(ctx context.Context, cmd *Command, def *CommandDefinition) []ValidationError {
    var errors []ValidationError
    
    for _, constraint := range def.Constraints {
        if constraint.Type != ConstraintPrecondition {
            continue
        }
        
        // Evaluate constraint expression
        satisfied, err := cv.ruleEngine.EvaluateExpression(ctx, constraint.Expression, cmd.Parameters)
        if err != nil {
            errors = append(errors, ValidationError{
                Code:    "CONSTRAINT_ERROR",
                Message: fmt.Sprintf("failed to evaluate constraint: %v", err),
            })
            continue
        }
        
        if !satisfied {
            errors = append(errors, ValidationError{
                Code:    "CONSTRAINT_VIOLATED",
                Message: constraint.Description,
            })
        }
    }
    
    return errors
}

// checkSpacecraftState validates against current spacecraft state
func (cv *CommandValidator) checkSpacecraftState(ctx context.Context, cmd *Command, def *CommandDefinition) []ValidationError {
    var errors []ValidationError
    
    // Get current spacecraft state
    state, err := cv.stateChecker.GetState(ctx, cmd.SpacecraftID)
    if err != nil {
        errors = append(errors, ValidationError{
            Code:    "STATE_CHECK_FAILED",
            Message: fmt.Sprintf("failed to get spacecraft state: %v", err),
        })
        return errors
    }
    
    // Check if command is allowed in current mode
    if !cv.isCommandAllowedInMode(def, state.Mode) {
        errors = append(errors, ValidationError{
            Code:    "INVALID_MODE",
            Message: fmt.Sprintf("command not allowed in mode '%s'", state.Mode),
        })
    }
    
    // Check power state for payload commands
    if strings.HasPrefix(def.Mnemonic, "PAYLOAD_") {
        if state.PowerMode == "LOW_POWER" {
            errors = append(errors, ValidationError{
                Code:    "LOW_POWER",
                Message: "payload commands not allowed in low power mode",
            })
        }
    }
    
    // Check communication link
    if !state.LinkActive {
        errors = append(errors, ValidationError{
            Code:    "NO_LINK",
            Message: "no active communication link",
        })
    }
    
    return errors
}

// isCommandAllowedInMode checks if command is allowed in spacecraft mode
func (cv *CommandValidator) isCommandAllowedInMode(def *CommandDefinition, mode string) bool {
    // Critical commands always allowed for recovery
    if def.Criticality == CriticalityCritical {
        return true
    }
    
    // Mode restrictions
    modeRestrictions := map[string][]CriticalityLevel{
        "SAFE":     {CriticalityCritical},
        "STANDBY":  {CriticalityCritical, CriticalityHigh},
        "NOMINAL":  {CriticalityCritical, CriticalityHigh, CriticalityMedium, CriticalityLow},
        "SCIENCE":  {CriticalityCritical, CriticalityHigh, CriticalityMedium, CriticalityLow},
    }
    
    allowed, ok := modeRestrictions[mode]
    if !ok {
        return false
    }
    
    for _, level := range allowed {
        if def.Criticality == level {
            return true
        }
    }
    
    return false
}

// summarizeErrors creates error summary
func (cv *CommandValidator) summarizeErrors(errors []ValidationError) string {
    if len(errors) == 0 {
        return ""
    }
    
    messages := make([]string, len(errors))
    for i, err := range errors {
        messages[i] = err.Message
    }
    return strings.Join(messages, "; ")
}

    Authorized bool
    Reason     string
}
```

```go
}

// EncoderConfig configures the encoder
type EncoderConfig struct {
    SpacecraftID    uint16
    VirtualChannel  uint8
    BypassFlag      bool
    ControlCommand  bool
    UseCRC          bool
    UseEncryption   bool
}

// CCSDS TC Space Data Link constants
const (
    TCFrameHeaderSize   = 5
    TCFrameMinSize      = 7
    TCFrameMaxSize      = 1024
    TCSegmentHeaderSize = 1
    TCPacketHeaderSize  = 6
)

// NewCommandEncoder creates a new encoder
func NewCommandEncoder(catalog *CommandCatalog, encryptor Encryptor, config *EncoderConfig) *CommandEncoder {
    return &CommandEncoder{
        catalog:   catalog,
        encryptor: encryptor,
        config:    config,
    }
}

// Encode encodes command parameters to CCSDS TC Packet
func (ce *CommandEncoder) Encode(ctx context.Context, cmd *Command) ([]byte, error) {
    def, ok := ce.catalog.Get(cmd.Type)
    if !ok {
        return nil, fmt.Errorf("unknown command type: %s", cmd.Type)
    }
    
    // Build packet data field
    dataField, err := ce.encodeDataField(cmd, def)
    if err != nil {
        return nil, fmt.Errorf("failed to encode data field: %w", err)
    }
    
    // Build TC Packet
    packet := ce.buildTCPacket(def, dataField)
    
    // Encrypt if configured
    if ce.config.UseEncryption {
        encrypted, err := ce.encryptor.Encrypt(ctx, packet)
        if err != nil {
            return nil, fmt.Errorf("encryption failed: %w", err)
        }
        packet = encrypted
    }
    
    return packet, nil
}

// encodeDataField encodes command parameters
func (ce *CommandEncoder) encodeDataField(cmd *Command, def *CommandDefinition) ([]byte, error) {
    // Calculate total size
    totalBits := 0
    for _, param := range def.Parameters {
        totalBits += param.Size
    }
    totalBytes := (totalBits + 7) / 8
    
    data := make([]byte, totalBytes)
    
    for _, paramDef := range def.Parameters {
        value, exists := cmd.Parameters[paramDef.Name]
        if !exists {
            if paramDef.Default != nil {
                value = paramDef.Default
            } else if paramDef.Required {
                return nil, fmt.Errorf("missing required parameter: %s", paramDef.Name)
            } else {
                continue
            }
        }
        
        if err := ce.encodeParameter(data, value, paramDef); err != nil {
            return nil, fmt.Errorf("failed to encode %s: %w", paramDef.Name, err)
        }
    }
    
    return data, nil
}

// encodeParameter encodes a single parameter
func (ce *CommandEncoder) encodeParameter(data []byte, value interface{}, def ParameterDefinition) error {
    byteOffset := def.Offset / 8
    bitOffset := def.Offset % 8
    
    switch def.Type {
    case ParamTypeUint:
        return ce.encodeUint(data, value, byteOffset, bitOffset, def.Size)
    case ParamTypeInt:
        return ce.encodeInt(data, value, byteOffset, bitOffset, def.Size)
    case ParamTypeFloat:
        return ce.encodeFloat32(data, value, byteOffset)
    case ParamTypeDouble:
        return ce.encodeFloat64(data, value, byteOffset)
    case ParamTypeBoolean:
        return ce.encodeBool(data, value, byteOffset, bitOffset)
    case ParamTypeEnum:
        return ce.encodeUint(data, value, byteOffset, bitOffset, def.Size)
    case ParamTypeOctet:
        return ce.encodeOctets(data, value, byteOffset, def.Size/8)
    case ParamTypeTime:
        return ce.encodeTime(data, value, byteOffset)
    default:
        return fmt.Errorf("unsupported parameter type: %s", def.Type)
    }
}

// encodeUint encodes unsigned integer
func (ce *CommandEncoder) encodeUint(data []byte, value interface{}, byteOffset, bitOffset, bits int) error {
    var v uint64
    switch val := value.(type) {
    case int:
        v = uint64(val)
    case int64:
        v = uint64(val)
    case uint64:
        v = val
    case float64:
        v = uint64(val)
    default:
        return fmt.Errorf("invalid type for uint: %T", value)
    }
    
    // Pack bits
    for i := 0; i < bits; i++ {
        bit := (v >> (bits - 1 - i)) & 1
        byteIdx := byteOffset + (bitOffset+i)/8
        bitIdx := 7 - ((bitOffset + i) % 8)
        if bit == 1 {
            data[byteIdx] |= 1 << bitIdx
        }
    }
    
    return nil
}

// encodeFloat64 encodes double precision float
func (ce *CommandEncoder) encodeFloat64(data []byte, value interface{}, byteOffset int) error {
    var v float64
    switch val := value.(type) {
    case float64:
        v = val
    case float32:
        v = float64(val)
    default:
        return fmt.Errorf("invalid type for double: %T", value)
    }
    
    binary.BigEndian.PutUint64(data[byteOffset:], math.Float64bits(v))
    return nil
}

// buildTCPacket builds CCSDS TC Packet
func (ce *CommandEncoder) buildTCPacket(def *CommandDefinition, dataField []byte) []byte {
    packet := new(bytes.Buffer)
    
    // Packet Primary Header (6 bytes)
    // Version (3 bits) = 0
    // Type (1 bit) = 1 (TC)
    // Secondary Header Flag (1 bit) = 0
    // APID (11 bits)
    versionType := uint16(0x1000) | (def.APID & 0x07FF)
    binary.Write(packet, binary.BigEndian, versionType)
    
    // Sequence Flags (2 bits) = 11 (standalone)
    // Sequence Count (14 bits) = 0 (will be set later)
    seqControl := uint16(0xC000) // Standalone packet
    binary.Write(packet, binary.BigEndian, seqControl)
    
    // Packet Data Length (length of data field - 1)
    dataLength := uint16(len(dataField) + 3) // +3 for PUS header
    binary.Write(packet, binary.BigEndian, dataLength)
    
    // PUS TC Packet Data Field Header (3 bytes minimum)
    // Version = 1, ACK = 0xF (all acknowledgments)
    pusVersion := uint8(0x1F) // Version 1, ACK all
    packet.WriteByte(pusVersion)
    
    // Service Type
    packet.WriteByte(def.ServiceType)
    
    // Service Subtype
    packet.WriteByte(def.ServiceSubtype)
    
    // Application Data
    packet.Write(dataField)
    
    return packet.Bytes()
}

// Frame creates CCSDS TC Frame
func (ce *CommandEncoder) Frame(ctx context.Context, cmd *Command) ([]byte, error) {
    if cmd.EncodedData == nil {
        return nil, fmt.Errorf("command not encoded")
    }
    
    frame := new(bytes.Buffer)
    
    // TC Frame Primary Header (5 bytes)
    // Transfer Frame Version (2 bits) = 00
    // Bypass Flag (1 bit)
    // Control Command Flag (1 bit)
    // Reserved (2 bits) = 00
    // Spacecraft ID (10 bits)
    header1 := uint16(ce.config.SpacecraftID & 0x03FF)
    if ce.config.BypassFlag {
        header1 |= 0x2000
    }
    if ce.config.ControlCommand {
        header1 |= 0x1000
    }
    binary.Write(frame, binary.BigEndian, header1)
    
    // Virtual Channel ID (6 bits)
    // Frame Length (10 bits)
    frameLength := uint16(len(cmd.EncodedData) + TCFrameHeaderSize + 2) // +2 for CRC
    header2 := uint16(ce.config.VirtualChannel&0x3F)<<10 | (frameLength & 0x03FF)
    binary.Write(frame, binary.BigEndian, header2)
    
    // Frame Sequence Number (8 bits)
    frame.WriteByte(uint8(cmd.SequenceNumber))
    
    // TC Segment Header (1 byte)
    // Sequence Flags (2 bits) = 11 (unsegmented)
    // MAP ID (6 bits) = 0
    segmentHeader := uint8(0xC0)
    frame.WriteByte(segmentHeader)
    
    // Frame Data Field (TC Packet)
    frame.Write(cmd.EncodedData)
    
    // Frame Error Control Field (CRC-16)
    if ce.config.UseCRC {
        crc := ce.calculateCRC(frame.Bytes())
        binary.Write(frame, binary.BigEndian, crc)
    }
    
    return frame.Bytes(), nil
}

// calculateCRC calculates CRC-16-CCITT
func (ce *CommandEncoder) calculateCRC(data []byte) uint16 {
    crc := uint16(0xFFFF)
    poly := uint16(0x1021)
    
    for _, b := range data {
        crc ^= uint16(b) << 8
        for i := 0; i < 8; i++ {
            if crc&0x8000 != 0 {
                crc = (crc << 1) ^ poly
            } else {
                crc <<= 1
            }
        }
    }
    
    return crc
}

// Encryptor interface for command encryption
type Encryptor interface {
    Encrypt(ctx context.Context, data []byte) ([]byte, error)
    Decrypt(ctx context.Context, data []byte) ([]byte, error)
}
```

```go
// NewUplinkController creates a new controller

        antenna:      antenna,

// StartPass starts a command uplink pass
func (uc *UplinkController) StartPass(ctx context.Context, passID string, commands []*Command) error {
    uc.mu.Lock()
    defer uc.mu.Unlock()
    
    if len(commands) == 0 {
        return fmt.Errorf("no commands to uplink")
    }
    
    // Create active pass
    pass := &ActivePass{
        PassID:       passID,
        SpacecraftID: commands[0].SpacecraftID,
        StartTime:    time.Now(),
        Commands:     commands,
        CurrentIndex: 0,
        Status:       "active",
    }
    
    uc.activePasses[passID] = pass
    
    // Configure modem
    if err := uc.configureModem(ctx); err != nil {
        return fmt.Errorf("failed to configure modem: %w", err)
    }
    
    // Start uplink sequence
    go uc.runUplinkSequence(ctx, pass)
    
    return nil
}

// configureModem configures the modem for uplink
func (uc *UplinkController) configureModem(ctx context.Context) error {
    config := &ModemConfig{
        Mode:       "TX",
        Frequency:  uc.config.Frequency,
        DataRate:   uc.config.DataRate,
        Modulation: uc.config.Modulation,
        CodingRate: uc.config.CodingRate,
        Power:      uc.config.TransmitPower,
    }
    
    return uc.modem.Configure(ctx, config)
}

// runUplinkSequence executes the uplink sequence
func (uc *UplinkController) runUplinkSequence(ctx context.Context, pass *ActivePass) {
    for i, cmd := range pass.Commands {
        pass.CurrentIndex = i
        
        // Check if pass still active
        if pass.Status != "active" {
            break
        }
        
        // Uplink command
        result, err := uc.uplinkSingleCommand(ctx, cmd)
        if err != nil {
            uc.eventBus.Publish(ctx, &CommandUplinkErrorEvent{
                CommandID: cmd.ID,
                PassID:    pass.PassID,
                Error:     err.Error(),
            })
            continue
        }
        
        // Emit uplink event
        uc.eventBus.Publish(ctx, &CommandUplinkEvent{
            CommandID:   cmd.ID,
            PassID:      pass.PassID,
            Result:      result,
        })
        
        // Inter-command delay
        if i < len(pass.Commands)-1 {
            time.Sleep(uc.config.InterCommandDelay)
        }
    }
    
    pass.Status = "completed"
    pass.EndTime = time.Now()
}

// uplinkSingleCommand uplinks a single command
func (uc *UplinkController) uplinkSingleCommand(ctx context.Context, cmd *Command) (*UplinkResult, error) {
    result := &UplinkResult{}
    
    for attempt := 0; attempt <= uc.config.MaxRetransmits; attempt++ {
        result.Retransmits = attempt
        
        // Transmit frame
        txCtx, cancel := context.WithTimeout(ctx, uc.config.UplinkTimeout)
        
        err := uc.modem.Transmit(txCtx, cmd.FrameData)
        cancel()
        
        if err != nil {
            result.Error = err.Error()
            continue
        }
        
        result.TransmitTime = time.Now()
        
        // Wait for CLTU confirmation
        confCtx, confCancel := context.WithTimeout(ctx, uc.config.ConfirmTimeout)
        
        confirmed, err := uc.waitForConfirmation(confCtx, cmd)
        confCancel()
        
        if err != nil {
            result.Error = err.Error()
            continue
        }
        
        if confirmed {
            result.Confirmed = true
            result.ConfirmTime = time.Now()
            return result, nil
        }
    }
    
    if result.Error == "" {
        result.Timeout = true
        result.Error = "confirmation timeout"
    }
    
    return result, fmt.Errorf("uplink failed: %s", result.Error)
}

// waitForConfirmation waits for CLTU confirmation
func (uc *UplinkController) waitForConfirmation(ctx context.Context, cmd *Command) (bool, error) {
    // Subscribe to modem status
    statusChan := uc.modem.StatusChannel()
    
    for {
        select {
        case status := <-statusChan:
            if status.Type == "CLTU_CONFIRM" && status.SequenceNumber == cmd.SequenceNumber {
                return true, nil
            }
            if status.Type == "CLTU_REJECT" && status.SequenceNumber == cmd.SequenceNumber {
                return false, fmt.Errorf("CLTU rejected: %s", status.Reason)
            }
        case <-ctx.Done():
            return false, ctx.Err()
        }
    }
}

// Uplink uplinks a command (called by CommandManager)
func (uc *UplinkController) Uplink(ctx context.Context, cmd *Command) (*UplinkResult, error) {
    // Check if there's an active pass
    uc.mu.RLock()
    var activePass *ActivePass
    for _, pass := range uc.activePasses {
        if pass.SpacecraftID == cmd.SpacecraftID && pass.Status == "active" {
            activePass = pass
            break
        }
    }
    uc.mu.RUnlock()
    
    if activePass == nil {
        return nil, fmt.Errorf("no active pass for spacecraft %s", cmd.SpacecraftID)
    }
    
    return uc.uplinkSingleCommand(ctx, cmd)
}

// AbortPass aborts an active pass
func (uc *UplinkController) AbortPass(ctx context.Context, passID string) error {
    uc.mu.Lock()
    defer uc.mu.Unlock()
    
    pass, ok := uc.activePasses[passID]
    if !ok {
        return fmt.Errorf("pass not found: %s", passID)
    }
    
    pass.Status = "aborted"
    
    // Stop transmission
    uc.modem.StopTransmit(ctx)
    
    return nil
}
```

```go
    Actual      interface{}

// NewVerificationEngine creates a new engine
func NewVerificationEngine(telemetry TelemetryClient, stateStore StateStore, catalog *CommandCatalog) *VerificationEngine {

        },
    }
}

// Verify verifies command execution
func (ve *VerificationEngine) Verify(ctx context.Context, cmd *Command) (*VerificationResult, error) {
    def, ok := ve.catalog.Get(cmd.Type)
    if !ok {
        return nil, fmt.Errorf("unknown command type: %s", cmd.Type)
    }
    
    verifyDef := def.Verification
    result := &VerificationResult{
        Method:      verifyDef.Method,
        ActualState: make(map[string]interface{}),
        Checks:      make([]VerificationCheck, 0),
    }
    
    startTime := time.Now()
    timeout := verifyDef.Timeout
    if timeout == 0 {
        timeout = ve.config.DefaultTimeout
    }
    
    // Create timeout context
    verifyCtx, cancel := context.WithTimeout(ctx, timeout)
    defer cancel()
    
    switch verifyDef.Method {
    case VerifyByAck:
        result.Verified = true // Already ACKed
        
    case VerifyByState:
        err := ve.verifyByState(verifyCtx, cmd, def, result)
        if err != nil {
            return nil, err
        }
        
    case VerifyByTelemetry:
        err := ve.verifyByTelemetry(verifyCtx, cmd, def, result)
        if err != nil {
            return nil, err
        }
        
    case VerifyByExecution:
        err := ve.verifyByExecution(verifyCtx, cmd, def, result)
        if err != nil {
            return nil, err
        }
        
    case VerifyByCompletion:
        err := ve.verifyByCompletion(verifyCtx, cmd, def, result)
        if err != nil {
            return nil, err
        }
        
    default:
        return nil, fmt.Errorf("unknown verification method: %s", verifyDef.Method)
    }
    
    result.Duration = time.Since(startTime)
    
    return result, nil
}

// verifyByState verifies by checking spacecraft state
func (ve *VerificationEngine) verifyByState(ctx context.Context, cmd *Command, def *CommandDefinition, result *VerificationResult) error {
    verifyDef := def.Verification
    
    ticker := time.NewTicker(ve.config.PollingInterval)
    defer ticker.Stop()
    
    for checkCount := 0; checkCount < ve.config.MaxStateChecks; checkCount++ {
        select {
        case <-ctx.Done():
            result.Reason = "verification timeout"
            return nil
            
        case <-ticker.C:
            // Get current state
            state, err := ve.stateStore.GetState(ctx, cmd.SpacecraftID)
            if err != nil {
                continue
            }
            
            // Check each parameter
            allMatched := true
            for _, paramName := range verifyDef.Parameters {
                actual, ok := state.Parameters[paramName]
                if !ok {
                    allMatched = false
                    continue
                }
                
                expected, ok := cmd.ExpectedState[paramName]
                if !ok {
                    continue
                }
                
                matched := ve.compareValues(expected, actual)
                
                result.Checks = append(result.Checks, VerificationCheck{
                    Parameter: paramName,
                    Expected:  expected,
                    Actual:    actual,
                    Matched:   matched,
                    Timestamp: time.Now(),
                })
                
                result.ActualState[paramName] = actual
                
                if !matched {
                    allMatched = false
                }
            }
            
            if allMatched {
                result.Verified = true
                return nil
            }
        }
    }
    
    result.Reason = "state mismatch after max checks"
    return nil
}

// verifyByTelemetry verifies by checking telemetry values
func (ve *VerificationEngine) verifyByTelemetry(ctx context.Context, cmd *Command, def *CommandDefinition, result *VerificationResult) error {
    verifyDef := def.Verification
    
    // Subscribe to telemetry
    tmChan, err := ve.telemetryClient.Subscribe(ctx, cmd.SpacecraftID, verifyDef.Parameters)
    if err != nil {
        return fmt.Errorf("failed to subscribe to telemetry: %w", err)
    }
    defer ve.telemetryClient.Unsubscribe(tmChan)
    
    for {
        select {
        case <-ctx.Done():
            result.Reason = "verification timeout"
            return nil
            
        case tm := <-tmChan:
            // Check telemetry values
            allMatched := true
            for paramName, expected := range cmd.ExpectedState {
                actual, ok := tm.Parameters[paramName]
                if !ok {
                    continue
                }
                
                matched := ve.compareValues(expected, actual)
                
                result.Checks = append(result.Checks, VerificationCheck{
                    Parameter: paramName,
                    Expected:  expected,
                    Actual:    actual,
                    Matched:   matched,
                    Timestamp: tm.Timestamp,
                })
                
                result.ActualState[paramName] = actual
                
                if !matched {
                    allMatched = false
                }
            }
            
            if allMatched {
                result.Verified = true
                return nil
            }
        }
    }
}

// verifyByExecution verifies by checking execution telemetry
func (ve *VerificationEngine) verifyByExecution(ctx context.Context, cmd *Command, def *CommandDefinition, result *VerificationResult) error {
    verifyDef := def.Verification
    
    // Wait for execution acknowledgment
    ackChan := ve.eventBus.Subscribe(fmt.Sprintf("cmd.exec.%s", cmd.ID))
    defer ve.eventBus.Unsubscribe(ackChan)
    
    select {
    case <-ctx.Done():
        result.Reason = "execution verification timeout"
        return nil
        
    case event := <-ackChan:
        execEvent, ok := event.(*CommandExecutionEvent)
        if !ok {
            result.Reason = "invalid execution event"
            return nil
        }
        
        if execEvent.Success {
            result.Verified = true
            result.ActualState = execEvent.ResultState
        } else {
            result.Reason = execEvent.Error
            result.ActualState = execEvent.ResultState
        }
    }
    
    return nil
}

// verifyByCompletion waits for completion report
func (ve *VerificationEngine) verifyByCompletion(ctx context.Context, cmd *Command, def *CommandDefinition, result *VerificationResult) error {
    // Subscribe to completion events from TM
    completionChan := ve.eventBus.Subscribe(fmt.Sprintf("tm.completion.%s.%d", cmd.SpacecraftID, cmd.SequenceNumber))
    defer ve.eventBus.Unsubscribe(completionChan)
    
    select {
    case <-ctx.Done():
        result.Reason = "completion verification timeout"
        return nil
        
    case event := <-completionChan:
        compEvent, ok := event.(*CommandCompletionEvent)
        if !ok {
            result.Reason = "invalid completion event"
            return nil
        }
        
        if compEvent.Status == "SUCCESS" {
            result.Verified = true
        } else {
            result.Reason = compEvent.FailureReason
        }
        result.ActualState = compEvent.ResultState
    }
    
    return nil
}

// compareValues compares expected and actual values
func (ve *VerificationEngine) compareValues(expected, actual interface{}) bool {
    switch exp := expected.(type) {
    case string:
        act, ok := actual.(string)
        return ok && exp == act
        
    case float64:
        act, ok := actual.(float64)
        if !ok {
            return false
        }
        // Use tolerance for floating point
        diff := exp - act
        if diff < 0 {
            diff = -diff
        }
        return diff <= ve.config.StateTolerance*exp
        
    case int:
        switch act := actual.(type) {
        case int:
            return exp == act
        case int64:
            return int64(exp) == act
        case float64:
            return float64(exp) == act
        }
        
    case bool:
        act, ok := actual.(bool)
        return ok && exp == act
    }
    
    return fmt.Sprintf("%v", expected) == fmt.Sprintf("%v", actual)
}
```

```go
        queue:         NewPriorityQueue(),
        scheduled:     make(map[string]*ScheduledCommand),
        conflicts:     NewConflictChecker(),
        config:        config,
    }
}

// Enqueue adds a command to the queue
func (cs *CommandScheduler) Enqueue(ctx context.Context, cmd *Command) error {
    cs.mu.Lock()
    defer cs.mu.Unlock()
    
    if cs.queue.Len() >= cs.config.MaxQueueSize {
        return fmt.Errorf("queue full: %d commands", cs.queue.Len())
    }
    
    item := &QueueItem{
        Command:   cmd,
        Priority:  cmd.Priority,
        Timestamp: cmd.CreatedAt,
    }
    
    heap.Push(cs.queue, item)
    
    cs.eventBus.Publish(ctx, &CommandQueuedEvent{
        CommandID:     cmd.ID,
        QueuePosition: cs.queue.Len(),
    })
    
    return nil
}

// Schedule schedules a command to a pass
func (cs *CommandScheduler) Schedule(ctx context.Context, cmd *Command, pass *Pass) error {
    cs.mu.Lock()
    defer cs.mu.Unlock()
    
    // Check if pass can accommodate more commands
    if len(pass.Commands) >= cs.config.MaxCommandsPerPass {
        return fmt.Errorf("pass %s full", pass.ID)
    }
    
    // Calculate scheduled time
    position := len(pass.Commands)
    scheduledTime := pass.AOS.Add(time.Duration(position) * time.Second)
    
    scheduled := &ScheduledCommand{
        Command:       cmd,
        Pass:          pass,
        ScheduledTime: scheduledTime,
        Position:      position,
    }
    
    cs.scheduled[cmd.ID] = scheduled
    pass.Commands = append(pass.Commands, cmd)
    
    cmd.ScheduledTime = &scheduledTime
    
    return nil
}

// FindNextPass finds the next available pass
func (cs *CommandScheduler) FindNextPass(ctx context.Context, spacecraftID string) (*Pass, error) {
    // Get predicted passes
    now := time.Now()
    endTime := now.Add(cs.config.LookAheadDuration)
    
    passes, err := cs.passPredictor.PredictPasses(ctx, spacecraftID, now, endTime)
    if err != nil {
        return nil, fmt.Errorf("failed to predict passes: %w", err)
    }
    
    // Find first pass with capacity
    for _, pass := range passes {
        if len(pass.Commands) < cs.config.MaxCommandsPerPass {
            return pass, nil
        }
    }
    
    return nil, fmt.Errorf("no available passes in next %v", cs.config.LookAheadDuration)
}

// CheckConflicts checks for scheduling conflicts
func (cs *CommandScheduler) CheckConflicts(ctx context.Context, cmd *Command) ([]ConflictInfo, error) {
    return cs.conflicts.Check(ctx, cmd, cs.scheduled)
}

// GetNextSequenceNumber returns the next sequence number
func (cs *CommandScheduler) GetNextSequenceNumber(spacecraftID string) uint16 {
    cs.mu.Lock()
    defer cs.mu.Unlock()
    
    // Track per-spacecraft sequence numbers
    // In real implementation, this would be persisted
    return uint16(time.Now().UnixNano() % 65536)
}

// GetPassCommands gets commands scheduled for a pass
func (cs *CommandScheduler) GetPassCommands(ctx context.Context, passID string) ([]*Command, error) {
    cs.mu.RLock()
    defer cs.mu.RUnlock()
    
    commands := make([]*Command, 0)
    for _, scheduled := range cs.scheduled {
        if scheduled.Pass.ID == passID {
            commands = append(commands, scheduled.Command)
        }
    }
    
    // Sort by position
    sort.Slice(commands, func(i, j int) bool {
        return cs.scheduled[commands[i].ID].Position < cs.scheduled[commands[j].ID].Position
    })
    
    return commands, nil
}

// Unschedule removes a command from schedule
func (cs *CommandScheduler) Unschedule(ctx context.Context, cmdID string) error {
    cs.mu.Lock()
    defer cs.mu.Unlock()
    
    scheduled, ok := cs.scheduled[cmdID]
    if !ok {
        return fmt.Errorf("command not scheduled: %s", cmdID)
    }
    
    // Remove from pass
    pass := scheduled.Pass
    for i, cmd := range pass.Commands {
        if cmd.ID == cmdID {
            pass.Commands = append(pass.Commands[:i], pass.Commands[i+1:]...)
            break
        }
    }
    
    delete(cs.scheduled, cmdID)
    
    return nil
}

// PriorityQueue implements a priority queue for commands
type PriorityQueue struct {

// QueueItem represents an item in the queue

    Command   *Command
    Priority  int
    Timestamp time.Time
    index     int
}

// NewPriorityQueue creates a new priority queue
func NewPriorityQueue() *PriorityQueue {
    pq := &PriorityQueue{
        items: make([]*QueueItem, 0),
    }
    heap.Init(pq)
    return pq
}

// Len returns queue length
func (pq *PriorityQueue) Len() int { return len(pq.items) }

// Less compares priorities (higher priority first, then earlier timestamp)
func (pq *PriorityQueue) Less(i, j int) bool {
    if pq.items[i].Priority != pq.items[j].Priority {
        return pq.items[i].Priority > pq.items[j].Priority
    }
    return pq.items[i].Timestamp.Before(pq.items[j].Timestamp)
}

// Swap swaps items
func (pq *PriorityQueue) Swap(i, j int) {
    pq.items[i], pq.items[j] = pq.items[j], pq.items[i]
    pq.items[i].index = i
    pq.items[j].index = j
}

// Push adds an item
func (pq *PriorityQueue) Push(x interface{}) {
    item := x.(*QueueItem)
    item.index = len(pq.items)
    pq.items = append(pq.items, item)
}

// Pop removes and returns the highest priority item
func (pq *PriorityQueue) Pop() interface{} {
    n := len(pq.items)
    item := pq.items[n-1]
    pq.items = pq.items[:n-1]
    return item
}
```

```go
// retry.go

package command

import (
    "context"
    "fmt"
    "math"
    "time"
)

// RetryManager manages command retry logic
type RetryManager struct {
    config       *RetryConfig
    commandMgr   *CommandManager
    eventBus     EventBus
    retryTracker map[string]*RetryState
}

// RetryConfig configures retry behavior
type RetryConfig struct {
    MaxRetries        int
    InitialDelay      time.Duration
    MaxDelay          time.Duration
    BackoffMultiplier float64
    RetryableStates   []CommandState
    RetryableCodes    []string
}

// RetryState tracks retry state for a command
type RetryState struct {
    CommandID    string
    RetryCount   int
    LastRetryAt  time.Time
    NextRetryAt  time.Time
    LastError    string
    Backoff      time.Duration
}

// NewRetryManager creates a new retry manager
func NewRetryManager(config *RetryConfig, commandMgr *CommandManager) *RetryManager {
    return &RetryManager{
        config:       config,
        commandMgr:   commandMgr,
        retryTracker: make(map[string]*RetryState),
    }
}

// ShouldRetry determines if a command should be retried
func (rm *RetryManager) ShouldRetry(cmd *Command, err error) bool {
    // Check max retries
    if cmd.RetryCount >= rm.config.MaxRetries {
        return false
    }
    
    // Check if state is retryable
    retryable := false
    for _, state := range rm.config.RetryableStates {
        if cmd.State == state {
            retryable = true
            break
        }
    }
    
    if !retryable {
        return false
    }
    
    // Check if error is retryable
    if err != nil {
        errStr := err.Error()
        for _, code := range rm.config.RetryableCodes {
            if strings.Contains(errStr, code) {
                return true
            }
        }
    }
    
    return retryable
}

// ScheduleRetry schedules a command retry
func (rm *RetryManager) ScheduleRetry(ctx context.Context, cmd *Command, err error) (*RetryState, error) {
    if !rm.ShouldRetry(cmd, err) {
        return nil, fmt.Errorf("command not retryable")
    }
    
    // Calculate backoff delay
    delay := rm.calculateBackoff(cmd.RetryCount)
    
    state := &RetryState{
        CommandID:   cmd.ID,
        RetryCount:  cmd.RetryCount + 1,
        LastRetryAt: time.Now(),
        NextRetryAt: time.Now().Add(delay),
        LastError:   err.Error(),
        Backoff:     delay,
    }
    
    rm.retryTracker[cmd.ID] = state
    
    // Schedule retry
    time.AfterFunc(delay, func() {
        rm.executeRetry(context.Background(), cmd.ID)
    })
    
    rm.eventBus.Publish(ctx, &CommandRetryScheduledEvent{
        CommandID:   cmd.ID,
        RetryCount:  state.RetryCount,
        NextRetryAt: state.NextRetryAt,
        Delay:       delay,
    })
    
    return state, nil
}

// calculateBackoff calculates exponential backoff delay
func (rm *RetryManager) calculateBackoff(retryCount int) time.Duration {
    delay := float64(rm.config.InitialDelay) * math.Pow(rm.config.BackoffMultiplier, float64(retryCount))
    
    if delay > float64(rm.config.MaxDelay) {
        delay = float64(rm.config.MaxDelay)
    }
    
    // Add jitter (±10%)
    jitter := delay * 0.1 * (2*rand.Float64() - 1)
    delay += jitter
    
    return time.Duration(delay)
}

// executeRetry executes a retry
func (rm *RetryManager) executeRetry(ctx context.Context, cmdID string) {
    state, ok := rm.retryTracker[cmdID]
    if !ok {
        return
    }
    
    // Get command
    cmd, err := rm.commandMgr.store.Get(ctx, cmdID)
    if err != nil {
        return
    }
    
    // Update retry count
    cmd.RetryCount = state.RetryCount
    
    // Re-submit for scheduling
    rm.commandMgr.scheduleCommand(ctx, cmd)
    
    rm.eventBus.Publish(ctx, &CommandRetryExecutedEvent{
        CommandID:  cmdID,
        RetryCount: state.RetryCount,
    })
}

// CancelRetry cancels a scheduled retry
func (rm *RetryManager) CancelRetry(ctx context.Context, cmdID string) error {
    state, ok := rm.retryTracker[cmdID]
    if !ok {
        return fmt.Errorf("no retry scheduled for command: %s", cmdID)
    }
    
    delete(rm.retryTracker, cmdID)
    
    rm.eventBus.Publish(ctx, &CommandRetryCancelledEvent{
        CommandID:  cmdID,
        RetryCount: state.RetryCount,
    })
    
    return nil
}

// GetRetryState gets retry state for a command
func (rm *RetryManager) GetRetryState(cmdID string) (*RetryState, bool) {
    state, ok := rm.retryTracker[cmdID]
    return state, ok
}

// DefaultRetryConfig returns default retry configuration
func DefaultRetryConfig() *RetryConfig {
    return &RetryConfig{
        MaxRetries:        3,
        InitialDelay:      5 * time.Second,
        MaxDelay:          5 * time.Minute,
        BackoffMultiplier: 2.0,
        RetryableStates: []CommandState{
            StateUplinkTimeout,
            StateUplinkFailed,
            StateVerifyTimeout,
        },
        RetryableCodes: []string{
            "TIMEOUT",
            "LINK_LOST",
            "NO_ACK",
            "TEMPORARY_FAILURE",
        },
    }
}

// CommandRetryScheduledEvent is emitted when retry is scheduled
type CommandRetryScheduledEvent struct {
    CommandID   string
    RetryCount  int
    NextRetryAt time.Time
    Delay       time.Duration
}

// CommandRetryExecutedEvent is emitted when retry is executed

}
```

```go
// security.go

package command

import (
    "context"

// CommandSecurity handles command security
type CommandSecurity struct {
    keyStore      KeyStore
    config        *SecurityConfig
    antiReplay    *AntiReplayProtection
}

// SecurityConfig configures command security
type SecurityConfig struct {
    EncryptionEnabled     bool
    AuthenticationEnabled bool
    Algorithm             string // AES-GCM, AES-CCM
    KeyID                 string
    AntiReplayWindow      time.Duration
}

// NewCommandSecurity creates a new security handler
func NewCommandSecurity(keyStore KeyStore, config *SecurityConfig) *CommandSecurity {
    return &CommandSecurity{
        keyStore:   keyStore,
        config:     config,
        antiReplay: NewAntiReplayProtection(config.AntiReplayWindow),
    }
}

// Encrypt encrypts a command packet
func (cs *CommandSecurity) Encrypt(ctx context.Context, plaintext []byte) ([]byte, error) {
    if !cs.config.EncryptionEnabled {
        return plaintext, nil
    }
    
    // Get encryption key
    key, err := cs.keyStore.GetKey(ctx, cs.config.KeyID)
    if err != nil {
        return nil, fmt.Errorf("failed to get key: %w", err)
    }
    
    // Create cipher
    block, err := aes.NewCipher(key.Data)
    if err != nil {
        return nil, fmt.Errorf("failed to create cipher: %w", err)
    }
    
    gcm, err := cipher.NewGCM(block)
    if err != nil {
        return nil, fmt.Errorf("failed to create GCM: %w", err)
    }
    
    // Generate nonce
    nonce := make([]byte, gcm.NonceSize())
    if _, err := io.ReadFull(rand.Reader, nonce); err != nil {
        return nil, fmt.Errorf("failed to generate nonce: %w", err)
    }
    
    // Encrypt
    ciphertext := gcm.Seal(nonce, nonce, plaintext, nil)
    
    // Prepend key ID
    result := make([]byte, 4+len(ciphertext))
    binary.BigEndian.PutUint32(result[:4], uint32(key.ID))
    copy(result[4:], ciphertext)
    
    return result, nil
}

// Decrypt decrypts a command packet
func (cs *CommandSecurity) Decrypt(ctx context.Context, ciphertext []byte) ([]byte, error) {
    if !cs.config.EncryptionEnabled {
        return ciphertext, nil
    }
    
    if len(ciphertext) < 4 {
        return nil, fmt.Errorf("ciphertext too short")
    }
    
    // Extract key ID
    keyID := binary.BigEndian.Uint32(ciphertext[:4])
    encData := ciphertext[4:]
    
    // Get decryption key
    key, err := cs.keyStore.GetKeyByID(ctx, int(keyID))
    if err != nil {
        return nil, fmt.Errorf("failed to get key: %w", err)
    }
    
    // Create cipher
    block, err := aes.NewCipher(key.Data)
    if err != nil {
        return nil, fmt.Errorf("failed to create cipher: %w", err)
    }
    
    gcm, err := cipher.NewGCM(block)
    if err != nil {
        return nil, fmt.Errorf("failed to create GCM: %w", err)
    }
    
    // Extract nonce
    nonceSize := gcm.NonceSize()
    if len(encData) < nonceSize {
        return nil, fmt.Errorf("ciphertext too short for nonce")
    }
    
    nonce, cipherData := encData[:nonceSize], encData[nonceSize:]
    
    // Decrypt
    plaintext, err := gcm.Open(nil, nonce, cipherData, nil)
    if err != nil {
        return nil, fmt.Errorf("decryption failed: %w", err)
    }
    
    return plaintext, nil
}

// Sign signs a command packet
func (cs *CommandSecurity) Sign(ctx context.Context, data []byte) ([]byte, error) {
    if !cs.config.AuthenticationEnabled {
        return nil, nil
    }
    
    // Get signing key
    key, err := cs.keyStore.GetKey(ctx, cs.config.KeyID)
    if err != nil {
        return nil, fmt.Errorf("failed to get key: %w", err)
    }
    
    // Compute HMAC-SHA256
    mac := hmac.New(sha256.New, key.Data)
    mac.Write(data)
    
    return mac.Sum(nil), nil
}

// Verify verifies a command signature
func (cs *CommandSecurity) Verify(ctx context.Context, data, signature []byte) (bool, error) {
    if !cs.config.AuthenticationEnabled {
        return true, nil
    }
    
    // Compute expected signature
    expected, err := cs.Sign(ctx, data)
    if err != nil {
        return false, err
    }
    
    // Compare
    return hmac.Equal(signature, expected), nil
}

// CheckAntiReplay checks for replay attacks
func (cs *CommandSecurity) CheckAntiReplay(seqNum uint16, timestamp time.Time) error {
    return cs.antiReplay.Check(seqNum, timestamp)
}

// AntiReplayProtection protects against replay attacks
type AntiReplayProtection struct {
    window    time.Duration
    seen      map[uint16]time.Time
    lastClean time.Time
    mu        sync.RWMutex
}

// NewAntiReplayProtection creates new anti-replay protection
func NewAntiReplayProtection(window time.Duration) *AntiReplayProtection {
    return &AntiReplayProtection{
        window:    window,
        seen:      make(map[uint16]time.Time),
        lastClean: time.Now(),
    }
}

// Check checks for replay
func (arp *AntiReplayProtection) Check(seqNum uint16, timestamp time.Time) error {
    arp.mu.Lock()
    defer arp.mu.Unlock()
    
    // Clean old entries periodically
    if time.Since(arp.lastClean) > arp.window {
        arp.clean()
    }
    
    // Check if seen
    if prevTime, ok := arp.seen[seqNum]; ok {
        if timestamp.Sub(prevTime) < arp.window {
            return fmt.Errorf("replay detected: sequence %d", seqNum)
        }
    }
    
    // Record
    arp.seen[seqNum] = timestamp
    
    return nil
}

// clean removes old entries
func (arp *AntiReplayProtection) clean() {
    cutoff := time.Now().Add(-arp.window)
    for seq, ts := range arp.seen {
        if ts.Before(cutoff) {
            delete(arp.seen, seq)
        }
    }
    arp.lastClean = time.Now()
}
```

```go
        LastUpdated:   cmd.UpdatedAt,
    }, nil
}

// GetActiveCommands gets all active commands
func (cm *CommandMonitor) GetActiveCommands(ctx context.Context, spacecraftID string) ([]*CommandStatus, error) {
    activeStates := []CommandState{
        StateQueued, StateScheduled, StateUplinking, StateSent, StateAcked, StateVerifying,
    }
    
    commands, err := cm.store.ListByStates(ctx, spacecraftID, activeStates)
    if err != nil {
        return nil, err
    }
    
    statuses := make([]*CommandStatus, len(commands))
    for i, cmd := range commands {
        statuses[i] = &CommandStatus{
            CommandID:    cmd.ID,
            SpacecraftID: cmd.SpacecraftID,
            State:        cmd.State,
            UplinkTime:   cmd.UplinkTime,
            AckTime:      cmd.AckTime,
            RetryCount:   cmd.RetryCount,
            LastUpdated:  cmd.UpdatedAt,
        }
    }
    
    return statuses, nil
}
```

```go
// NewAuditLogger creates a new audit logger
func NewAuditLogger(store AuditStore, eventBus EventBus) *AuditLogger {
    al := &AuditLogger{
        store:    store,
        eventBus: eventBus,
    }
    
    // Subscribe to command events
    go al.handleEvents()
    
    return al

// handleEvents processes events for auditing
func (al *AuditLogger) handleEvents() {
    eventChan := al.eventBus.Subscribe("command.*")
    
    for event := range eventChan {
        switch e := event.(type) {
        case *CommandStateChangedEvent:
            al.logStateChange(context.Background(), e)
        case *CommandUplinkEvent:
            al.logUplink(context.Background(), e)
        case *CommandRetryScheduledEvent:
            al.logRetry(context.Background(), e)
        }
    }
}

// LogCommandCreated logs command creation
func (al *AuditLogger) LogCommandCreated(ctx context.Context, cmd *Command, userID string) error {
    event := &AuditEvent{
        ID:           generateID(),
        Timestamp:    time.Now(),
        EventType:    "COMMAND_CREATED",
        CommandID:    cmd.ID,
        SpacecraftID: cmd.SpacecraftID,
        UserID:       userID,
        Action:       "create",
        Details: map[string]interface{}{
            "type":       cmd.Type,
            "priority":   cmd.Priority,
            "parameters": cmd.Parameters,
        },
        NewState: cmd.State,
        TenantID: cmd.TenantID,
    }
    
    return al.store.Log(ctx, event)
}

// logStateChange logs state change
func (al *AuditLogger) logStateChange(ctx context.Context, e *CommandStateChangedEvent) {
    event := &AuditEvent{
        ID:            generateID(),
        Timestamp:     e.Timestamp,
        EventType:     "STATE_CHANGED",
        CommandID:     e.CommandID,
        Action:        string(e.Event),
        PreviousState: e.PreviousState,
        NewState:      e.CurrentState,
    }
    
    al.store.Log(ctx, event)
}

// logUplink logs uplink event
func (al *AuditLogger) logUplink(ctx context.Context, e *CommandUplinkEvent) {
    event := &AuditEvent{
        ID:        generateID(),
        Timestamp: time.Now(),
        EventType: "COMMAND_UPLINKED",
        CommandID: e.CommandID,
        Action:    "uplink",
        Details: map[string]interface{}{
            "confirmed":    e.Result.Confirmed,
            "transmit_time": e.Result.TransmitTime,
            "retransmits":  e.Result.Retransmits,
        },
    }
    
    al.store.Log(ctx, event)
}

// GetCommandHistory retrieves command audit history
func (al *AuditLogger) GetCommandHistory(ctx context.Context, cmdID string) ([]*AuditEvent, error) {
    return al.store.GetByCommandID(ctx, cmdID)
}

// GenerateComplianceReport generates compliance report
func (al *AuditLogger) GenerateComplianceReport(ctx context.Context, start, end time.Time, spacecraftID string) (*ComplianceReport, error) {
    events, err := al.store.GetByTimeRange(ctx, start, end, spacecraftID)
    if err != nil {
        return nil, err
    }
    
    report := &ComplianceReport{
        SpacecraftID:  spacecraftID,
        StartTime:     start,
        EndTime:       end,
        GeneratedAt:   time.Now(),
        TotalCommands: 0,
        ByState:       make(map[CommandState]int),
        ByUser:        make(map[string]int),
        ByType:        make(map[CommandType]int),
        Events:        events,
    }
    
    // Analyze events
    commandsSeen := make(map[string]bool)
    for _, event := range events {
        if event.EventType == "COMMAND_CREATED" {
            commandsSeen[event.CommandID] = true
            report.ByUser[event.UserID]++
            if cmdType, ok := event.Details["type"].(CommandType); ok {
                report.ByType[cmdType]++
            }
        }
        if event.NewState != "" {
            report.ByState[event.NewState]++
        }
    }
    report.TotalCommands = len(commandsSeen)
    
    return report, nil
}

// ComplianceReport contains compliance information
type ComplianceReport struct {
    SpacecraftID  string
    StartTime     time.Time
    EndTime       time.Time
    GeneratedAt   time.Time
    TotalCommands int
    ByState       map[CommandState]int
    ByUser        map[string]int
    ByType        map[CommandType]int
    Events        []*AuditEvent
}

// AuditStore interface for audit storage
type AuditStore interface {
    Log(ctx context.Context, event *AuditEvent) error
    GetByCommandID(ctx context.Context, cmdID string) ([]*AuditEvent, error)
    GetByTimeRange(ctx context.Context, start, end time.Time, spacecraftID string) ([]*AuditEvent, error)
}
```
