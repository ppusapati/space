## 1. Executive Summary

### 1.1 Purpose

The Change Detection Pipeline provides automated detection and analysis of changes in Earth Observation imagery across time:

- **Multi-Algorithm Support:** Pixel-based, object-based, and deep learning methods
- **Temporal Flexibility:** Bi-temporal and multi-temporal time series analysis
- **Scalable Processing:** Distributed pipeline for large-area monitoring
- **Rich Attribution:** Change type classification and confidence scoring
- **STAC Integration:** Seamless catalog search and result publishing

### 1.2 Key Metrics

| Metric | Target | Description |
|--------|--------|-------------|
| **Detection Accuracy** | > 90% F1 | On validation dataset |
| **Processing Time** | < 5 min | Per Sentinel-2 tile pair |
| **False Positive Rate** | < 5% | Spurious detections |
| **Spatial Resolution** | 10m | Minimum detectable |
| **Temporal Latency** | < 24h | Acquisition to detection |
| **Area Coverage** | 1M km²/day | Processing capacity |

### 3.1 Co-Registration Implementation

```go
// coregistration.go

package changedetection

import (
    "context"
    "math"
)

// CoRegistration handles image alignment
type CoRegistration struct {
    config *CoRegConfig
}

// CoRegConfig configures co-registration
type CoRegConfig struct {
    MaxShiftPixels    int     // Maximum allowed shift
    MinMatchPoints    int     // Minimum tie points
    ResamplingMethod  string  // bilinear, cubic, nearest
    FeatureMethod     string  // sift, orb
    RANSACThreshold   float64 // Outlier threshold
}

// ImagePair represents a temporal image pair
type ImagePair struct {
    Reference   *Image
    Target      *Image
    Transform   *AffineTransform
    AlignedPair *AlignedImagePair
    Quality     *CoRegQuality
}

// Image represents a satellite image
type Image struct {
    ID           string
    Path         string
    Width        int
    Height       int
    GeoTransform []float64
    Projection   string
    DateTime     time.Time
    CloudMask    []byte
    Data         [][]float32  // [band][pixels]
}

// CoRegQuality contains quality metrics
type CoRegQuality struct {
    NumMatchPoints int
    RMSE           float64
    MaxResidual    float64
    ShiftX         float64
    ShiftY         float64
}

// Process performs co-registration
func (cr *CoRegistration) Process(ctx context.Context, pair *ImagePair) error {
    // 1. Extract features from both images
    refFeatures := cr.detectFeatures(pair.Reference)
    tgtFeatures := cr.detectFeatures(pair.Target)
    
    // 2. Match features
    matches := cr.matchFeatures(refFeatures, tgtFeatures)
    
    if len(matches) < cr.config.MinMatchPoints {
        return fmt.Errorf("insufficient matches: %d", len(matches))
    }
    
    // 3. Estimate transformation using RANSAC
    transform, inliers := cr.estimateTransform(matches)
    pair.Transform = transform
    
    // 4. Assess quality
    pair.Quality = cr.assessQuality(matches, inliers, transform)
    
    if pair.Quality.RMSE > cr.config.RANSACThreshold {
        return fmt.Errorf("RMSE too high: %.2f", pair.Quality.RMSE)
    }
    
    // 5. Apply transformation
    alignedTarget := cr.applyTransform(pair.Target, transform)
    
    // 6. Create aligned pair
    pair.AlignedPair = &AlignedImagePair{
        Reference: pair.Reference,
        Target:    alignedTarget,
        ValidMask: cr.computeValidMask(pair.Reference, alignedTarget),
    }
    
    return nil
}

// AffineTransform represents 2D affine transformation
type AffineTransform struct {
    A, B, C, D, E, F float64
}

// Apply applies transform to a point
func (t *AffineTransform) Apply(x, y float64) (float64, float64) {
    return t.A*x + t.B*y + t.E, t.C*x + t.D*y + t.F
}
```

### 3.2 Radiometric Normalization

```go
// radiometric.go

// RadiometricNormalizer handles normalization between images
type RadiometricNormalizer struct {
    config *RadiometricConfig
}

// RadiometricConfig configures normalization
type RadiometricConfig struct {
    Method       string  // histogram, pif, relative, mad
    PIFThreshold float64 // Pseudo-invariant feature threshold
    UseCloudMask bool
}

// Normalize normalizes target to reference
func (rn *RadiometricNormalizer) Normalize(ctx context.Context, pair *AlignedImagePair) error {
    switch rn.config.Method {
    case "histogram":
        return rn.histogramMatch(pair)
    case "pif":
        return rn.pifNormalize(pair)
    case "relative":
        return rn.relativeNormalize(pair)
    default:
        return rn.relativeNormalize(pair)
    }
}

// pifNormalize uses Pseudo-Invariant Features
func (rn *RadiometricNormalizer) pifNormalize(pair *AlignedImagePair) error {
    numBands := len(pair.Reference.Data)
    numPixels := len(pair.Reference.Data[0])
    
    // 1. Identify PIFs (unchanged pixels)
    pifMask := rn.identifyPIFs(pair)
    
    // 2. For each band, compute linear regression on PIFs
    for band := 0; band < numBands; band++ {
        refData := pair.Reference.Data[band]
        tgtData := pair.Target.Data[band]
        
        // Extract PIF values
        var refPIF, tgtPIF []float64
        for i := 0; i < numPixels; i++ {
            if pifMask[i] && pair.ValidMask[i] != 0 {
                refPIF = append(refPIF, float64(refData[i]))
                tgtPIF = append(tgtPIF, float64(tgtData[i]))
            }
        }
        
        // Compute regression: ref = slope * tgt + intercept
        slope, intercept := rn.linearRegression(tgtPIF, refPIF)
        
        // Apply to all target pixels
        for i := 0; i < numPixels; i++ {
            pair.Target.Data[band][i] = float32(slope*float64(tgtData[i]) + intercept)
        }
    }
    
    return nil
}

// identifyPIFs finds pseudo-invariant features
func (rn *RadiometricNormalizer) identifyPIFs(pair *AlignedImagePair) []bool {
    numPixels := len(pair.Reference.Data[0])
    numBands := len(pair.Reference.Data)
    pifMask := make([]bool, numPixels)
    
    for i := 0; i < numPixels; i++ {
        if pair.ValidMask[i] == 0 {
            continue
        }
        
        // Compute spectral difference
        var sumDiff, sumMag float64
        for b := 0; b < numBands; b++ {
            ref := float64(pair.Reference.Data[b][i])
            tgt := float64(pair.Target.Data[b][i])
            diff := ref - tgt
            sumDiff += diff * diff
            sumMag += ref*ref + tgt*tgt
        }
        
        if sumMag > 0 {
            normDiff := math.Sqrt(sumDiff / sumMag)
            pifMask[i] = normDiff < rn.config.PIFThreshold
        }
    }
    
    return pifMask
}
```

### 3.3 Cloud Masking

```go
// cloudmask.go

// CloudMaskProcessor handles cloud and shadow detection
type CloudMaskProcessor struct {
    config *CloudMaskConfig
}

// CloudMaskConfig configures cloud masking
type CloudMaskConfig struct {
    UseProvidedMask  bool
    CloudThreshold   float64
    ShadowThreshold  float64
    BufferPixels     int
    MinCloudSize     int
}

// MaskResult contains masking results
type MaskResult struct {
    CloudMask    []byte
    ShadowMask   []byte
    ValidMask    []byte
    CloudPercent float64
}

// Process generates cloud and shadow masks
func (cmp *CloudMaskProcessor) Process(ctx context.Context, image *Image) (*MaskResult, error) {
    numPixels := image.Width * image.Height
    
    result := &MaskResult{
        CloudMask:  make([]byte, numPixels),
        ShadowMask: make([]byte, numPixels),
        ValidMask:  make([]byte, numPixels),
    }
    
    // Initialize valid mask
    for i := range result.ValidMask {
        result.ValidMask[i] = 1
    }
    
    // Use provided SCL if available
    if cmp.config.UseProvidedMask && image.CloudMask != nil {
        result.CloudMask = image.CloudMask
    } else {
        cmp.detectClouds(image, result)
    }
    
    // Detect shadows
    cmp.detectShadows(image, result)
    
    // Buffer clouds
    if cmp.config.BufferPixels > 0 {
        result.CloudMask = cmp.bufferMask(result.CloudMask, image.Width, image.Height)
    }
    
    // Update valid mask
    for i := 0; i < numPixels; i++ {
        if result.CloudMask[i] != 0 || result.ShadowMask[i] != 0 {
            result.ValidMask[i] = 0
        }
    }
    
    // Calculate cloud percentage
    cloudCount := 0
    for _, v := range result.CloudMask {
        if v != 0 {
            cloudCount++
        }
    }
    result.CloudPercent = float64(cloudCount) / float64(numPixels) * 100
    
    return result, nil
}

// detectClouds detects clouds using spectral tests
func (cmp *CloudMaskProcessor) detectClouds(image *Image, result *MaskResult) {
    for i := 0; i < len(image.Data[0]); i++ {
        // Assuming: B02=Blue, B03=Green, B04=Red, B08=NIR
        blue := float64(image.Data[0][i]) / 10000.0
        green := float64(image.Data[1][i]) / 10000.0
        red := float64(image.Data[2][i]) / 10000.0
        nir := float64(image.Data[3][i]) / 10000.0
        
        // High brightness test
        brightness := (blue + green + red) / 3.0
        
        // Low NDVI test
        ndvi := 0.0
        if nir+red > 0 {
            ndvi = (nir - red) / (nir + red)
        }
        
        // Cloud detection
        if brightness > cmp.config.CloudThreshold && ndvi < 0.1 {
            result.CloudMask[i] = 1
        }
    }
}
```

### 4.1 Pixel-Based Change Detection

```go
// pixel_change.go

// PixelChangeDetector implements pixel-based methods
type PixelChangeDetector struct {
    config *PixelChangeConfig
}

// PixelChangeConfig configures detection
type PixelChangeConfig struct {
    Method            string  // cva, diff, ratio, mad
    ChangeThreshold   float64
    AdaptiveThreshold bool
}

// ChangeMap represents detected changes
type ChangeMap struct {
    Width           int
    Height          int
    ChangeMask      []byte    // Binary: 1 = change
    ChangeMagnitude []float32 // Magnitude of change
    ChangeDirection []float32 // Direction (for CVA)
    ChangeType      []int     // Classification
    Threshold       float64
    NumChanges      int
    ChangePercent   float64
}

// Detect performs pixel-based change detection
func (pcd *PixelChangeDetector) Detect(ctx context.Context, pair *AlignedImagePair) (*ChangeMap, error) {
    switch pcd.config.Method {
    case "cva":
        return pcd.changeVectorAnalysis(ctx, pair)
    case "diff":
        return pcd.imageDifferencing(ctx, pair)
    case "ratio":
        return pcd.imageRatioing(ctx, pair)
    case "mad":
        return pcd.multivariateAlterationDetection(ctx, pair)
    default:
        return pcd.changeVectorAnalysis(ctx, pair)
    }
}

// changeVectorAnalysis implements CVA
func (pcd *PixelChangeDetector) changeVectorAnalysis(ctx context.Context, pair *AlignedImagePair) (*ChangeMap, error) {
    numBands := len(pair.Reference.Data)
    numPixels := len(pair.Reference.Data[0])
    
    changeMap := &ChangeMap{
        Width:           pair.Reference.Width,
        Height:          pair.Reference.Height,
        ChangeMask:      make([]byte, numPixels),
        ChangeMagnitude: make([]float32, numPixels),
        ChangeDirection: make([]float32, numPixels),
    }
    
    // Calculate change vector for each pixel
    for i := 0; i < numPixels; i++ {
        if pair.ValidMask[i] == 0 {
            continue
        }
        
        var sumSq float64
        for b := 0; b < numBands; b++ {
            diff := float64(pair.Target.Data[b][i]) - float64(pair.Reference.Data[b][i])
            sumSq += diff * diff
        }
        
        // Magnitude is Euclidean distance
        changeMap.ChangeMagnitude[i] = float32(math.Sqrt(sumSq))
    }
    
    // Determine threshold
    threshold := pcd.config.ChangeThreshold
    if pcd.config.AdaptiveThreshold {
        threshold = pcd.computeAdaptiveThreshold(changeMap.ChangeMagnitude, pair.ValidMask)
    }
    changeMap.Threshold = threshold
    
    // Apply threshold
    changeCount := 0
    for i := 0; i < numPixels; i++ {
        if pair.ValidMask[i] != 0 && float64(changeMap.ChangeMagnitude[i]) > threshold {
            changeMap.ChangeMask[i] = 1
            changeCount++
        }
    }
    
    changeMap.NumChanges = changeCount
    changeMap.ChangePercent = float64(changeCount) / float64(numPixels) * 100
    
    return changeMap, nil
}

// imageDifferencing implements simple differencing
func (pcd *PixelChangeDetector) imageDifferencing(ctx context.Context, pair *AlignedImagePair) (*ChangeMap, error) {
    numPixels := len(pair.Reference.Data[0])
    
    changeMap := &ChangeMap{
        Width:           pair.Reference.Width,
        Height:          pair.Reference.Height,
        ChangeMask:      make([]byte, numPixels),
        ChangeMagnitude: make([]float32, numPixels),
        ChangeType:      make([]int, numPixels),
    }
    
    // Use NDVI differencing
    refNDVI := pcd.computeNDVI(pair.Reference.Data[2], pair.Reference.Data[3])
    tgtNDVI := pcd.computeNDVI(pair.Target.Data[2], pair.Target.Data[3])
    
    for i := 0; i < numPixels; i++ {
        if pair.ValidMask[i] == 0 {
            continue
        }
        
        diff := tgtNDVI[i] - refNDVI[i]
        changeMap.ChangeMagnitude[i] = float32(math.Abs(diff))
        
        if diff > 0 {
            changeMap.ChangeType[i] = 1  // Gain (greening)
        } else if diff < 0 {
            changeMap.ChangeType[i] = -1 // Loss (browning)
        }
    }
    
    // Apply threshold
    threshold := pcd.config.ChangeThreshold
    changeCount := 0
    for i := 0; i < numPixels; i++ {
        if pair.ValidMask[i] != 0 && float64(changeMap.ChangeMagnitude[i]) > threshold {
            changeMap.ChangeMask[i] = 1
            changeCount++
        }
    }
    
    changeMap.NumChanges = changeCount
    changeMap.ChangePercent = float64(changeCount) / float64(numPixels) * 100
    
    return changeMap, nil
}

// computeNDVI computes NDVI
func (pcd *PixelChangeDetector) computeNDVI(red, nir []float32) []float64 {
    ndvi := make([]float64, len(red))
    for i := range red {
        r := float64(red[i])
        n := float64(nir[i])
        if n+r > 0 {
            ndvi[i] = (n - r) / (n + r)
        }
    }
    return ndvi
}

// computeAdaptiveThreshold uses Otsu's method
func (pcd *PixelChangeDetector) computeAdaptiveThreshold(magnitude []float32, validMask []byte) float64 {
    var values []float64
    for i, v := range magnitude {
        if validMask[i] != 0 && v > 0 {
            values = append(values, float64(v))
        }
    }
    
    if len(values) == 0 {
        return pcd.config.ChangeThreshold
    }
    
    sort.Float64s(values)
    
    // Use 95th percentile
    idx := int(float64(len(values)) * 0.95)
    return values[idx]
}
```

```go
// object_change.go

// ObjectChangeDetector implements object-based detection
type ObjectChangeDetector struct {
    config    *ObjectChangeConfig
    segmenter *ImageSegmenter
}

// ObjectChangeConfig configures object-based detection
type ObjectChangeConfig struct {
    SegmentationScale float64
    MinSegmentSize    int
    ChangeThreshold   float64
    ComparisonMethod  string // spectral, shape, combined
}

// Segment represents an image segment
type Segment struct {
    ID           int
    Pixels       []int
    Centroid     [2]float64
    Area         int
    MeanSpectral []float64
    StdSpectral  []float64
}

// ObjectChange represents changed object
type ObjectChange struct {
    RefSegment   *Segment
    TgtSegment   *Segment
    ChangeScore  float64
    ChangeType   string
    SpectralDiff float64
}

// Detect performs object-based detection
func (ocd *ObjectChangeDetector) Detect(ctx context.Context, pair *AlignedImagePair) (*ObjectChangeMap, error) {
    // 1. Segment both images
    refSegments, refLabels := ocd.segmenter.Segment(pair.Reference)
    tgtSegments, tgtLabels := ocd.segmenter.Segment(pair.Target)
    
    // 2. Compute segment features
    ocd.computeFeatures(pair.Reference, refSegments)
    ocd.computeFeatures(pair.Target, tgtSegments)
    
    // 3. Match segments
    matches := ocd.matchSegments(refSegments, tgtSegments, refLabels, tgtLabels)
    
    // 4. Compare matched segments
    changes := make([]*ObjectChange, 0)
    for _, match := range matches {
        change := ocd.compareSegments(match.RefSegment, match.TgtSegment)
        if change.ChangeScore > ocd.config.ChangeThreshold {
            changes = append(changes, change)
        }
    }
    
    return &ObjectChangeMap{
        RefSegments: refSegments,
        TgtSegments: tgtSegments,
        Changes:     changes,
    }, nil
}

// compareSegments compares two matched segments
func (ocd *ObjectChangeDetector) compareSegments(ref, tgt *Segment) *ObjectChange {
    change := &ObjectChange{
        RefSegment: ref,
        TgtSegment: tgt,
    }
    
    // Spectral difference
    var spectralDist float64
    for b := 0; b < len(ref.MeanSpectral); b++ {
        diff := ref.MeanSpectral[b] - tgt.MeanSpectral[b]
        spectralDist += diff * diff
    }
    change.SpectralDiff = math.Sqrt(spectralDist)
    change.ChangeScore = change.SpectralDiff / 1000.0
    
    // Classify change type
    refNDVI := (ref.MeanSpectral[3] - ref.MeanSpectral[2]) / (ref.MeanSpectral[3] + ref.MeanSpectral[2] + 1)
    tgtNDVI := (tgt.MeanSpectral[3] - tgt.MeanSpectral[2]) / (tgt.MeanSpectral[3] + tgt.MeanSpectral[2] + 1)
    
    if tgtNDVI-refNDVI > 0.15 {
        change.ChangeType = "vegetation_increase"
    } else if tgtNDVI-refNDVI < -0.15 {
        change.ChangeType = "vegetation_decrease"
    } else {
        change.ChangeType = "modification"
    }
    
    return change
}
```

## 5. Deep Learning Models

### 5.1 DL Change Detector

```go
// dl_change.go

// DLChangeDetector implements deep learning change detection
type DLChangeDetector struct {
    config    *DLChangeConfig
    model     InferenceClient
    processor *TileProcessor
}

// DLChangeConfig configures DL detection
type DLChangeConfig struct {
    ModelID      string
    ModelVersion string
    TileSize     int
    TileOverlap  int
    BatchSize    int
    Threshold    float64
    UseGPU       bool
}

// Supported DL architectures
// - Siamese CNN: Twin networks with shared weights
// - FC-EF: Fully Convolutional Early Fusion
// - FC-Siam-Diff: Siamese with difference features
// - BIT: Bitemporal Image Transformer
// - ChangeFormer: Transformer-based architecture

// Detect performs DL-based change detection
func (dld *DLChangeDetector) Detect(ctx context.Context, pair *AlignedImagePair) (*ChangeMap, error) {
    // 1. Create tiles from both images
    tiles := dld.processor.CreateBitemporalTiles(pair)
    
    changeMap := &ChangeMap{
        Width:           pair.Reference.Width,
        Height:          pair.Reference.Height,
        ChangeMask:      make([]byte, pair.Reference.Width*pair.Reference.Height),
        ChangeMagnitude: make([]float32, pair.Reference.Width*pair.Reference.Height),
    }
    
    // 2. Process tiles in batches
    for start := 0; start < len(tiles); start += dld.config.BatchSize {
        end := start + dld.config.BatchSize
        if end > len(tiles) {
            end = len(tiles)
        }
        
        batch := tiles[start:end]
        
        // 3. Prepare input tensors
        inputs := dld.prepareBatchInput(batch)
        
        // 4. Run inference
        outputs, err := dld.model.Infer(ctx, inputs)
        if err != nil {
            return nil, err
        }
        
        // 5. Process outputs
        dld.processBatchOutput(batch, outputs, changeMap)
    }
    
    // 6. Apply threshold
    changeCount := 0
    for i, mag := range changeMap.ChangeMagnitude {
        if float64(mag) > dld.config.Threshold {
            changeMap.ChangeMask[i] = 1
            changeCount++
        }
    }
    
    changeMap.NumChanges = changeCount
    changeMap.ChangePercent = float64(changeCount) / float64(len(changeMap.ChangeMask)) * 100
    
    return changeMap, nil
}

// BitemporalTile represents a tile pair
type BitemporalTile struct {
    X, Y    int
    Width   int
    Height  int
    RefData []float32
    TgtData []float32
}

// prepareBatchInput prepares input tensors
func (dld *DLChangeDetector) prepareBatchInput(tiles []*BitemporalTile) []*Tensor {
    batchSize := len(tiles)
    numBands := len(tiles[0].RefData) / (tiles[0].Width * tiles[0].Height)
    tileSize := dld.config.TileSize
    
    // Concatenate: [batch, 2*bands, height, width]
    inputSize := batchSize * 2 * numBands * tileSize * tileSize
    inputData := make([]float32, inputSize)
    
    for b, tile := range tiles {
        offset := b * 2 * numBands * tileSize * tileSize
        copy(inputData[offset:], tile.RefData)
        copy(inputData[offset+len(tile.RefData):], tile.TgtData)
    }
    
    return []*Tensor{{
        Name:        "input",
        Shape:       []int64{int64(batchSize), int64(2 * numBands), int64(tileSize), int64(tileSize)},
        DataType:    "FP32",
        Float32Data: inputData,
    }}
}
```

```go
// MultiTemporalAnalyzer handles time series analysis
type MultiTemporalAnalyzer struct {
    config *MultiTemporalConfig
}

// MultiTemporalConfig configures analysis
type MultiTemporalConfig struct {
    MinObservations  int
    SeasonalAdjust   bool
    OutlierThreshold float64
    TrendMethod      string // linear, mann_kendall
    BreakpointMethod string // cusum, pelt
    ConfidenceLevel  float64
}

// TimeSeriesChange represents change in time series
type TimeSeriesChange struct {
    X, Y           int
    ChangeDate     time.Time
    Magnitude      float64
    Direction      int
    Confidence     float64
    PreChangeMean  float64
    PostChangeMean float64
}

// Analyze performs multi-temporal analysis
func (mta *MultiTemporalAnalyzer) Analyze(ctx context.Context, timeSeries []*Image) (*MultiTemporalResult, error) {
    if len(timeSeries) < mta.config.MinObservations {
        return nil, fmt.Errorf("insufficient observations")
    }
    
    // Sort by date
    sort.Slice(timeSeries, func(i, j int) bool {
        return timeSeries[i].DateTime.Before(timeSeries[j].DateTime)
    })
    
    width := timeSeries[0].Width
    height := timeSeries[0].Height
    
    result := &MultiTemporalResult{
        Width:    width,
        Height:   height,
        Changes:  make([]*TimeSeriesChange, 0),
        TrendMap: make([]float64, width*height),
    }
    
    // Process each pixel
    for y := 0; y < height; y++ {
        for x := 0; x < width; x++ {
            idx := y*width + x
            
            // Extract time series
            pixelTS := mta.extractPixelTimeSeries(timeSeries, x, y)
            
            if pixelTS.ValidCount < mta.config.MinObservations {
                continue
            }
            
            // Remove seasonality
            if mta.config.SeasonalAdjust {
                mta.removeSeasonality(pixelTS)
            }
            
            // Detect trend
            trend := mta.detectTrend(pixelTS)
            result.TrendMap[idx] = trend.Slope
            
            // Detect breakpoints
            breakpoints := mta.detectBreakpoints(pixelTS)
            for _, bp := range breakpoints {
                result.Changes = append(result.Changes, &TimeSeriesChange{
                    X:              x,
                    Y:              y,
                    ChangeDate:     bp.Date,
                    Magnitude:      bp.Magnitude,
                    Direction:      bp.Direction,
                    Confidence:     bp.Confidence,
                    PreChangeMean:  bp.PreMean,
                    PostChangeMean: bp.PostMean,
                })
            }
        }
    }
    
    return result, nil
}

// detectBreakpoints using CUSUM
func (mta *MultiTemporalAnalyzer) detectBreakpoints(ts *TimeSeriesPixel) []*Breakpoint {
    ndvi := mta.computeNDVITimeSeries(ts)
    
    // Collect valid observations
    var values []float64
    var indices []int
    for t := 0; t < len(ts.Dates); t++ {
        if !ts.CloudMask[t] && !math.IsNaN(ndvi[t]) {
            values = append(values, ndvi[t])
            indices = append(indices, t)
        }
    }
    
    n := len(values)
    if n < mta.config.MinObservations {
        return nil
    }
    
    // Compute mean
    mean := 0.0
    for _, v := range values {
        mean += v
    }
    mean /= float64(n)
    
    // Compute CUSUM
    cusum := make([]float64, n)
    cusum[0] = values[0] - mean
    for i := 1; i < n; i++ {
        cusum[i] = cusum[i-1] + (values[i] - mean)
    }
    
    // Find maximum deviation
    maxDev := 0.0
    maxIdx := 0
    for i, c := range cusum {
        if math.Abs(c) > maxDev {
            maxDev = math.Abs(c)
            maxIdx = i
        }
    }
    
    // Test significance
    stdDev := 0.0
    for _, v := range values {
        d := v - mean
        stdDev += d * d
    }
    stdDev = math.Sqrt(stdDev / float64(n-1))
    
    threshold := 1.36 * stdDev * math.Sqrt(float64(n)) // 95% confidence
    
    if maxDev > threshold && maxIdx > 0 && maxIdx < n-1 {
        // Compute pre/post means
        preMean := 0.0
        for i := 0; i <= maxIdx; i++ {
            preMean += values[i]
        }
        preMean /= float64(maxIdx + 1)
        
        postMean := 0.0
        for i := maxIdx + 1; i < n; i++ {
            postMean += values[i]
        }
        postMean /= float64(n - maxIdx - 1)
        
        direction := 1
        if postMean < preMean {
            direction = -1
        }
        
        return []*Breakpoint{{
            Date:       ts.Dates[indices[maxIdx]],
            Magnitude:  math.Abs(postMean - preMean),
            Direction:  direction,
            Confidence: maxDev / threshold,
            PreMean:    preMean,
            PostMean:   postMean,
        }}
    }
    
    return nil
}
```

```go
                classified[i] = 1 // Moderate loss
            }
        } else {
            classified[i] = -1
```

```go
// Extract extracts vector polygons
func (ve *VectorExtractor) Extract(ctx context.Context, changeMap *ChangeMap, geoTransform []float64) (*ChangeFeatureCollection, error) {
    // 1. Connected component labeling
    labels, numComponents := ve.labelComponents(changeMap)
    
    features := make([]*ChangeFeature, 0)
    
    // 2. Extract contours for each component
    for compID := 1; compID <= numComponents; compID++ {
        compMask := ve.getComponentMask(labels, compID, changeMap.Width, changeMap.Height)
        
        // Check minimum area
        area := ve.countPixels(compMask)
        if area < ve.config.MinAreaPixels {
            continue
        }
        
        // Extract contour
        contour := ve.extractContour(compMask, changeMap.Width, changeMap.Height)
        if len(contour) < 4 {
            continue
        }
        
        // Convert to geo coordinates
        geoContour := ve.pixelToGeo(contour, geoTransform)
        
        // Simplify
        if ve.config.SimplifyTolerance > 0 {
            geoContour = ve.douglasPeucker(geoContour, ve.config.SimplifyTolerance)
        }
        
        // Compute attributes
        props := ve.computeAttributes(changeMap, labels, compID, area, geoTransform)
        
        features = append(features, &ChangeFeature{
            Type: "Feature",
            ID:   fmt.Sprintf("change_%d", compID),
            Geometry: &GeoJSONGeometry{
                Type:        "Polygon",
                Coordinates: [][][]float64{geoContour},
            },
            Properties: props,
        })
    }
    
    return &ChangeFeatureCollection{
        Type:     "FeatureCollection",
        Features: features,
    }, nil
}

// labelComponents performs connected component labeling
func (ve *VectorExtractor) labelComponents(changeMap *ChangeMap) ([]int, int) {
    width := changeMap.Width
    height := changeMap.Height
    labels := make([]int, width*height)
    nextLabel := 1
    
    for y := 0; y < height; y++ {
        for x := 0; x < width; x++ {
            idx := y*width + x
            
            if changeMap.ChangeMask[idx] == 0 {
                continue
            }
            
            // Check neighbors
            neighbors := make([]int, 0)
            if x > 0 && labels[idx-1] > 0 {
                neighbors = append(neighbors, labels[idx-1])
            }
            if y > 0 && labels[idx-width] > 0 {
                neighbors = append(neighbors, labels[idx-width])
            }
            
            if len(neighbors) == 0 {
                labels[idx] = nextLabel
                nextLabel++
            } else {
                minLabel := neighbors[0]
                for _, l := range neighbors {
                    if l < minLabel {
                        minLabel = l
                    }
                }
                labels[idx] = minLabel
            }
        }
    }
    
    return labels, nextLabel - 1
}

// computeAttributes computes polygon attributes
func (ve *VectorExtractor) computeAttributes(changeMap *ChangeMap, labels []int, compID, areaPixels int, geoTransform []float64) map[string]interface{} {
    props := make(map[string]interface{})
    
    props["area_pixels"] = areaPixels
    
    // Area in square meters
    pixelAreaSqM := math.Abs(geoTransform[1] * geoTransform[5])
    props["area_sqm"] = float64(areaPixels) * pixelAreaSqM
    props["area_ha"] = float64(areaPixels) * pixelAreaSqM / 10000
    
    // Mean magnitude
    var sumMag float64
    count := 0
    for i, l := range labels {
        if l == compID {
            sumMag += float64(changeMap.ChangeMagnitude[i])
            count++
        }
    }
    if count > 0 {
        props["mean_magnitude"] = sumMag / float64(count)
    }
    
    return props
}
```

```go
// PipelineConfig configures the pipeline
type PipelineConfig struct {
    DetectionMethod    string  // pixel, object, dl, ensemble
    PreprocessingLevel string  // basic, full
    MaxCloudPercent    float64
    MinTemporalGap     time.Duration
    MaxTemporalGap     time.Duration
}

// ChangeDetectionJob represents a job
type ChangeDetectionJob struct {
    ID         string
    AOI        *GeoJSON
    DateRange  [2]time.Time
    Collections []string
    Config     *PipelineConfig
    Status     string
    Progress   float64
    Results    []*ChangeDetectionResult
    Error      string
}

// Execute runs the change detection pipeline
func (cdp *ChangeDetectionPipeline) Execute(ctx context.Context, job *ChangeDetectionJob) error {
    job.Status = "running"
    
    // 1. Query STAC for images
    images, err := cdp.stacClient.QueryImages(ctx, job.AOI, job.DateRange, job.Collections)
    if err != nil {
        return cdp.failJob(job, err)
    }
    
    if len(images) < 2 {
        return cdp.failJob(job, fmt.Errorf("insufficient images"))
    }
    
    // 2. Select image pairs
    pairs := cdp.selectImagePairs(images, job.Config)
    
    // 3. Process each pair
    results := make([]*ChangeDetectionResult, 0)
    for i, pair := range pairs {
        job.Progress = float64(i) / float64(len(pairs))
        
        result, err := cdp.processPair(ctx, pair, job.Config)
        if err != nil {
            continue
        }
        results = append(results, result)
    }
    
    // 4. Publish results
    for _, result := range results {
        cdp.stacPublisher.Publish(ctx, result)
    }
    
    job.Status = "completed"
    job.Results = results
    job.Progress = 1.0
    
    return nil
}

// processPair processes a single image pair
func (cdp *ChangeDetectionPipeline) processPair(ctx context.Context, pair *ImagePairInfo, config *PipelineConfig) (*ChangeDetectionResult, error) {
    result := &ChangeDetectionResult{
        ID:             generateID(),
        ReferenceImage: pair.Reference.ID,
        TargetImage:    pair.Target.ID,
    }
    
    // 1. Load images
    refImage, _ := cdp.imageLoader.Load(ctx, pair.Reference)
    tgtImage, _ := cdp.imageLoader.Load(ctx, pair.Target)
    
    // 2. Cloud masking
    refMask, _ := cdp.cloudMask.Process(ctx, refImage)
    tgtMask, _ := cdp.cloudMask.Process(ctx, tgtImage)
    
    refImage.CloudMask = refMask.CloudMask
    tgtImage.CloudMask = tgtMask.CloudMask
    
    if refMask.CloudPercent > config.MaxCloudPercent {
        return nil, fmt.Errorf("cloud coverage too high")
    }
    
    // 3. Co-registration
    imagePair := &ImagePair{Reference: refImage, Target: tgtImage}
    if err := cdp.coRegistration.Process(ctx, imagePair); err != nil {
        return nil, err
    }
    
    // 4. Radiometric normalization
    if config.PreprocessingLevel == "full" {
        cdp.radiometric.Normalize(ctx, imagePair.AlignedPair)
    }
    
    // 5. Change detection
    var changeMap *ChangeMap
    var err error
    
    switch config.DetectionMethod {
    case "pixel":
        changeMap, err = cdp.pixelDetector.Detect(ctx, imagePair.AlignedPair)
    case "dl":
        changeMap, err = cdp.dlDetector.Detect(ctx, imagePair.AlignedPair)
    default:
        changeMap, err = cdp.pixelDetector.Detect(ctx, imagePair.AlignedPair)
    }
    
    if err != nil {
        return nil, err
    }
    
    result.ChangeMap = changeMap
    
    // 6. Generate outputs
    metadata := &ChangeMapMetadata{
        SourceImages:  []string{pair.Reference.ID, pair.Target.ID},
        DateRange:     [2]time.Time{pair.Reference.DateTime, pair.Target.DateTime},
        Algorithm:     config.DetectionMethod,
        ChangedPixels: changeMap.NumChanges,
        ChangePercent: changeMap.ChangePercent,
        GeoTransform:  refImage.GeoTransform,
    }
    
    output, _ := cdp.changeMapGen.Generate(ctx, changeMap, metadata)
    result.OutputPaths = output
    
    // 7. Extract vectors
    vectors, _ := cdp.vectorExtractor.Extract(ctx, changeMap, refImage.GeoTransform)
    result.Vectors = vectors
    
    return result, nil
}
```

```go
// ProcessParallel processes tiles in parallel
func (pp *ParallelProcessor) ProcessParallel(ctx context.Context, pair *AlignedImagePair, detector ChangeDetector) (*ChangeMap, error) {
    // Create tiles
    tiles := pp.createTiles(pair)
    
    // Result channel
    results := make(chan *TileResult, len(tiles))
    
    // Worker pool
    sem := make(chan struct{}, pp.numWorkers)
    var wg sync.WaitGroup
    
    for _, tile := range tiles {
        wg.Add(1)
        go func(t *Tile) {
            defer wg.Done()
            
            sem <- struct{}{}
            defer func() { <-sem }()
            
            // Process tile
            result := detector.DetectTile(ctx, t, pair)
            results <- result
        }(tile)
    }
    
    // Wait and close
    go func() {
        wg.Wait()
        close(results)
    }()
    
    // Merge results
    return pp.mergeResults(results, pair.Reference.Width, pair.Reference.Height)
}
```

```go
        report.Valid = false
        report.CloudCoverage = "high"
        report.Warnings = append(report.Warnings, "High cloud coverage")
    }
    
    // Sanity check on change percentage
    if result.ChangeMap.ChangePercent > qc.config.MaxChangePercent {
        report.ChangeReasonable = false
        report.Warnings = append(report.Warnings, "Unusually high change percentage")
    } else {
        report.ChangeReasonable = true
    }
    
    return report
}
```
