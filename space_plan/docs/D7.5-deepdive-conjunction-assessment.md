## 1. Executive Summary

### 1.1 Purpose

The Conjunction Assessment Pipeline provides automated collision risk analysis for satellite operations:

- **Conjunction Screening:** Identify close approaches between objects
- **Risk Quantification:** Calculate collision probability using covariance data
- **CDM Processing:** Ingest and analyze conjunction data messages
- **Decision Support:** Recommend avoidance maneuvers when necessary
- **Historical Analysis:** Track conjunction trends and patterns

### 1.2 Key Metrics

| Metric | Target | Description |
|--------|--------|-------------|
| **Screening Rate** | 10K objects/min | All-on-all conjunction check |
| **Pc Accuracy** | ±0.5 order of magnitude | Probability of collision |
| **Alert Latency** | < 5 minutes | CDM to operator notification |
| **Prediction Horizon** | 7 days | Standard screening window |
| **Update Frequency** | Every 8 hours | Routine screening cycle |

### 1.3 Risk Thresholds

| Threshold | Pc Value | Action |
|-----------|----------|--------|
| **Green** | < 1e-7 | No action required |
| **Yellow** | 1e-7 to 1e-5 | Monitor closely |
| **Orange** | 1e-5 to 1e-4 | Evaluate maneuver options |
| **Red** | > 1e-4 | Maneuver recommended |

## 4. Screening Algorithm

### 4.1 Multi-Stage Filtering

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                      CONJUNCTION SCREENING ALGORITHM                        │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│  CHALLENGE: N satellites × M objects = N×M comparisons                     │
│             100 satellites × 30,000 objects = 3,000,000 pairs              │
│                                                                             │
│  SOLUTION: Multi-stage filtering                                           │
│                                                                             │
│  STAGE 1: ORBITAL REGIME FILTER                                           │
│  • Skip if: Apogee₁ + margin < Perigee₂                                   │
│  • Margin: 50 km typical                                                   │
│  • Effectiveness: Eliminates ~60-70% of pairs                             │
│                                                                             │
│  STAGE 2: TEMPORAL FILTER                                                  │
│  • Propagate over 7-day window                                             │
│  • Coarse step: orbital_period / 10                                        │
│  • Check spatial proximity at each step                                    │
│  • Effectiveness: Eliminates ~80% of remaining                            │
│                                                                             │
│  STAGE 3: MISS DISTANCE CALCULATION                                       │
│  • Iterative refinement (Newton-Raphson)                                   │
│  • Find precise TCA                                                        │
│  • Threshold: 5 km alert, 25 km monitor                                   │
│                                                                             │
│  STAGE 4: PROBABILITY CALCULATION                                         │
│  • For conjunctions within threshold                                       │
│  • Calculate Pc using covariance data                                      │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 4.2 Screening Implementation

```go
// ScreeningService performs conjunction screening
type ScreeningService struct {
    propagatorPool *PropagatorPool
    catalogDB      *CatalogDatabase
    config         *ScreeningConfig
}

// ScreeningConfig configuration
type ScreeningConfig struct {
    ScreeningWindow       time.Duration // 7 days
    ApogeePerigeeMargin   float64       // 50 km
    MissDistanceThreshold float64       // 5 km
    MonitorThreshold      float64       // 25 km
    ParallelWorkers       int           // 8
}

// ScreenSatellite screens one satellite against catalog
func (ss *ScreeningService) ScreenSatellite(
    ctx context.Context,
    satelliteID string,
    startTime time.Time,
) ([]ScreeningResult, error) {
    // Get satellite TLE
    primaryTLE, err := ss.catalogDB.GetTLE(satelliteID)
    if err != nil {
        return nil, err
    }
    
    // Stage 1: Filter by orbital regime
    candidates, _ := ss.catalogDB.GetObjectsInRegime(
        primaryTLE.Perigee() - ss.config.ApogeePerigeeMargin,
        primaryTLE.Apogee() + ss.config.ApogeePerigeeMargin,
    )
    
    // Stage 2 & 3: Parallel processing
    results := ss.processInParallel(primaryTLE, candidates, startTime)
    
    return results, nil
}

// findCloseApproaches finds all close approaches
func (ss *ScreeningService) findCloseApproaches(
    primary, secondary *SGP4Propagator,
    start, end time.Time,
) []CloseApproach {
    approaches := make([]CloseApproach, 0)
    step := time.Minute // Coarse step
    
    prevDist := math.MaxFloat64
    
    for t := start; t.Before(end); t = t.Add(step) {
        state1, _ := primary.Propagate(t)
        state2, _ := secondary.Propagate(t)
        
        dist := distance(state1.Position, state2.Position)
        
        // Detect local minimum
        if dist < prevDist && dist < ss.config.MonitorThreshold {
            // Refine TCA
            tca, minDist := ss.refineTCA(primary, secondary, t.Add(-step), t.Add(step))
            
            approaches = append(approaches, CloseApproach{
                TCA:          tca,
                MissDistance: minDist,
            })
        }
        
        prevDist = dist
    }
    
    return approaches
}
```

### 5.1 Covariance Matrix Structure

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                      COVARIANCE MATRIX STRUCTURE                            │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│  6×6 covariance matrix representing uncertainty:                           │
│                                                                             │
│         │  x    y    z    vx   vy   vz  │                                  │
│     ────┼──────────────────────────────────                                │
│      x  │ σ²ₓ   σₓᵧ  σₓᵤ  ...              │  Position                     │
│      y  │      σ²ᵧ   σᵧᵤ  ...              │  variance                     │
│  P = z  │           σ²ᵤ   ...              │  (km²)                        │
│      vx │                σ²ᵥₓ  ...         │  Velocity                     │
│      vy │                     σ²ᵥᵧ  ...    │  variance                     │
│      vz │                          σ²ᵥᵤ   │  (km²/s²)                     │
│                                                                             │
│  TYPICAL VALUES (LEO with good tracking):                                  │
│  • Position 1-sigma: 100-500 m                                             │
│  • Velocity 1-sigma: 0.01-0.1 m/s                                          │
│                                                                             │
│  TLE-derived (no explicit uncertainty):                                    │
│  • Position 1-sigma: 500-2000 m                                            │
│  • Velocity 1-sigma: 0.1-1 m/s                                             │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 5.2 Covariance Propagation

```go
// CovarianceMatrix 6x6 covariance
type CovarianceMatrix [6][6]float64

// LinearCovariancePropagator propagates covariance
type LinearCovariancePropagator struct {
    mu float64 // Gravitational parameter
}

// Propagate using state transition matrix
// P(t+dt) = Φ * P * Φᵀ + Q
func (lcp *LinearCovariancePropagator) Propagate(
    cov *CovarianceMatrix,
    state *StateVector,
    dt float64,
) *CovarianceMatrix {
    phi := lcp.stateTransitionMatrix(state, dt)
    q := lcp.processNoiseMatrix(dt)
    
    // Matrix multiplication: result = Φ * P * Φᵀ + Q
    return matrixMultiply(phi, cov, transpose(phi), q)
}

// ScaleCovarianceFromTLE estimates covariance from TLE age
func ScaleCovarianceFromTLE(tle *TLE, age time.Duration) *CovarianceMatrix {
    basePos := 0.5  // km at epoch
    baseVel := 0.0001 // km/s
    
    // Scale with age (10% growth per hour)
    scaleFactor := 1.0 + 0.1*age.Hours()
    
    // Create diagonal covariance
    cov := &CovarianceMatrix{}
    posUnc := basePos * scaleFactor
    velUnc := baseVel * scaleFactor
    
    cov[0][0] = posUnc * posUnc       // Radial
    cov[1][1] = posUnc * posUnc * 4   // Along-track (2x larger)
    cov[2][2] = posUnc * posUnc * 1.5 // Cross-track
    cov[3][3] = velUnc * velUnc
    cov[4][4] = velUnc * velUnc
    cov[5][5] = velUnc * velUnc
    
    return cov
}
```

## 6. Probability of Collision

### 6.1 Pc Calculation Methods

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                    PROBABILITY OF COLLISION METHODS                         │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│  METHOD 1: FOSTER (2D Gaussian Integration)                               │
│  ═══════════════════════════════════════════                               │
│                                                                             │
│         1                    ⌠⌠                                            │
│  Pc = ─────────────────────  ⎮⎮ exp(-½ xᵀ C⁻¹ x) dx dy                    │
│       2π √(det C)            ⌡⌡                                            │
│                            collision disk                                   │
│                                                                             │
│  • Assumes Gaussian covariance                                             │
│  • Instantaneous encounter                                                 │
│                                                                             │
│  METHOD 2: ALFANO (Analytical Approximation)                              │
│  ════════════════════════════════════════════                              │
│                                                                             │
│          R²                 m²                                              │
│  Pc = ────────── exp( - ────────── )                                       │
│       2 σ₁ σ₂           2 σ₁² σ₂²                                          │
│                                                                             │
│  • σ₁, σ₂ = covariance eigenvalues                                        │
│  • m = miss distance                                                       │
│  • R = hard body radius                                                    │
│  • Valid when m >> R                                                       │
│                                                                             │
│  METHOD 3: MONTE CARLO                                                     │
│  ══════════════════════                                                    │
│                                                                             │
│  • Sample N points from covariance                                         │
│  • Count samples where distance < R                                        │
│  • Pc ≈ count / N                                                          │
│  • Handles non-Gaussian cases                                              │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 6.2 Pc Implementation

```go
// PcCalculator calculates probability of collision
type PcCalculator struct {
    method PcMethod
}

type PcMethod int
const (
    PcMethodFoster PcMethod = iota
    PcMethodAlfano
    PcMethodMonteCarlo
)

// PcResult contains calculation results
type PcResult struct {
    Pc             float64
    Method         PcMethod
    MissDistance   float64
    HardBodyRadius float64
    Mahalanobis    float64
}

// CalculatePc computes collision probability
func (calc *PcCalculator) CalculatePc(
    state1, state2 *StateVector,
    cov1, cov2 *CovarianceMatrix,
    hbr1, hbr2 float64,
) *PcResult {
    hbr := hbr1 + hbr2
    
    // Create encounter frame
    encFrame := NewEncounterFrame(state1, state2)
    
    // Combine covariances
    combinedCov := CombineCovariances(cov1, cov2)
    
    // Project to 2D encounter plane
    cov2D := ProjectToEncounterPlane(combinedCov, encFrame)
    
    // Calculate miss distance
    missDist := distance(state1.Position, state2.Position)
    
    // Calculate Pc based on method
    var pc float64
    switch calc.method {
    case PcMethodFoster:
        pc = calc.fosterMethod(missDist, hbr, cov2D)
    case PcMethodAlfano:
        pc = calc.alfanoMethod(missDist, hbr, cov2D)
    case PcMethodMonteCarlo:
        pc = calc.monteCarloMethod(missDist, hbr, cov2D)
    }
    
    return &PcResult{
        Pc:             pc,
        Method:         calc.method,
        MissDistance:   missDist,
        HardBodyRadius: hbr,
        Mahalanobis:    calc.mahalanobisDistance(encFrame.Origin, cov2D),
    }
}

// alfanoMethod analytical approximation
func (calc *PcCalculator) alfanoMethod(miss, hbr float64, cov *CovarianceMatrix2D) float64 {
    sigma1, sigma2, _ := eigenDecomposition(cov)
    
    u := miss * miss / (2 * sigma1 * sigma1)
    v := miss * miss / (2 * sigma2 * sigma2)
    
    pc := (hbr * hbr / (2 * sigma1 * sigma2)) * math.Exp(-u - v)
    
    return min(pc, 1.0)
}
```

```go
// CDM represents a parsed Conjunction Data Message
type CDM struct {
    MessageID      string
    CreationDate   time.Time
    Originator     string
    TCA            time.Time
    MissDistance   float64
    RelativeSpeed  float64
    Pc             float64
    PcMethod       string
    Object1        CDMObject
    Object2        CDMObject
}

// CDMProcessor processes CDMs into conjunction events
type CDMProcessor struct {
    eventStore   *ConjunctionEventStore
    alertService *AlertService
    pcCalculator *PcCalculator
}

// ProcessCDM processes a received CDM
func (cp *CDMProcessor) ProcessCDM(cdm *CDM) (*ConjunctionEvent, error) {
    event := &ConjunctionEvent{
        EventID:      generateEventID(cdm),
        SatelliteID:  cdm.Object1.Designator,
        SecondaryID:  cdm.Object2.Designator,
        TCA:          cdm.TCA,
        MissDistance: cdm.MissDistance,
        Pc:           cdm.Pc,
    }
    
    // Recalculate Pc if covariance available
    if cdm.Object1.Covariance != nil && cdm.Object2.Covariance != nil {
        pcResult := cp.pcCalculator.CalculatePc(...)
        event.PcCalculated = pcResult.Pc
    }
    
    // Determine risk level
    event.RiskLevel = cp.classifyRisk(event.Pc)
    
    // Store and alert
    cp.eventStore.Save(event)
    if event.RiskLevel >= RiskLevelYellow {
        cp.alertService.CreateAlert(event)
    }
    
    return event, nil
}
```

## 9. Maneuver Planning

### 9.1 Maneuver Types

| Type | Direction | Effect | Typical ΔV |
|------|-----------|--------|------------|
| **Along-track** | Velocity direction | Changes period, shifts timing | 0.01-0.1 m/s |
| **Cross-track** | Perpendicular to plane | Lateral separation | 0.05-0.5 m/s |
| **Radial** | Toward/away Earth | Changes altitude | Rarely used alone |
| **Optimal** | Opposite Pc gradient | Minimum fuel | Computed |

### 9.2 Maneuver Planning Implementation

```go
// ManeuverPlanner plans collision avoidance maneuvers
type ManeuverPlanner struct {
    propagator   *SGP4Propagator
    pcCalculator *PcCalculator
    constraints  *ManeuverConstraints
}

// ManeuverOption represents one maneuver possibility
type ManeuverOption struct {
    Type           string    // ALONG_TRACK, CROSS_TRACK, OPTIMAL
    Direction      Vector3   // RTN direction
    DeltaV         float64   // km/s
    BurnTime       time.Time
    ResultingPc    float64
    ResultingMiss  float64
    FuelRequired   float64
    SecondaryConjs []ScreeningResult // New conjunctions created
}

// GetOptions generates maneuver options
func (mp *ManeuverPlanner) GetOptions(event *ConjunctionEvent) []ManeuverOption {
    options := make([]ManeuverOption, 0)
    
    burnTime := event.TCA.Add(-time.Duration(event.Object1.OrbitalPeriod()/2) * time.Minute)
    
    // Try different directions and magnitudes
    directions := []string{"ALONG_TRACK_POS", "ALONG_TRACK_NEG", "CROSS_TRACK_POS", "CROSS_TRACK_NEG"}
    magnitudes := []float64{0.01, 0.02, 0.05, 0.1} // m/s
    
    for _, dir := range directions {
        for _, mag := range magnitudes {
            option := mp.evaluateManeuver(event, dir, mag/1000, burnTime)
            if option != nil {
                options = append(options, *option)
            }
        }
    }
    
    // Add optimal direction
    optDir, optMag := mp.calculateOptimalDirection(event, burnTime)
    optimal := mp.evaluateManeuver(event, "OPTIMAL", optMag, burnTime)
    if optimal != nil {
        optimal.Direction = optDir
        options = append(options, *optimal)
    }
    
    // Sort by effectiveness
    sort.Slice(options, func(i, j int) bool {
        return options[i].PcReduction/options[i].FuelRequired > options[j].PcReduction/options[j].FuelRequired
    })
    
    return options
}
```

```go
package conjunction

import (
    "context"

// ScreeningService performs conjunction screening
type ScreeningService struct {
    propagatorPool *PropagatorPool
    catalogDB      *CatalogDatabase
    eventStore     *ConjunctionEventStore
    config         *ScreeningConfig
    metrics        *ScreeningMetrics
}

// ScreeningConfig configures the screening process
type ScreeningConfig struct {
    ScreeningWindow       time.Duration // 7 days typical
    CoarseStepFraction    float64       // Fraction of orbital period
    ApogeePerigeeMargin   float64       // km
    MissDistanceThreshold float64       // km for detailed analysis
    MonitorThreshold      float64       // km for monitoring
    ParallelWorkers       int
    BatchSize             int
}

// DefaultScreeningConfig returns typical configuration
func DefaultScreeningConfig() *ScreeningConfig {
    return &ScreeningConfig{
        ScreeningWindow:       7 * 24 * time.Hour,
        CoarseStepFraction:    0.1,
        ApogeePerigeeMargin:   50.0,
        MissDistanceThreshold: 5.0,
        MonitorThreshold:      25.0,
        ParallelWorkers:       8,
        BatchSize:             100,
    }
}

// ScreeningResult represents a potential conjunction
type ScreeningResult struct {
    PrimarySatellite  string
    PrimaryNorad      int
    SecondaryObject   string
    SecondaryNorad    int
    SecondaryType     string
    TCA               time.Time
    MissDistance      float64
    RelativeVelocity  float64
    NeedsAnalysis     bool
}

// ScreeningMetrics tracks screening performance
type ScreeningMetrics struct {
    TotalPairs        int64
    Stage1Filtered    int64
    Stage2Filtered    int64
    ConjunctionsFound int64
    ProcessingTime    time.Duration
}

// ScreenAllSatellites screens all protected satellites
func (ss *ScreeningService) ScreenAllSatellites(
    ctx context.Context,
    satelliteIDs []string,
    startTime time.Time,
) (*ScreeningReport, error) {
    
    report := &ScreeningReport{
        StartTime:   startTime,
        EndTime:     startTime.Add(ss.config.ScreeningWindow),
        Satellites:  len(satelliteIDs),
        Results:     make([]ScreeningResult, 0),
    }
    
    startProcess := time.Now()
    
    // Process each satellite
    for _, satID := range satelliteIDs {
        results, err := ss.ScreenSatellite(ctx, satID, startTime)
        if err != nil {
            log.Error("Screening failed for satellite", "id", satID, "error", err)
            continue
        }
        
        report.Results = append(report.Results, results...)
    }
    
    report.ProcessingTime = time.Since(startProcess)
    report.ConjunctionsFound = len(report.Results)
    
    // Sort by TCA
    sort.Slice(report.Results, func(i, j int) bool {
        return report.Results[i].TCA.Before(report.Results[j].TCA)
    })
    
    return report, nil
}

// ScreenSatellite screens one satellite against the catalog
func (ss *ScreeningService) ScreenSatellite(
    ctx context.Context,
    satelliteID string,
    startTime time.Time,
) ([]ScreeningResult, error) {
    
    // Get satellite TLE/ephemeris
    primaryTLE, err := ss.catalogDB.GetTLE(satelliteID)
    if err != nil {
        return nil, fmt.Errorf("failed to get primary TLE: %w", err)
    }
    
    // Get orbital parameters for filtering
    primaryApogee := primaryTLE.Apogee()
    primaryPerigee := primaryTLE.Perigee()
    primaryPeriod := primaryTLE.OrbitalPeriod()
    
    // Stage 1: Filter by orbital regime
    candidates, err := ss.catalogDB.GetObjectsInRegime(
        primaryPerigee - ss.config.ApogeePerigeeMargin,
        primaryApogee + ss.config.ApogeePerigeeMargin,
    )
    if err != nil {
        return nil, err
    }
    
    ss.metrics.TotalPairs += int64(len(candidates))
    ss.metrics.Stage1Filtered += int64(len(candidates))
    
    // Stage 2 & 3: Parallel temporal filtering and miss distance
    results := make([]ScreeningResult, 0)
    resultsMu := sync.Mutex{}
    
    // Create work channel
    work := make(chan *TLE, len(candidates))
    for _, tle := range candidates {
        work <- tle
    }
    close(work)
    
    // Process in parallel
    var wg sync.WaitGroup
    for i := 0; i < ss.config.ParallelWorkers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            
            primaryProp, _ := ss.propagatorPool.Get(primaryTLE)
            
            for secondaryTLE := range work {
                if secondaryTLE.CatalogNumber == primaryTLE.CatalogNumber {
                    continue
                }
                
                secondaryProp, _ := ss.propagatorPool.Get(secondaryTLE)
                
                // Find close approaches
                approaches := ss.findCloseApproaches(
                    primaryProp, secondaryProp,
                    startTime, startTime.Add(ss.config.ScreeningWindow),
                    primaryPeriod,
                )
                
                for _, approach := range approaches {
                    if approach.MissDistance <= ss.config.MonitorThreshold {
                        result := ScreeningResult{
                            PrimarySatellite:  satelliteID,
                            PrimaryNorad:      primaryTLE.CatalogNumber,
                            SecondaryObject:   secondaryTLE.Name,
                            SecondaryNorad:    secondaryTLE.CatalogNumber,
                            SecondaryType:     ss.classifyObject(secondaryTLE),
                            TCA:               approach.TCA,
                            MissDistance:      approach.MissDistance,
                            RelativeVelocity:  approach.RelativeVelocity,
                            NeedsAnalysis:     approach.MissDistance <= ss.config.MissDistanceThreshold,
                        }
                        
                        resultsMu.Lock()
                        results = append(results, result)
                        resultsMu.Unlock()
                    }
                }
            }
        }()
    }
    
    wg.Wait()
    
    ss.metrics.ConjunctionsFound += int64(len(results))
    
    return results, nil
}

// CloseApproach represents a potential conjunction point
type CloseApproach struct {
    TCA              time.Time
    MissDistance     float64
    RelativeVelocity float64
}

// findCloseApproaches finds all close approaches in a time window
func (ss *ScreeningService) findCloseApproaches(
    primary, secondary *SGP4Propagator,
    start, end time.Time,
    period float64,
) []CloseApproach {
    
    approaches := make([]CloseApproach, 0)
    
    // Coarse step based on shorter orbital period
    step := time.Duration(period * ss.config.CoarseStepFraction * float64(time.Minute))
    if step < time.Minute {
        step = time.Minute
    }
    
    // Track distance trend for finding minima
    prevDist := math.MaxFloat64
    prevPrevDist := math.MaxFloat64
    prevTime := start
    
    for t := start; t.Before(end); t = t.Add(step) {
        state1, err1 := primary.Propagate(t)
        state2, err2 := secondary.Propagate(t)
        
        if err1 != nil || err2 != nil {
            continue
        }
        
        // Calculate distance
        dist := distance(state1.Position, state2.Position)
        
        // Check for local minimum
        if prevDist < prevPrevDist && prevDist < dist && prevDist < ss.config.MonitorThreshold*2 {
            // Refine to find exact TCA
            tca, minDist, relVel := ss.refineTCA(
                primary, secondary,
                prevTime.Add(-step), t,
            )
            
            approaches = append(approaches, CloseApproach{
                TCA:              tca,
                MissDistance:     minDist,
                RelativeVelocity: relVel,
            })
        }
        
        prevPrevDist = prevDist
        prevDist = dist
        prevTime = t
    }
    
    return approaches
}

// refineTCA uses golden section search for exact TCA
func (ss *ScreeningService) refineTCA(
    primary, secondary *SGP4Propagator,
    start, end time.Time,
) (time.Time, float64, float64) {
    
    phi := (1.0 + math.Sqrt(5.0)) / 2.0
    tolerance := 100 * time.Millisecond
    
    a := start
    b := end
    
    for b.Sub(a) > tolerance {
        x1 := b.Add(-time.Duration(float64(b.Sub(a)) / phi))
        x2 := a.Add(time.Duration(float64(b.Sub(a)) / phi))
        
        d1 := ss.distanceAt(primary, secondary, x1)
        d2 := ss.distanceAt(primary, secondary, x2)
        
        if d1 < d2 {
            b = x2
        } else {
            a = x1
        }
    }
    
    tca := a.Add(b.Sub(a) / 2)
    
    state1, _ := primary.Propagate(tca)
    state2, _ := secondary.Propagate(tca)
    
    minDist := distance(state1.Position, state2.Position)
    relVel := relativeVelocity(state1, state2)
    
    return tca, minDist, relVel
}

// distanceAt calculates distance at a given time
func (ss *ScreeningService) distanceAt(p1, p2 *SGP4Propagator, t time.Time) float64 {
    s1, err1 := p1.Propagate(t)
    s2, err2 := p2.Propagate(t)
    
    if err1 != nil || err2 != nil {
        return math.MaxFloat64
    }
    
    return distance(s1.Position, s2.Position)
}

// classifyObject determines object type
func (ss *ScreeningService) classifyObject(tle *TLE) string {
    name := strings.ToUpper(tle.Name)
    
    if strings.Contains(name, "DEB") {
        return "DEBRIS"
    }
    if strings.Contains(name, "R/B") {
        return "ROCKET_BODY"
    }
    if strings.Contains(name, "STARLINK") {
        return "STARLINK"
    }
    if strings.Contains(name, "ONEWEB") {
        return "ONEWEB"
    }
    
    return "PAYLOAD"
}
```

```go
// probability_calculator.go

package conjunction

import (
    "math"
    "math/rand"
)

// PcCalculator calculates probability of collision
type PcCalculator struct {
    method      PcMethod
    monteCarlo  *MonteCarloConfig
}

// PcMethod specifies calculation method
type PcMethod int

const (
    PcMethodFoster    PcMethod = iota
    PcMethodAlfano
    PcMethodMonteCarlo
    PcMethodChan
)

// MonteCarloConfig configures Monte Carlo simulation
type MonteCarloConfig struct {
    Samples     int
    RandomSeed  int64
    UseParallel bool
    Workers     int
}

// PcResult contains probability calculation results
type PcResult struct {
    Pc              float64
    PcUpper         float64
    PcLower         float64
    Method          PcMethod
    MissDistance    float64
    HardBodyRadius  float64
    CombinedSigma   float64
    Mahalanobis     float64
    Iterations      int
    Confidence      float64
}

// NewPcCalculator creates a new probability calculator
func NewPcCalculator(method PcMethod) *PcCalculator {
    return &PcCalculator{
        method: method,
        monteCarlo: &MonteCarloConfig{
            Samples:     100000,
            UseParallel: true,
            Workers:     4,
        },
    }
}

// CalculatePc computes collision probability
func (calc *PcCalculator) CalculatePc(
    state1, state2 *StateVector,
    cov1, cov2 *CovarianceMatrix,
    hbr1, hbr2 float64,
) *PcResult {
    
    hbr := hbr1 + hbr2
    
    // Create encounter frame
    encFrame := NewEncounterFrame(state1, state2)
    
    // Combine covariances
    combinedCov := CombineCovariances(cov1, cov2)
    
    // Project to encounter plane (2D)
    cov2D := ProjectToEncounterPlane(combinedCov, encFrame)
    
    // Calculate miss distance
    missDist := distance(state1.Position, state2.Position)
    
    var pc float64
    var iterations int
    
    switch calc.method {
    case PcMethodFoster:
        pc, iterations = calc.fosterMethod(missDist, hbr, cov2D)
    case PcMethodAlfano:
        pc = calc.alfanoMethod(missDist, hbr, cov2D)
    case PcMethodChan:
        pc = calc.chanMethod(missDist, hbr, cov2D)
    case PcMethodMonteCarlo:
        pc = calc.monteCarloMethod(missDist, hbr, cov2D)
    default:
        pc, iterations = calc.fosterMethod(missDist, hbr, cov2D)
    }
    
    // Calculate Mahalanobis distance
    mahal := calc.mahalanobisDistance(encFrame.Origin, cov2D)
    
    // Calculate combined sigma
    det := cov2D.SigmaXX*cov2D.SigmaYY - cov2D.SigmaXY*cov2D.SigmaXY
    combinedSigma := math.Sqrt(det)
    
    return &PcResult{
        Pc:             pc,
        Method:         calc.method,
        MissDistance:   missDist,
        HardBodyRadius: hbr,
        CombinedSigma:  combinedSigma,
        Mahalanobis:    mahal,
        Iterations:     iterations,
        Confidence:     calc.estimateConfidence(mahal, cov2D),
    }
}

// fosterMethod implements Foster's 2D Gaussian integration
func (calc *PcCalculator) fosterMethod(miss, hbr float64, cov *CovarianceMatrix2D) (float64, int) {
    // Eigenvalue decomposition
    sigma1, sigma2, theta := calc.eigenDecomposition(cov)
    
    if sigma1 < 1e-10 || sigma2 < 1e-10 {
        return 0, 0
    }
    
    // Normalize coordinates
    cosTheta := math.Cos(theta)
    sinTheta := math.Sin(theta)
    
    // Number of integration points
    nRadial := 50
    nAngular := 100
    
    dr := hbr / float64(nRadial)
    dtheta := 2 * math.Pi / float64(nAngular)
    
    pc := 0.0
    iterations := nRadial * nAngular
    
    // Integrate over collision disk
    for i := 0; i < nRadial; i++ {
        r := (float64(i) + 0.5) * dr
        
        for j := 0; j < nAngular; j++ {
            th := float64(j) * dtheta
            
            // Point in collision disk (centered at miss point origin)
            x := r * math.Cos(th)
            y := r * math.Sin(th)
            
            // Transform to principal axes
            xp := x*cosTheta + y*sinTheta
            yp := -x*sinTheta + y*cosTheta
            
            // Gaussian probability density
            z := xp*xp/(sigma1*sigma1) + yp*yp/(sigma2*sigma2)
            prob := math.Exp(-0.5*z) / (2 * math.Pi * sigma1 * sigma2)
            
            // Area element
            area := r * dr * dtheta
            
            pc += prob * area
        }
    }
    
    return pc, iterations
}

// alfanoMethod implements Alfano's analytical approximation
func (calc *PcCalculator) alfanoMethod(miss, hbr float64, cov *CovarianceMatrix2D) float64 {
    sigma1, sigma2, _ := calc.eigenDecomposition(cov)
    
    if sigma1 < 1e-10 || sigma2 < 1e-10 {
        return 0
    }
    
    if miss < 1e-10 {
        miss = 1e-10
    }
    
    // Alfano's formula
    u := miss * miss / (2 * sigma1 * sigma1)
    v := miss * miss / (2 * sigma2 * sigma2)
    
    exponent := -u - v
    if exponent < -700 {
        return 0
    }
    
    pc := (hbr * hbr / (2 * sigma1 * sigma2)) * math.Exp(exponent)
    
    if pc > 1 {
        pc = 1
    }
    
    return pc
}

// chanMethod implements Chan's formula (more accurate for small miss distances)
func (calc *PcCalculator) chanMethod(miss, hbr float64, cov *CovarianceMatrix2D) float64 {
    sigma1, sigma2, _ := calc.eigenDecomposition(cov)
    
    if sigma1 < 1e-10 || sigma2 < 1e-10 {
        return 0
    }
    
    // Normalized quantities
    u := hbr / sigma1
    v := hbr / sigma2
    x := miss / sigma1
    y := miss / sigma2
    
    // Chan's approximation with Rice distribution
    // More accurate when miss distance is comparable to HBR
    bessel := math.I0(x * u / (sigma1 * sigma1))
    
    pc := (u * v / 2) * math.Exp(-(x*x+u*u)/(2)) * bessel
    
    return math.Min(pc, 1.0)
}

// monteCarloMethod uses Monte Carlo simulation
func (calc *PcCalculator) monteCarloMethod(miss, hbr float64, cov *CovarianceMatrix2D) float64 {
    n := calc.monteCarlo.Samples
    if n == 0 {
        n = 100000
    }
    
    // Cholesky decomposition for sampling
    L := calc.choleskyDecomposition(cov)
    
    collisions := 0
    rng := rand.New(rand.NewSource(calc.monteCarlo.RandomSeed))
    
    for i := 0; i < n; i++ {
        // Sample standard normal
        z1 := rng.NormFloat64()
        z2 := rng.NormFloat64()
        
        // Transform to covariance
        x := L[0][0]*z1 + miss
        y := L[1][0]*z1 + L[1][1]*z2
        
        // Check if within HBR
        dist := math.Sqrt(x*x + y*y)
        if dist < hbr {
            collisions++
        }
    }
    
    return float64(collisions) / float64(n)
}

// eigenDecomposition computes eigenvalues and rotation angle
func (calc *PcCalculator) eigenDecomposition(cov *CovarianceMatrix2D) (sigma1, sigma2, theta float64) {
    a := cov.SigmaXX
    b := cov.SigmaXY
    c := cov.SigmaYY
    
    trace := a + c
    det := a*c - b*b
    
    if det < 0 {
        det = 0
    }
    
    discriminant := math.Sqrt(math.Max(0, trace*trace/4-det))
    lambda1 := trace/2 + discriminant
    lambda2 := trace/2 - discriminant
    
    sigma1 = math.Sqrt(math.Abs(lambda1))
    sigma2 = math.Sqrt(math.Abs(lambda2))
    
    if math.Abs(b) < 1e-10 {
        theta = 0
    } else {
        theta = 0.5 * math.Atan2(2*b, a-c)
    }
    
    return sigma1, sigma2, theta
}

// choleskyDecomposition for Monte Carlo sampling
func (calc *PcCalculator) choleskyDecomposition(cov *CovarianceMatrix2D) [2][2]float64 {
    var L [2][2]float64
    
    L[0][0] = math.Sqrt(math.Max(0, cov.SigmaXX))
    if L[0][0] > 1e-10 {
        L[1][0] = cov.SigmaXY / L[0][0]
    }
    
    inner := cov.SigmaYY - L[1][0]*L[1][0]
    L[1][1] = math.Sqrt(math.Max(0, inner))
    
    return L
}

// mahalanobisDistance calculates Mahalanobis distance
func (calc *PcCalculator) mahalanobisDistance(miss Vector3, cov *CovarianceMatrix2D) float64 {
    det := cov.SigmaXX*cov.SigmaYY - cov.SigmaXY*cov.SigmaXY
    if det < 1e-20 {
        det = 1e-20
    }
    
    invXX := cov.SigmaYY / det
    invYY := cov.SigmaXX / det
    invXY := -cov.SigmaXY / det
    
    x := miss.X
    y := miss.Y
    
    mahal2 := x*x*invXX + 2*x*y*invXY + y*y*invYY
    
    return math.Sqrt(math.Max(0, mahal2))
}

// estimateConfidence estimates confidence in Pc calculation
func (calc *PcCalculator) estimateConfidence(mahal float64, cov *CovarianceMatrix2D) float64 {
    // Higher Mahalanobis distance = lower confidence in Pc
    // Very close approaches have higher uncertainty
    
    sigma1, sigma2, _ := calc.eigenDecomposition(cov)
    
    // Check condition number
    if sigma2 < 1e-10 {
        return 0.3
    }
    
    conditionNumber := sigma1 / sigma2
    if conditionNumber > 100 {
        return 0.5
    }
    
    // Based on Mahalanobis distance
    if mahal < 1 {
        return 0.7 // Very close, higher uncertainty
    } else if mahal < 3 {
        return 0.9
    }
    
    return 0.95
}
```

```go
// PlanManeuver generates and analyzes maneuver options
func (mp *ManeuverPlanner) PlanManeuver(event *ConjunctionEvent) (*ManeuverPlan, error) {
    plan := &ManeuverPlan{
        EventID:          event.EventID,
        AnalyzedAt:       time.Now(),
        Options:          make([]ManeuverOption, 0),
        DecisionDeadline: event.TCA.Add(-mp.constraints.MinLeadTime),
        Notes:            make([]string, 0),
    }
    
    // Check if maneuver is still possible
    if time.Until(event.TCA) < mp.constraints.MinLeadTime {
        plan.Recommendation = "NO_ACTION_POSSIBLE"
        plan.Notes = append(plan.Notes, "Insufficient time for maneuver execution")
        return plan, nil
    }
    
    // Determine optimal burn time
    burnTime := mp.selectBurnTime(event)
    
    // Generate options in each direction
    directions := []struct {
        name string
        dir  Vector3
    }{
        {"ALONG_TRACK_POS", Vector3{0, 1, 0}},
        {"ALONG_TRACK_NEG", Vector3{0, -1, 0}},
        {"CROSS_TRACK_POS", Vector3{0, 0, 1}},
        {"CROSS_TRACK_NEG", Vector3{0, 0, -1}},
        {"RADIAL_POS", Vector3{1, 0, 0}},
        {"RADIAL_NEG", Vector3{-1, 0, 0}},
    }
    
    // Try different magnitudes
    magnitudes := []float64{0.005, 0.01, 0.02, 0.05, 0.1, 0.2, 0.5} // m/s
    
    for _, dir := range directions {
        for _, mag := range magnitudes {
            if mag > mp.constraints.MaxDeltaV {
                continue
            }
            
            option := mp.evaluateManeuver(event, dir.name, dir.dir, mag/1000, burnTime)
            if option != nil && option.IsFeasible {
                plan.Options = append(plan.Options, *option)
            }
        }
    }
    
    // Calculate optimal direction
    optimalDir, optimalMag := mp.calculateOptimalDirection(event, burnTime)
    if optimalMag*1000 <= mp.constraints.MaxDeltaV {
        optimalOption := mp.evaluateManeuver(event, "OPTIMAL", optimalDir, optimalMag, burnTime)
        if optimalOption != nil && optimalOption.IsFeasible {
            plan.Options = append(plan.Options, *optimalOption)
        }
    }
    
    // Score and sort options
    mp.scoreOptions(plan.Options)
    sort.Slice(plan.Options, func(i, j int) bool {
        return plan.Options[i].Score > plan.Options[j].Score
    })
    
    // Select recommendation
    if len(plan.Options) > 0 {
        best := plan.Options[0]
        plan.RecommendedID = best.ID
        
        if best.ResultingPc < 1e-6 {
            plan.Recommendation = "EXECUTE_RECOMMENDED"
        } else if best.ResultingPc < 1e-5 {
            plan.Recommendation = "CONSIDER_EXECUTION"
        } else {
            plan.Recommendation = "INSUFFICIENT_REDUCTION"
            plan.Notes = append(plan.Notes, "Best option does not reduce Pc to acceptable level")
        }
    } else {
        plan.Recommendation = "NO_FEASIBLE_OPTIONS"
    }
    
    return plan, nil
}

// selectBurnTime chooses optimal maneuver time
func (mp *ManeuverPlanner) selectBurnTime(event *ConjunctionEvent) time.Time {
    // Default: half orbit before TCA
    period := event.Object1.OrbitalPeriod()
    idealLead := time.Duration(period/2) * time.Minute
    
    burnTime := event.TCA.Add(-idealLead)
    
    // Adjust for constraints
    now := time.Now()
    minBurnTime := now.Add(mp.constraints.PreferredBurn)
    maxBurnTime := event.TCA.Add(-mp.constraints.MinLeadTime)
    
    if burnTime.Before(minBurnTime) {
        burnTime = minBurnTime
    }
    if burnTime.After(maxBurnTime) {
        burnTime = maxBurnTime
    }
    
    return burnTime
}

// evaluateManeuver assesses a specific maneuver option
func (mp *ManeuverPlanner) evaluateManeuver(
    event *ConjunctionEvent,
    typeName string,
    direction Vector3,
    deltaV float64,
    burnTime time.Time,
) *ManeuverOption {
    
    option := &ManeuverOption{
        ID:           fmt.Sprintf("%s-%.3f", typeName, deltaV*1000),
        Type:         typeName,
        DirectionRTN: direction,
        DeltaV:       deltaV,
        DeltaVMetric: deltaV * 1000,
        BurnTime:     burnTime,
        OriginalPc:   event.Pc,
        IsFeasible:   true,
    }
    
    // Calculate fuel required
    option.FuelRequired = mp.calculateFuel(deltaV * 1000)
    option.FuelPct = (option.FuelRequired / mp.constraints.FuelBudget) * 100
    
    // Check fuel constraint
    if option.FuelRequired > mp.constraints.FuelBudget {
        option.IsFeasible = false
        option.FeasibilityNote = "Exceeds fuel budget"
        return option
    }
    
    // Propagate maneuver effect
    newState, err := mp.propagateWithManeuver(event, direction, deltaV, burnTime)
    if err != nil {
        option.IsFeasible = false
        option.FeasibilityNote = "Propagation failed"
        return option
    }
    
    // Get secondary state at TCA
    secondaryState := &StateVector{
        Position: event.Object2.Position,
        Velocity: event.Object2.Velocity,
        Time:     event.TCA,
    }
    
    // Calculate new miss distance
    option.ResultingMiss = distance(newState.Position, secondaryState.Position)
    
    // Calculate new Pc
    cov1 := ScaleCovarianceFromTLE(nil, time.Hour)
    cov2 := ScaleCovarianceFromTLE(nil, time.Hour)
    
    pcResult := mp.pcCalculator.CalculatePc(
        newState, secondaryState,
        cov1, cov2,
        event.Object1.HardBodyRadius, event.Object2.HardBodyRadius,
    )
    
    option.ResultingPc = pcResult.Pc
    option.PcReduction = event.Pc - pcResult.Pc
    if event.Pc > 0 {
        option.PcReductionPct = (option.PcReduction / event.Pc) * 100
    }
    
    // Calculate orbit changes
    option.NewOrbit = mp.calculateOrbitDelta(event, direction, deltaV)
    
    // Calculate burn duration
    thrust := 0.1 // N typical for cold gas
    option.BurnDuration = time.Duration(option.FuelRequired * mp.constraints.Isp * 9.81 / thrust) * time.Second
    
    // Check for secondary conjunctions
    option.SecondaryConjs = mp.checkSecondaryConjunctions(event, newState)
    option.HasSecondaryRisk = len(option.SecondaryConjs) > 0
    
    // Calculate efficiency
    if option.FuelRequired > 0 {
        option.Efficiency = option.PcReduction / option.FuelRequired
    }
    
    return option
}

// calculateOptimalDirection finds minimum fuel direction
func (mp *ManeuverPlanner) calculateOptimalDirection(
    event *ConjunctionEvent,
    burnTime time.Time,
) (Vector3, float64) {
    
    // Calculate Pc gradient numerically
    epsilon := 0.001 // km/s
    basePc := event.Pc
    
    gradients := Vector3{}
    
    directions := []Vector3{
        {1, 0, 0}, // R
        {0, 1, 0}, // T
        {0, 0, 1}, // N
    }
    
    for i, dir := range directions {
        opt := mp.evaluateManeuver(event, "test", dir, epsilon, burnTime)
        if opt != nil {
            dPc := (opt.ResultingPc - basePc) / epsilon
            switch i {
            case 0:
                gradients.X = dPc
            case 1:
                gradients.Y = dPc
            case 2:
                gradients.Z = dPc
            }
        }
    }
    
    gradMag := math.Sqrt(gradients.X*gradients.X + gradients.Y*gradients.Y + gradients.Z*gradients.Z)
    
    if gradMag < 1e-10 {
        return Vector3{0, -1, 0}, 0.01 / 1000
    }
    
    // Optimal direction is opposite to gradient
    optDir := Vector3{
        X: -gradients.X / gradMag,
        Y: -gradients.Y / gradMag,
        Z: -gradients.Z / gradMag,
    }
    
    // Estimate magnitude needed
    targetPc := 1e-6
    estimatedMag := math.Abs(basePc-targetPc) / gradMag
    
    if estimatedMag*1000 > mp.constraints.MaxDeltaV {
        estimatedMag = mp.constraints.MaxDeltaV / 1000
    }
    
    return optDir, estimatedMag
}

// calculateFuel estimates fuel required using Tsiolkovsky equation
func (mp *ManeuverPlanner) calculateFuel(deltaVms float64) float64 {
    g0 := 9.80665
    isp := mp.constraints.Isp
    mass := mp.constraints.SatelliteMass
    
    mf := mass * math.Exp(-deltaVms/(isp*g0))
    return mass - mf
}

// scoreOptions calculates overall score for each option
func (mp *ManeuverPlanner) scoreOptions(options []ManeuverOption) {
    for i := range options {
        opt := &options[i]
        
        // Score components (0-1 scale)
        pcScore := 0.0
        if opt.ResultingPc < 1e-7 {
            pcScore = 1.0
        } else if opt.ResultingPc < 1e-6 {
            pcScore = 0.9
        } else if opt.ResultingPc < 1e-5 {
            pcScore = 0.7
        } else if opt.ResultingPc < 1e-4 {
            pcScore = 0.4
        }
        
        fuelScore := 1.0 - opt.FuelPct/100
        if fuelScore < 0 {
            fuelScore = 0
        }
        
        secondaryScore := 1.0
        if opt.HasSecondaryRisk {
            secondaryScore = 0.5
        }
        
        // Weighted combination
        opt.Score = 0.5*pcScore + 0.3*fuelScore + 0.2*secondaryScore
    }
}

// rtnToECI converts RTN vector to ECI
func (mp *ManeuverPlanner) rtnToECI(rtn Vector3, state *StateVector, magnitude float64) Vector3 {
    // R: Radial (position direction)
    r := normalize(state.Position)
    
    // N: Normal (orbit normal = r × v)
    n := cross(state.Position, state.Velocity)
    n = normalize(n)
    
    // T: Along-track (N × R)
    t := cross(n, r)
    t = normalize(t)
    
    return Vector3{
        X: magnitude * (rtn.X*r.X + rtn.Y*t.X + rtn.Z*n.X),
        Y: magnitude * (rtn.X*r.Y + rtn.Y*t.Y + rtn.Z*n.Y),
        Z: magnitude * (rtn.X*r.Z + rtn.Y*t.Z + rtn.Z*n.Z),
    }
}

    }
    
    return delta
```

```go
// monitorEscalation handles escalation progression
func (as *AlertService) monitorEscalation(eventID string) {
    ticker := time.NewTicker(10 * time.Minute)
    defer ticker.Stop()
    
    for range ticker.C {
        state, ok := as.escalations[eventID]
        if !ok {
            return
        }
        
        if state.Acknowledged {
            delete(as.escalations, eventID)
            return
        }
        
        // Check if time for next escalation level
        elapsed := time.Since(state.StartTime)
        
        for i, level := range as.config.EscalationPath {
            if i > state.CurrentLevel && elapsed >= level.Delay {
                as.escalateAlert(eventID, i)
                state.CurrentLevel = i
            }
        }
        
        // Check if event has passed
        event, err := as.alertStore.GetEvent(eventID)
        if err != nil || event.TCA.Before(time.Now()) {
            delete(as.escalations, eventID)
            return
        }
    }
}

    password  string
    from      string
    templates *TemplateEngine

func (en *EmailNotifier) Channel() string { return "email" }

func (en *EmailNotifier) Send(ctx context.Context, alert *Alert) error {
    subject := fmt.Sprintf("[%s] Conjunction Alert: %s vs %s - TCA %s",
        alert.RiskLevel,
        alert.Satellite,
        alert.Secondary,
        formatDuration(alert.TimeToTCA),
    )
    
    body := en.templates.Render("conjunction_alert_email", map[string]interface{}{
        "Alert":    alert,
        "RiskColor": riskLevelColor(alert.RiskLevel),
    })
    
    auth := smtp.PlainAuth("", en.username, en.password, en.smtpHost)
    
    for _, recipient := range alert.Recipients {
        msg := fmt.Sprintf("From: %s\r\nTo: %s\r\nSubject: %s\r\nContent-Type: text/html\r\n\r\n%s",
            en.from, recipient, subject, body)
        
        addr := fmt.Sprintf("%s:%d", en.smtpHost, en.smtpPort)
        if err := smtp.SendMail(addr, auth, en.from, []string{recipient}, []byte(msg)); err != nil {
            return err
        }
    }
    
    return nil
}

type WebhookNotifier struct {
    webhooks []WebhookConfig
    client   *http.Client
}

type WebhookConfig struct {
    URL     string
    Secret  string
    Events  []string
    Headers map[string]string
}

func (wn *WebhookNotifier) Channel() string { return "webhook" }

func (wn *WebhookNotifier) Send(ctx context.Context, alert *Alert) error {
    payload := map[string]interface{}{
        "event_type":     "conjunction_alert",
        "event_id":       alert.EventID,
        "alert_id":       alert.ID,
        "alert_type":     alert.Type,
        "risk_level":     alert.RiskLevel.String(),
        "satellite":      alert.Satellite,
        "secondary":      alert.Secondary,
        "secondary_type": alert.SecondaryType,
        "tca":            alert.TCA.Format(time.RFC3339),
        "time_to_tca_sec": int(alert.TimeToTCA.Seconds()),
        "miss_distance_km": alert.MissDistance,
        "pc":             alert.Pc,
        "action":         alert.Action,
        "urgency":        alert.Urgency,
        "timestamp":      time.Now().Format(time.RFC3339),
    }
    
    jsonPayload, _ := json.Marshal(payload)
    
    for _, webhook := range wn.webhooks {
        req, err := http.NewRequestWithContext(ctx, "POST", webhook.URL, bytes.NewBuffer(jsonPayload))
        if err != nil {
            continue
        }
        
        req.Header.Set("Content-Type", "application/json")
        req.Header.Set("User-Agent", "P9E-Conjunction-Alerter/1.0")
        
        // Custom headers
        for k, v := range webhook.Headers {
            req.Header.Set(k, v)
        }
        
        // Sign payload
        if webhook.Secret != "" {
            signature := hmacSign(jsonPayload, webhook.Secret)
            req.Header.Set("X-Signature-256", "sha256="+signature)
        }
        
        resp, err := wn.client.Do(req)
        if err != nil {
            log.Error("Webhook delivery failed", "url", webhook.URL, "error", err)
            continue
        }
        resp.Body.Close()
        
        if resp.StatusCode >= 400 {
            log.Error("Webhook returned error", "url", webhook.URL, "status", resp.StatusCode)
        }
    }
    
    return nil
}

func hmacSign(payload []byte, secret string) string {
    h := hmac.New(sha256.New, []byte(secret))
    h.Write(payload)
    return hex.EncodeToString(h.Sum(nil))
}

func formatDuration(d time.Duration) string {
    if d < time.Hour {
        return fmt.Sprintf("%dm", int(d.Minutes()))
    }
    if d < 24*time.Hour {
        return fmt.Sprintf("%dh%dm", int(d.Hours()), int(d.Minutes())%60)
    }
    return fmt.Sprintf("%dd%dh", int(d.Hours())/24, int(d.Hours())%24)
}

func riskLevelColor(level RiskLevel) string {
    switch level {
    case RiskLevelGreen:
        return "#00ff00"
    case RiskLevelYellow:
        return "#ffff00"
    case RiskLevelOrange:
        return "#ffa500"
    case RiskLevelRed:
        return "#ff0000"
    }
    return "#808080"
}

func unique(strs []string) []string {
    seen := make(map[string]bool)
    result := make([]string, 0)
    for _, s := range strs {
        if !seen[s] {
            seen[s] = true
            result = append(result, s)
        }
    }
    return result
}
```
