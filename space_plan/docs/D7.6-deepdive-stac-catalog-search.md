## 1. Executive Summary

### 1.1 Purpose

The STAC Catalog & Search system provides a standards-compliant, high-performance geospatial data discovery and access platform for Earth Observation data:

- **Standards Compliance:** Full STAC 1.0 specification implementation
- **High-Performance Search:** Sub-second queries across millions of items
- **Spatial Intelligence:** Advanced geospatial queries with PostGIS and H3
- **Cloud-Native Access:** Direct access to Cloud-Optimized GeoTIFFs (COG)
- **Extensibility:** Support for EO, SAR, and custom extensions

| Metric | Target | Description |
|--------|--------|-------------|
| **Search Latency** | < 200ms | P95 for typical queries |

```go
// ParseDatetime parses STAC datetime string
// Formats: single instant, closed interval, open interval
// Examples: "2024-01-15T00:00:00Z"
//           "2024-01-01/2024-01-31"
//           "../2024-01-31" (before)
//           "2024-01-01/.." (after)
func ParseDatetime(s string) (*DatetimeRange, error) {
    parts := strings.Split(s, "/")
    
    result := &DatetimeRange{}
    
    if len(parts) == 1 {
        // Single instant
        t, err := time.Parse(time.RFC3339, parts[0])
        if err != nil {
            return nil, err
        }
        result.Start = &t
        result.End = &t
    } else if len(parts) == 2 {
        // Interval
        if parts[0] != ".." {
            t, _ := time.Parse(time.RFC3339, parts[0])
            result.Start = &t
        }
        if parts[1] != ".." {
            t, _ := time.Parse(time.RFC3339, parts[1])
            result.End = &t
        }
    }
    
    return result, nil
}
```

### 5.3 CQL2 Filter Support

```go
// CQL2Filter represents a CQL2 filter expression
type CQL2Filter struct {
    Op   string        `json:"op"`
    Args []interface{} `json:"args"`
}

// CQL2 operators
const (
    OpAnd           = "and"
    OpOr            = "or"
    OpNot           = "not"
    OpEqual         = "="
    OpNotEqual      = "<>"
    OpLess          = "<"
    OpLessEqual     = "<="
    OpGreater       = ">"
    OpGreaterEqual  = ">="
    OpLike          = "like"
    OpBetween       = "between"
    OpIn            = "in"
    OpIsNull        = "isNull"
    
    // Spatial operators
    OpSIntersects = "s_intersects"
    OpSContains   = "s_contains"
    OpSWithin     = "s_within"
    
    // Temporal operators
    OpTAfter      = "t_after"
    OpTBefore     = "t_before"
    OpTDuring     = "t_during"
    OpTIntersects = "t_intersects"
)

// CQL2ToSQL converts CQL2 filter to SQL WHERE clause
func (c *CQL2ToSQL) Convert(filter *CQL2Filter) (string, []interface{}, error) {
    switch filter.Op {
    case OpAnd:
        return c.convertLogical(filter, "AND")
    case OpOr:
        return c.convertLogical(filter, "OR")
    case OpEqual, OpNotEqual, OpLess, OpLessEqual, OpGreater, OpGreaterEqual:
        return c.convertComparison(filter)
    case OpSIntersects, OpSContains, OpSWithin:
        return c.convertSpatial(filter)
    case OpTAfter, OpTBefore, OpTDuring:
        return c.convertTemporal(filter)
    default:
        return "", nil, fmt.Errorf("unsupported operator: %s", filter.Op)
    }
}

// convertSpatial converts spatial operators to PostGIS
func (c *CQL2ToSQL) convertSpatial(filter *CQL2Filter) (string, []interface{}, error) {
    prop := extractProperty(filter.Args[0])
    geom, _ := json.Marshal(filter.Args[1])
    
    pgFunc := map[string]string{
        OpSIntersects: "ST_Intersects",
        OpSContains:   "ST_Contains",
        OpSWithin:     "ST_Within",
    }[filter.Op]
    
    return fmt.Sprintf("%s(%s, ST_GeomFromGeoJSON($1))", pgFunc, prop), 
           []interface{}{string(geom)}, nil
}
```

### 6.2 H3 Hexagonal Index

```go
// H3Indexer provides H3 hexagonal spatial indexing
type H3Indexer struct {
    resolutions []int // H3 resolutions to index (e.g., 4, 6, 8)
}

// IndexGeometry generates H3 indexes for a geometry
func (h *H3Indexer) IndexGeometry(geojson *GeoJSON) ([]H3Index, error) {
    indexes := make([]H3Index, 0)
    
    switch geojson.Type {
    case "Point":
        coords := geojson.Coordinates.([]float64)
        lat, lng := coords[1], coords[0]
        
        for _, res := range h.resolutions {
            cell := h3.LatLngToCell(h3.NewLatLng(lat, lng), res)
            indexes = append(indexes, H3Index{
                Resolution: res,
                CellID:     uint64(cell),
            })
        }
        
    case "Polygon", "MultiPolygon":
        for _, res := range h.resolutions {
            cells := h.polygonToCells(geojson, res)
            for _, cell := range cells {
                indexes = append(indexes, H3Index{
                    Resolution: res,
                    CellID:     uint64(cell),
                })
            }
        }
    }
    
    return indexes, nil
}

// SearchByH3 searches using H3 index for fast bbox queries
func (h *H3Indexer) SearchByH3(db *sql.DB, bbox []float64, resolution int) ([]string, error) {
    cells := h.bboxToCells(bbox, resolution)
    
    cellIDs := make([]uint64, len(cells))
    for i, cell := range cells {
        cellIDs[i] = uint64(cell)
    }
    
    query := `
        SELECT DISTINCT item_id 
        FROM stac_item_h3_index 
        WHERE resolution = $1 AND cell_id = ANY($2)
    `
    
    rows, err := db.Query(query, resolution, pq.Array(cellIDs))
    // ... process results
}
```

### 7.2 Temporal Query Optimization

```go
// TemporalQueryBuilder builds optimized temporal queries
type TemporalQueryBuilder struct {
    startTime   *time.Time
    endTime     *time.Time
    collections []string
}

// BuildQuery generates optimized SQL based on time range
func (tqb *TemporalQueryBuilder) BuildQuery() (string, []interface{}) {
    var conditions []string
    var params []interface{}
    
    if tqb.startTime != nil && tqb.endTime != nil {
        rangeSize := tqb.endTime.Sub(*tqb.startTime)
        
        if rangeSize < 24*time.Hour {
            // Very small range: exact B-tree lookup
            conditions = append(conditions, 
                "datetime >= $1 AND datetime < $2")
        } else if rangeSize < 7*24*time.Hour {
            // Small range: B-tree range scan
            conditions = append(conditions,
                "datetime BETWEEN $1 AND $2")
        } else {
            // Large range: BRIN + partition pruning
            conditions = append(conditions,
                "datetime >= $1 AND datetime <= $2")
        }
        params = append(params, tqb.startTime, tqb.endTime)
    }
    
    return strings.Join(conditions, " AND "), params
}
```

```sql
BEGIN
    NEW.search_vector = 
        setweight(to_tsvector('english', COALESCE(NEW.stac_id, '')), 'A') ||
        setweight(to_tsvector('english', COALESCE(NEW.properties->>'title', '')), 'A') ||
        setweight(to_tsvector('english', COALESCE(NEW.properties->>'description', '')), 'B') ||
        setweight(to_tsvector('english', COALESCE(NEW.properties->>'platform', '')), 'B') ||
        setweight(to_tsvector('english', COALESCE(NEW.properties->>'constellation', '')), 'C');
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_stac_item_search_vector
    BEFORE INSERT OR UPDATE ON stac_items

-- Search function with ranking
CREATE OR REPLACE FUNCTION search_stac_items(query_text TEXT, p_limit INTEGER DEFAULT 10)
RETURNS TABLE (id UUID, stac_id VARCHAR, rank REAL) AS $$
BEGIN
    RETURN QUERY
    SELECT si.id, si.stac_id,
           ts_rank(si.search_vector, plainto_tsquery('english', query_text)) as rank
    FROM stac_items si
    WHERE si.search_vector @@ plainto_tsquery('english', query_text)
    ORDER BY rank DESC
    LIMIT p_limit;
```

## 9. Query Optimization

### 9.1 Query Planner

```go
// QueryPlanner optimizes STAC queries
type QueryPlanner struct {
    db    *sql.DB
    stats *TableStats
}

// Plan creates optimized execution plan
func (qp *QueryPlanner) Plan(req *SearchRequest) (*ExecutionPlan, error) {
    plan := &ExecutionPlan{}
    
    // Estimate selectivity
    selectivity := qp.estimateSelectivity(req)
    
    // Choose strategy based on selectivity
    if selectivity < 0.001 {
        // Very selective: use index scan
        plan.Strategy = "INDEX_SCAN"
        plan.IndexHints = qp.selectBestIndexes(req)
    } else if selectivity < 0.1 {
        // Moderately selective: bitmap scan
        plan.Strategy = "BITMAP_SCAN"
        plan.Parallelism = 4
    } else {
        // Low selectivity: parallel seq scan
        plan.Strategy = "SEQ_SCAN"
        plan.Parallelism = 8
    }
    
    return plan, nil
}

// estimateSelectivity estimates query selectivity
func (qp *QueryPlanner) estimateSelectivity(req *SearchRequest) float64 {
    selectivity := 1.0
    
    // Collection filter
    if len(req.Collections) > 0 {
        for _, coll := range req.Collections {
            if count, ok := qp.stats.CollectionCounts[coll]; ok {
                selectivity *= float64(count) / float64(qp.stats.TotalItems)
            }
        }
    }
    
    // BBox filter (estimate by area ratio)
    if len(req.BBox) == 4 {
        bboxArea := (req.BBox[2] - req.BBox[0]) * (req.BBox[3] - req.BBox[1])
        totalArea := 360.0 * 180.0 // World coverage
        selectivity *= bboxArea / totalArea
    }
    
    // Datetime filter (estimate by time ratio)
    if req.Datetime != "" {
        dt, _ := ParseDatetime(req.Datetime)
        if dt != nil && dt.Start != nil && dt.End != nil {
            totalRange := qp.stats.DatetimeRange.End.Sub(*qp.stats.DatetimeRange.Start)
            queryRange := dt.End.Sub(*dt.Start)
            selectivity *= float64(queryRange) / float64(totalRange)
        }
    }
    
    return selectivity
}

// selectBestIndexes chooses optimal indexes for query
func (qp *QueryPlanner) selectBestIndexes(req *SearchRequest) []string {
    indexes := make([]string, 0)
    
    // Spatial: use GIST index
    if len(req.BBox) > 0 || req.Intersects != nil {
        indexes = append(indexes, "idx_stac_items_geometry")
    }
    
    // Temporal: choose B-tree or BRIN based on range
    if req.Datetime != "" {
        dt, _ := ParseDatetime(req.Datetime)
        if dt != nil {
            rangeSize := dt.End.Sub(*dt.Start)
            if rangeSize > 30*24*time.Hour {
                indexes = append(indexes, "idx_stac_items_datetime_brin")
            } else {
                indexes = append(indexes, "idx_stac_items_datetime")
            }
        }
    }
    
    // Collection filter
    if len(req.Collections) > 0 {
        indexes = append(indexes, "idx_stac_items_collection_datetime")
    }
    
    return indexes
}
```

### 9.2 Query Execution

```go
// QueryExecutor executes optimized STAC queries
type QueryExecutor struct {
    db      *sql.DB
    planner *QueryPlanner
    cache   *QueryCache
}

// ExecuteSearch executes a search request
func (qe *QueryExecutor) ExecuteSearch(ctx context.Context, req *SearchRequest) (*SearchResponse, error) {
    // Check cache
    cacheKey := req.CacheKey()
    if cached, ok := qe.cache.Get(cacheKey); ok {
        return cached, nil
    }
    
    // Create execution plan
    plan, _ := qe.planner.Plan(req)
    
    // Build SQL query
    query, params := qe.buildSearchQuery(req, plan)
    
    // Execute with appropriate parallelism
    var items []*Item
    if plan.Parallelism > 1 {
        items = qe.executeParallel(ctx, query, params, plan.Parallelism)
    } else {
        items = qe.executeSerial(ctx, query, params)
    }
    
    // Build response
    response := &SearchResponse{
        Type:     "FeatureCollection",
        Features: items,
        Context:  &SearchContext{Returned: len(items), Limit: req.Limit},
    }
    
    // Add pagination links
    if len(items) >= req.Limit {
        response.Links = append(response.Links, Link{
            Rel:  "next",
            Href: qe.buildNextLink(req, items),
        })
    }
    
    // Cache result
    qe.cache.Set(cacheKey, response, 5*time.Minute)
    
    return response, nil
}
```

```go
// ReadTile reads a specific tile using HTTP range request
func (cr *COGReader) ReadTile(tileX, tileY, overview int) ([]byte, error) {
    offset, size := cr.calculateTileRange(tileX, tileY, overview)
    
    req, _ := http.NewRequest("GET", cr.url, nil)
    req.Header.Set("Range", fmt.Sprintf("bytes=%d-%d", offset, offset+size-1))
    
    resp, err := cr.httpClient.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()
    
    return io.ReadAll(resp.Body)
}

// ReadRegion reads a region at best resolution for given pixel limit
func (cr *COGReader) ReadRegion(bbox []float64, maxPixels int) ([]byte, error) {
    overview := cr.selectOverview(bbox, maxPixels)
    tiles := cr.calculateTiles(bbox, overview)
    return cr.readAndMergeTiles(tiles, overview)
}

// selectOverview selects best overview level for pixel budget
func (cr *COGReader) selectOverview(bbox []float64, maxPixels int) int {
    for i, ov := range cr.header.Overviews {
        scale := float64(cr.header.Width) / float64(ov.Width)
        pixels := estimatePixels(bbox, cr.header, scale)
        if pixels <= float64(maxPixels) {
            return i
        }
    }
    return len(cr.header.Overviews) - 1 // Lowest resolution
}
```

```go
type AssetURLGenerator struct {

    signingKey []byte

}

// GenerateSignedURL creates a signed URL for asset access
func (aug *AssetURLGenerator) GenerateSignedURL(assetPath string, opts *SignOptions) (string, error) {
    u, _ := url.Parse(aug.baseURL)
    u.Path = assetPath
    
    // Add expiry
    expiry := time.Now().Add(aug.expiry)
    q := u.Query()
    q.Set("expires", fmt.Sprintf("%d", expiry.Unix()))
    
    // Add options (format, width, bands, etc.)
    if opts != nil {
        if opts.Format != "" {
            q.Set("format", opts.Format)
        }
        if opts.Width > 0 {
            q.Set("width", fmt.Sprintf("%d", opts.Width))
        }
        if len(opts.Bands) > 0 {
            q.Set("bands", strings.Join(opts.Bands, ","))
        }
    }
    
    // Generate HMAC signature
    stringToSign := fmt.Sprintf("%s?%s", u.Path, q.Encode())
    signature := aug.sign(stringToSign)
    q.Set("signature", signature)
    
    u.RawQuery = q.Encode()
    return u.String(), nil
}

// sign creates HMAC-SHA256 signature
func (aug *AssetURLGenerator) sign(data string) string {
    h := hmac.New(sha256.New, aug.signingKey)
    h.Write([]byte(data))
    return base64.URLEncoding.EncodeToString(h.Sum(nil))
}
```

## 14. Caching Strategy

### 14.1 Multi-Layer Cache

```go
// STACCache provides multi-layer caching
type STACCache struct {
    memory *MemoryCache // L1: In-memory LRU
    redis  *redis.Client // L2: Redis
    config *CacheConfig
}

// Get retrieves from cache (L1 → L2)
func (c *STACCache) Get(ctx context.Context, key string) (interface{}, bool) {
    // Try L1
    if val, ok := c.memory.Get(key); ok {
        return val, true
    }
    
    // Try L2
    if c.config.EnableL2 {
        if val, ok := c.getFromRedis(ctx, key); ok {
            c.memory.Set(key, val, c.config.MemoryTTL) // Promote to L1
            return val, true
        }
    }
    
    return nil, false
}

// Set stores in both layers
func (c *STACCache) Set(ctx context.Context, key string, value interface{}) {
    c.memory.Set(key, value, c.config.MemoryTTL)
    if c.config.EnableL2 {
        c.setInRedis(ctx, key, value, c.config.RedisTTL)
    }
}

// InvalidateCollection clears collection-related cache
func (c *STACCache) InvalidateCollection(ctx context.Context, collectionID string) {
    c.memory.Delete("collection:" + collectionID)
    c.memory.Delete("collections")
    
    if c.config.EnableL2 {
        c.redis.Del(ctx, "collection:"+collectionID, "collections")
        // Invalidate search results for this collection
        pattern := "search:*collection=" + collectionID + "*"
        keys, _ := c.redis.Keys(ctx, pattern).Result()
        if len(keys) > 0 {
            c.redis.Del(ctx, keys...)
        }
    }
}
```

```sql
-- Full-text search with ranking
CREATE OR REPLACE FUNCTION search_stac_items_text(
    query_text TEXT,
    p_collection_id VARCHAR DEFAULT NULL,
    p_limit INTEGER DEFAULT 100
)
RETURNS TABLE (
    id UUID,
    stac_id VARCHAR,
    collection_id VARCHAR,
    rank REAL
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        si.id,
        si.stac_id,
        si.collection_id,
        ts_rank(si.search_vector, plainto_tsquery('english', query_text)) as rank
    FROM stac_items si
    WHERE si.search_vector @@ plainto_tsquery('english', query_text)
    AND (p_collection_id IS NULL OR si.collection_id = p_collection_id)
    ORDER BY rank DESC
    LIMIT p_limit;
END;
$$ LANGUAGE plpgsql;

-- Spatial search with bbox
CREATE OR REPLACE FUNCTION search_stac_items_bbox(
    p_min_lon DOUBLE PRECISION,
    p_min_lat DOUBLE PRECISION,
    p_max_lon DOUBLE PRECISION,
    p_max_lat DOUBLE PRECISION,
    p_collection_id VARCHAR DEFAULT NULL,
    p_start_datetime TIMESTAMPTZ DEFAULT NULL,
    p_end_datetime TIMESTAMPTZ DEFAULT NULL,
    p_max_cloud_cover REAL DEFAULT NULL,
    p_limit INTEGER DEFAULT 100
)
RETURNS TABLE (
    id UUID,
    stac_id VARCHAR,
    collection_id VARCHAR,
    datetime TIMESTAMPTZ
) AS $$
DECLARE
    search_bbox GEOMETRY;
BEGIN
    search_bbox := ST_MakeEnvelope(p_min_lon, p_min_lat, p_max_lon, p_max_lat, 4326);
    
    RETURN QUERY
    SELECT 
        si.id,
        si.stac_id,
        si.collection_id,
        si.datetime
    FROM stac_items si
    WHERE ST_Intersects(si.geometry, search_bbox)
    AND (p_collection_id IS NULL OR si.collection_id = p_collection_id)
    AND (p_start_datetime IS NULL OR si.datetime >= p_start_datetime)
    AND (p_end_datetime IS NULL OR si.datetime <= p_end_datetime)
    AND (p_max_cloud_cover IS NULL OR si.cloud_cover <= p_max_cloud_cover)
    ORDER BY si.datetime DESC
    LIMIT p_limit;
END;
$$ LANGUAGE plpgsql;
```

```go
// Create inserts a new STAC item
func (s *ItemStore) Create(ctx context.Context, item *Item) error {
    geomJSON, _ := json.Marshal(item.Geometry)
    propsJSON, _ := json.Marshal(item.Properties)
    assetsJSON, _ := json.Marshal(item.Assets)
    linksJSON, _ := json.Marshal(item.Links)
    
    query := `
        INSERT INTO stac_items (
            stac_id, collection_id, geometry, 
            datetime, start_datetime, end_datetime,
            properties, assets, links, stac_extensions, tenant_id
        ) VALUES (
            $1, $2, ST_GeomFromGeoJSON($3),
            $4, $5, $6,
            $7, $8, $9, $10, $11
        )
        RETURNING id
    `
    
    // Extract datetime from properties
    datetime := s.extractDatetime(item.Properties)
    startDatetime := s.extractTime(item.Properties, "start_datetime")
    endDatetime := s.extractTime(item.Properties, "end_datetime")
    
    var id string
    err := s.db.QueryRowContext(ctx, query,
        item.ID,
        item.Collection,
        string(geomJSON),
        datetime,
        startDatetime,
        endDatetime,
        propsJSON,
        assetsJSON,
        linksJSON,
        pq.Array(item.StacExtensions),
        item.TenantID,
    ).Scan(&id)
    
    if err != nil {
        return fmt.Errorf("failed to create item: %w", err)
    }
    
    return nil
}

// Get retrieves a STAC item by ID
func (s *ItemStore) Get(ctx context.Context, collectionID, itemID string) (*Item, error) {
    query := `
        SELECT 
            id, stac_id, collection_id,
            ST_AsGeoJSON(geometry)::jsonb as geometry,
            array_to_json(ARRAY[ST_XMin(bbox), ST_YMin(bbox), ST_XMax(bbox), ST_YMax(bbox)]) as bbox,
            datetime, start_datetime, end_datetime,
            properties, assets, links, stac_extensions
        FROM stac_items
        WHERE collection_id = $1 AND stac_id = $2
    `
    
    var item Item
    var geomJSON, propsJSON, assetsJSON, linksJSON, bboxJSON []byte
    var extensions pq.StringArray
    
    err := s.db.QueryRowContext(ctx, query, collectionID, itemID).Scan(
        &item.DBId,
        &item.ID,
        &item.Collection,
        &geomJSON,
        &bboxJSON,
        &item.Datetime,
        &item.StartDatetime,
        &item.EndDatetime,
        &propsJSON,
        &assetsJSON,
        &linksJSON,
        &extensions,
    )
    
    if err == sql.ErrNoRows {
        return nil, ErrNotFound
    }
    if err != nil {
        return nil, fmt.Errorf("failed to get item: %w", err)
    }
    
    // Parse JSON fields
    json.Unmarshal(geomJSON, &item.Geometry)
    json.Unmarshal(bboxJSON, &item.BBox)
    json.Unmarshal(propsJSON, &item.Properties)
    json.Unmarshal(assetsJSON, &item.Assets)
    json.Unmarshal(linksJSON, &item.Links)
    item.StacExtensions = []string(extensions)
    item.Type = "Feature"
    item.StacVersion = "1.0.0"
    
    return &item, nil
}

// List retrieves items from a collection
func (s *ItemStore) List(ctx context.Context, collectionID string, opts *ListOptions) ([]*Item, string, error) {
    query := s.buildListQuery(collectionID, opts)
    
    rows, err := s.db.QueryContext(ctx, query.SQL, query.Params...)
    if err != nil {
        return nil, "", fmt.Errorf("failed to list items: %w", err)
    }
    defer rows.Close()
    
    items := make([]*Item, 0)
    var lastCursor string
    
    for rows.Next() {
        item, cursor, err := s.scanItem(rows)
        if err != nil {
            return nil, "", err
        }
        items = append(items, item)
        lastCursor = cursor
    }
    
    // Check if there are more results
    var nextToken string
    if len(items) >= opts.Limit {
        nextToken = lastCursor
    }
    
    return items, nextToken, nil
}

// Search performs a STAC search
func (s *ItemStore) Search(ctx context.Context, req *SearchRequest) ([]*Item, int64, string, error) {
    // Build search query
    query := s.buildSearchQuery(req)
    
    // Execute count query (optional)
    var matched int64
    if req.IncludeMatched {
        countQuery := s.buildCountQuery(req)
        s.db.QueryRowContext(ctx, countQuery.SQL, countQuery.Params...).Scan(&matched)
    }
    
    // Execute search query
    rows, err := s.db.QueryContext(ctx, query.SQL, query.Params...)
    if err != nil {
        return nil, 0, "", fmt.Errorf("search failed: %w", err)
    }
    defer rows.Close()
    
    items := make([]*Item, 0)
    var lastCursor string
    
    for rows.Next() {
        item, cursor, err := s.scanItem(rows)
        if err != nil {
            return nil, 0, "", err
        }
        items = append(items, item)
        lastCursor = cursor
    }
    
    // Build next token
    var nextToken string
    if len(items) >= req.Limit {
        nextToken = lastCursor
    }
    
    return items, matched, nextToken, nil
}

// buildSearchQuery constructs the search SQL
func (s *ItemStore) buildSearchQuery(req *SearchRequest) *Query {
    var conditions []string
    var params []interface{}
    paramIdx := 1
    
    // Collection filter
    if len(req.Collections) > 0 {
        if len(req.Collections) == 1 {
            conditions = append(conditions, fmt.Sprintf("collection_id = $%d", paramIdx))
            params = append(params, req.Collections[0])
        } else {
            conditions = append(conditions, fmt.Sprintf("collection_id = ANY($%d)", paramIdx))
            params = append(params, pq.Array(req.Collections))
        }
        paramIdx++
    }
    
    // ID filter
    if len(req.IDs) > 0 {
        conditions = append(conditions, fmt.Sprintf("stac_id = ANY($%d)", paramIdx))
        params = append(params, pq.Array(req.IDs))
        paramIdx++
    }
    
    // BBox filter
    if len(req.BBox) >= 4 {
        conditions = append(conditions, fmt.Sprintf(
            "geometry && ST_MakeEnvelope($%d, $%d, $%d, $%d, 4326)",
            paramIdx, paramIdx+1, paramIdx+2, paramIdx+3))
        params = append(params, req.BBox[0], req.BBox[1], req.BBox[2], req.BBox[3])
        paramIdx += 4
    }
    
    // Intersects filter
    if req.Intersects != nil {
        geojson, _ := json.Marshal(req.Intersects)
        conditions = append(conditions, fmt.Sprintf(
            "ST_Intersects(geometry, ST_GeomFromGeoJSON($%d))", paramIdx))
        params = append(params, string(geojson))
        paramIdx++
    }
    
    // Datetime filter
    if req.Datetime != "" {
        dtRange, _ := ParseDatetime(req.Datetime)
        if dtRange != nil {
            if dtRange.Start != nil && dtRange.End != nil {
                conditions = append(conditions, fmt.Sprintf(
                    "datetime BETWEEN $%d AND $%d", paramIdx, paramIdx+1))
                params = append(params, dtRange.Start, dtRange.End)
                paramIdx += 2
            } else if dtRange.Start != nil {
                conditions = append(conditions, fmt.Sprintf("datetime >= $%d", paramIdx))
                params = append(params, dtRange.Start)
                paramIdx++
            } else if dtRange.End != nil {
                conditions = append(conditions, fmt.Sprintf("datetime <= $%d", paramIdx))
                params = append(params, dtRange.End)
                paramIdx++
            }
        }
    }
    
    // CQL2 filter
    if req.Filter != nil {
        converter := NewCQL2ToSQL(paramIdx)
        filterSQL, filterParams, _ := converter.Convert(req.Filter)
        if filterSQL != "" {
            conditions = append(conditions, filterSQL)
            params = append(params, filterParams...)
            paramIdx += len(filterParams)
        }
    }
    
    // Cursor pagination
    if req.Token != "" {
        cursor, _ := DecodeCursor(req.Token)
        if cursor != nil {
            conditions = append(conditions, fmt.Sprintf(
                "(datetime, id) < ($%d, $%d)", paramIdx, paramIdx+1))
            params = append(params, cursor.Datetime, cursor.ID)
            paramIdx += 2
        }
    }
    
    // Build WHERE clause
    whereClause := ""
    if len(conditions) > 0 {
        whereClause = "WHERE " + strings.Join(conditions, " AND ")
    }
    
    // Build ORDER BY
    orderBy := "ORDER BY datetime DESC, id DESC"
    if len(req.Sortby) > 0 {
        sorts := make([]string, 0)
        for _, spec := range req.Sortby {
            dir := "ASC"
            if spec.Direction == "desc" {
                dir = "DESC"
            }
            sorts = append(sorts, fmt.Sprintf("%s %s", s.mapSortField(spec.Field), dir))
        }
        orderBy = "ORDER BY " + strings.Join(sorts, ", ")
    }
    
    // Build LIMIT
    limit := 10
    if req.Limit > 0 && req.Limit <= 1000 {
        limit = req.Limit
    }
    
    // Build SELECT with field filtering
    selectFields := s.buildSelectFields(req.Fields)
    
    sql := fmt.Sprintf(`
        SELECT %s
        FROM stac_items
        %s
        %s
        LIMIT %d
    `, selectFields, whereClause, orderBy, limit+1)
    
    return &Query{SQL: sql, Params: params}
}

// buildSelectFields builds SELECT clause based on field filtering
func (s *ItemStore) buildSelectFields(fields *FieldsSpec) string {
    baseFields := `
        id, stac_id, collection_id,
        ST_AsGeoJSON(geometry)::jsonb as geometry,
        array_to_json(ARRAY[ST_XMin(bbox), ST_YMin(bbox), ST_XMax(bbox), ST_YMax(bbox)]) as bbox,
        datetime,
        CONCAT(datetime::text, ':', id::text) as cursor
    `
    
    if fields == nil {
        return baseFields + ", properties, assets, links, stac_extensions"
    }
    
    // Handle include/exclude logic
    includedFields := make(map[string]bool)
    
    if len(fields.Include) > 0 {
        for _, f := range fields.Include {
            includedFields[f] = true
        }
    } else {
        // Include all by default
        includedFields["properties"] = true
        includedFields["assets"] = true
        includedFields["links"] = true
    }
    
    for _, f := range fields.Exclude {
        delete(includedFields, f)
    }
    
    var additionalFields []string
    if includedFields["properties"] {
        additionalFields = append(additionalFields, "properties")
    }
    if includedFields["assets"] {
        additionalFields = append(additionalFields, "assets")
    }
    if includedFields["links"] {
        additionalFields = append(additionalFields, "links")
    }
    
    if len(additionalFields) > 0 {
        return baseFields + ", " + strings.Join(additionalFields, ", ") + ", stac_extensions"
    }
    
    return baseFields + ", stac_extensions"
}

// BulkCreate inserts multiple items efficiently
func (s *ItemStore) BulkCreate(ctx context.Context, collectionID string, items []*Item) (int, []error) {
    tx, err := s.db.BeginTx(ctx, nil)
    if err != nil {
        return 0, []error{err}
    }
    defer tx.Rollback()
    
    stmt, err := tx.PrepareContext(ctx, `
        INSERT INTO stac_items (
            stac_id, collection_id, geometry, 
            datetime, properties, assets, links, 
            stac_extensions, tenant_id
        ) VALUES (
            $1, $2, ST_GeomFromGeoJSON($3),
            $4, $5, $6, $7, $8, $9
        )
    `)
    if err != nil {
        return 0, []error{err}
    }
    defer stmt.Close()
    
    created := 0
    var errors []error
    
    for _, item := range items {
        geomJSON, _ := json.Marshal(item.Geometry)
        propsJSON, _ := json.Marshal(item.Properties)
        assetsJSON, _ := json.Marshal(item.Assets)
        linksJSON, _ := json.Marshal(item.Links)
        datetime := s.extractDatetime(item.Properties)
        
        _, err := stmt.ExecContext(ctx,
            item.ID,
            collectionID,
            string(geomJSON),
            datetime,
            propsJSON,
            assetsJSON,
            linksJSON,
            pq.Array(item.StacExtensions),
            item.TenantID,
        )
        
        if err != nil {
            errors = append(errors, fmt.Errorf("item %s: %w", item.ID, err))
        } else {
            created++
        }
    }
    
    if err := tx.Commit(); err != nil {
        return 0, []error{err}
    }
    
    return created, errors
}

// Query represents a constructed SQL query
type Query struct {
    SQL    string
    Params []interface{}
}

// Cursor represents pagination cursor
type Cursor struct {
    Datetime time.Time
    ID       string
}

// EncodeCursor encodes cursor to string
func EncodeCursor(c *Cursor) string {
    return base64.StdEncoding.EncodeToString(
        []byte(fmt.Sprintf("%s:%s", c.Datetime.Format(time.RFC3339Nano), c.ID)))
}

// DecodeCursor decodes cursor from string
func DecodeCursor(s string) (*Cursor, error) {
    data, err := base64.StdEncoding.DecodeString(s)
    if err != nil {
        return nil, err
    }
    
    parts := strings.SplitN(string(data), ":", 2)
    if len(parts) != 2 {
        return nil, fmt.Errorf("invalid cursor format")
    }
    
    dt, err := time.Parse(time.RFC3339Nano, parts[0])
    if err != nil {
        return nil, err
    }
    
    return &Cursor{Datetime: dt, ID: parts[1]}, nil
}
```

```go
package stac

// AggregationService provides STAC aggregation capabilities
type AggregationService struct {

// NewAggregationService creates a new aggregation service
func NewAggregationService(db *sql.DB) *AggregationService {
    return &AggregationService{db: db}
}

// Aggregate performs aggregation queries
func (s *AggregationService) Aggregate(ctx context.Context, req *AggregateRequest) (*AggregateResponse, error) {
    response := &AggregateResponse{
        Aggregations: make(map[string]*Aggregation),
    }
    
    for _, aggSpec := range req.Aggregations {
        agg, err := s.executeAggregation(ctx, req, aggSpec)
        if err != nil {
            return nil, fmt.Errorf("aggregation %s failed: %w", aggSpec.Name, err)
        }
        response.Aggregations[aggSpec.Name] = agg
    }
    
    return response, nil
}

// executeAggregation executes a single aggregation
func (s *AggregationService) executeAggregation(ctx context.Context, req *AggregateRequest, spec *AggregationSpec) (*Aggregation, error) {
    switch spec.Type {
    case AggTypeTerms:
        return s.termsAggregation(ctx, req, spec)
    case AggTypeDateHistogram:
        return s.dateHistogramAggregation(ctx, req, spec)
    case AggTypeStats:
        return s.statsAggregation(ctx, req, spec)
    case AggTypeGeohashGrid:
        return s.geohashGridAggregation(ctx, req, spec)
    case AggTypeGeotileGrid:
        return s.geotileGridAggregation(ctx, req, spec)
    default:
        return nil, fmt.Errorf("unsupported aggregation type: %s", spec.Type)
    }
}

// termsAggregation performs terms aggregation
func (s *AggregationService) termsAggregation(ctx context.Context, req *AggregateRequest, spec *AggregationSpec) (*Aggregation, error) {
    whereClause, params := s.buildWhereClause(req)
    
    field := s.mapAggField(spec.Field)
    size := spec.Size
    if size == 0 {
        size = 10
    }
    
    query := fmt.Sprintf(`
        SELECT %s as key, COUNT(*) as count
        FROM stac_items
        %s
        GROUP BY 1
        ORDER BY 2 DESC
        LIMIT %d
    `, field, whereClause, size)
    
    rows, err := s.db.QueryContext(ctx, query, params...)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    
    agg := &Aggregation{
        Name:    spec.Name,
        Buckets: make([]*Bucket, 0),
    }
    
    for rows.Next() {
        var bucket Bucket
        err := rows.Scan(&bucket.Key, &bucket.Count)
        if err != nil {
            return nil, err
        }
        agg.Buckets = append(agg.Buckets, &bucket)
    }
    
    return agg, nil
}

// dateHistogramAggregation performs date histogram aggregation
func (s *AggregationService) dateHistogramAggregation(ctx context.Context, req *AggregateRequest, spec *AggregationSpec) (*Aggregation, error) {
    whereClause, params := s.buildWhereClause(req)
    
    interval := spec.CalendarInterval
    if interval == "" {
        interval = "month"
    }
    
    query := fmt.Sprintf(`
        SELECT 
            date_trunc('%s', datetime) as key,
            COUNT(*) as count
        FROM stac_items
        %s
        GROUP BY 1
        ORDER BY 1
    `, interval, whereClause)
    
    rows, err := s.db.QueryContext(ctx, query, params...)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    
    agg := &Aggregation{
        Name:    spec.Name,
        Buckets: make([]*Bucket, 0),
    }
    
    for rows.Next() {
        var key time.Time
        var count int64
        err := rows.Scan(&key, &count)
        if err != nil {
            return nil, err
        }
        agg.Buckets = append(agg.Buckets, &Bucket{
            Key:   key.Format(time.RFC3339),
            Count: count,
        })
    }
    
    return agg, nil
}

// statsAggregation performs statistics aggregation
func (s *AggregationService) statsAggregation(ctx context.Context, req *AggregateRequest, spec *AggregationSpec) (*Aggregation, error) {
    whereClause, params := s.buildWhereClause(req)
    
    field := s.mapAggField(spec.Field)
    
    query := fmt.Sprintf(`
        SELECT 
            COUNT(*) as count,
            MIN(%s::numeric) as min,

// geohashGridAggregation performs geohash grid aggregation
func (s *AggregationService) geohashGridAggregation(ctx context.Context, req *AggregateRequest, spec *AggregationSpec) (*Aggregation, error) {

        FROM stac_items

// geotileGridAggregation performs geotile grid aggregation (XYZ tiles)
func (s *AggregationService) geotileGridAggregation(ctx context.Context, req *AggregateRequest, spec *AggregationSpec) (*Aggregation, error) {
    whereClause, params := s.buildWhereClause(req)
    
    zoom := spec.Zoom
    if zoom == 0 {
        zoom = 6
    }
    
    // Convert to tile coordinates using mercator projection
    query := fmt.Sprintf(`
        WITH tile_coords AS (
            SELECT 
                floor((ST_X(centroid) + 180) / 360 * power(2, %d))::int as tile_x,
                floor((1 - ln(tan(radians(ST_Y(centroid))) + 1/cos(radians(ST_Y(centroid)))) / pi()) / 2 * power(2, %d))::int as tile_y,
                id
            FROM stac_items
            %s
        )
        SELECT 
            %d || '/' || tile_x || '/' || tile_y as key,
            COUNT(*) as count
        FROM tile_coords
        GROUP BY tile_x, tile_y
        ORDER BY 2 DESC
        LIMIT 1000
    `, zoom, zoom, whereClause, zoom)
    
    rows, err := s.db.QueryContext(ctx, query, params...)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    
    agg := &Aggregation{

// buildWhereClause builds WHERE clause from request
func (s *AggregationService) buildWhereClause(req *AggregateRequest) (string, []interface{}) {
    var conditions []string
    var params []interface{}
    paramIdx := 1
    
    if len(req.Collections) > 0 {
        conditions = append(conditions, fmt.Sprintf("collection_id = ANY($%d)", paramIdx))
        params = append(params, pq.Array(req.Collections))
        paramIdx++
    }
    
    if len(req.BBox) >= 4 {
        conditions = append(conditions, fmt.Sprintf(
            "geometry && ST_MakeEnvelope($%d, $%d, $%d, $%d, 4326)",
            paramIdx, paramIdx+1, paramIdx+2, paramIdx+3))
        params = append(params, req.BBox[0], req.BBox[1], req.BBox[2], req.BBox[3])
        paramIdx += 4
    }
    
    if req.Datetime != "" {
        dtRange, _ := ParseDatetime(req.Datetime)
        if dtRange != nil && dtRange.Start != nil && dtRange.End != nil {
            conditions = append(conditions, fmt.Sprintf(
                "datetime BETWEEN $%d AND $%d", paramIdx, paramIdx+1))
            params = append(params, dtRange.Start, dtRange.End)
            paramIdx += 2
        }
    }
    
    if len(conditions) > 0 {
        return "WHERE " + strings.Join(conditions, " AND "), params
    }
    
    return "", params
}
```

```go
    m.DBQueryDuration.WithLabelValues(queryType).Observe(duration)
```

```go
// health.go

package stac

import (
    "context"
    "time"
)

// HealthChecker provides health check functionality
type HealthChecker struct {
    db    *sql.DB
    cache *STACCache
}

// HealthStatus represents service health
type HealthStatus struct {
    Status     string            `json:"status"`
    Checks     map[string]Check  `json:"checks"`
    Timestamp  time.Time         `json:"timestamp"`
}

// Check represents individual health check
type Check struct {
    Status   string  `json:"status"`
    Duration float64 `json:"duration_ms"`
    Message  string  `json:"message,omitempty"`
}

// Check performs health check
func (h *HealthChecker) Check(ctx context.Context) *HealthStatus {
    status := &HealthStatus{
        Status:    "healthy",
        Checks:    make(map[string]Check),
        Timestamp: time.Now(),
    }
    
    // Database check
    dbCheck := h.checkDatabase(ctx)
    status.Checks["database"] = dbCheck
    if dbCheck.Status != "healthy" {
        status.Status = "unhealthy"
    }
    
    // Cache check
    cacheCheck := h.checkCache(ctx)
    status.Checks["cache"] = cacheCheck
    if cacheCheck.Status != "healthy" && status.Status == "healthy" {
        status.Status = "degraded"
    }
    
    // Search index check
    indexCheck := h.checkSearchIndex(ctx)
    status.Checks["search_index"] = indexCheck
    if indexCheck.Status != "healthy" && status.Status == "healthy" {
        status.Status = "degraded"
    }
    
    return status
}

// checkDatabase checks database connectivity
func (h *HealthChecker) checkDatabase(ctx context.Context) Check {
    start := time.Now()
    
    err := h.db.PingContext(ctx)
    duration := time.Since(start).Seconds() * 1000
    
    if err != nil {
        return Check{
            Status:   "unhealthy",
            Duration: duration,
            Message:  err.Error(),
        }
    }
    
    // Check query performance
    var count int
    err = h.db.QueryRowContext(ctx, "SELECT COUNT(*) FROM stac_items LIMIT 1").Scan(&count)
    if err != nil {
        return Check{
            Status:   "degraded",
            Duration: duration,
            Message:  "Query check failed: " + err.Error(),
        }
    }
    
    return Check{
        Status:   "healthy",
        Duration: duration,
    }
}

// checkCache checks cache connectivity
func (h *HealthChecker) checkCache(ctx context.Context) Check {
    start := time.Now()
    
    // Try to get a test key
    _, _ = h.cache.Get(ctx, "health_check")
    
    // Try to set a test key
    h.cache.Set(ctx, "health_check", "ok")
    
    duration := time.Since(start).Seconds() * 1000
    
    return Check{
        Status:   "healthy",
        Duration: duration,
    }
}

// checkSearchIndex checks search index health
func (h *HealthChecker) checkSearchIndex(ctx context.Context) Check {
    start := time.Now()

    err := h.db.QueryRowContext(ctx, `
        SELECT COUNT(*) FROM stac_items 
        WHERE search_vector @@ plainto_tsquery('test')
        LIMIT 1
    `).Scan(&count)
    
    duration := time.Since(start).Seconds() * 1000
    
    if err != nil {
        return Check{
            Status:   "degraded",
            Duration: duration,
            Message:  "Search index check failed",
        }
    }
    
    return Check{
        Status:   "healthy",
        Duration: duration,
    }
```
