```go
// go/internal/satellite/service/flight_dynamics.go

package service

import (
    "context"
    
    "p9e.space/satellite/domain"
    "p9e.space/satellite/julia"  // Julia interop
)

// FlightDynamicsService coordinates flight dynamics calculations
type FlightDynamicsService struct {
    repo        domain.OrbitRepository
    juliaClient *julia.Client  // gRPC client to Julia service
}

// PropagateOrbit delegates to Julia for numerical propagation
func (s *FlightDynamicsService) PropagateOrbit(ctx context.Context, req *PropagateRequest) (*PropagateResponse, error) {
    // Get TLE from database
    tle, err := s.repo.GetLatestTLE(ctx, req.SpacecraftID)
    if err != nil {
        return nil, err
    }
    
    // Delegate to Julia for SGP4/HPOP propagation
    result, err := s.juliaClient.Propagate(ctx, &julia.PropagateRequest{
        TLE:       tle.ToJulia(),
        StartTime: req.StartTime,
        EndTime:   req.EndTime,
        StepSize:  req.StepSize,
        Method:    req.PropagatorType,  // sgp4, sdp4, hpop
    })
    if err != nil {
        return nil, err
    }
    
    // Store ephemeris
    if err := s.repo.StoreEphemeris(ctx, req.SpacecraftID, result.States); err != nil {
        return nil, err
    }
    
    return &PropagateResponse{
        States: result.States,
    }, nil
}

// AssessConjunction delegates to Julia for collision assessment
func (s *FlightDynamicsService) AssessConjunction(ctx context.Context, req *ConjunctionRequest) (*ConjunctionAssessment, error) {
    return s.juliaClient.AssessConjunction(ctx, &julia.ConjunctionRequest{
        PrimaryTLE:   req.Primary.TLE,
        SecondaryTLE: req.Secondary.TLE,
        TCA:          req.TCA,
        Covariances:  req.Covariances,
        MonteCarloN:  10000,  // Julia handles Monte Carlo
    })
}
```

```rust
// rust/crates/p9e-telemetry/src/ccsds.rs

use bytes::{Buf, Bytes};

/// CCSDS Space Packet Header
#[derive(Debug, Clone)]
pub struct SpacePacketHeader {
    pub version: u8,
    pub packet_type: PacketType,
    pub secondary_header: bool,
    pub apid: u16,
    pub sequence_flags: SequenceFlags,
    pub sequence_count: u16,
    pub data_length: u16,
}

impl SpacePacketHeader {
    /// Parse CCSDS primary header (6 bytes)
    pub fn parse(buf: &mut Bytes) -> Result<Self, ParseError> {
        if buf.remaining() < 6 {
            return Err(ParseError::InsufficientData);
        }
        
        let word1 = buf.get_u16();
        let word2 = buf.get_u16();
        let word3 = buf.get_u16();
        
        Ok(Self {
            version: ((word1 >> 13) & 0x07) as u8,
            packet_type: PacketType::from((word1 >> 12) & 0x01),
            secondary_header: (word1 >> 11) & 0x01 == 1,
            apid: word1 & 0x07FF,
            sequence_flags: SequenceFlags::from((word2 >> 14) & 0x03),
            sequence_count: word2 & 0x3FFF,
            data_length: word3,
        })
    }
}

/// High-performance telemetry frame processor
pub struct FrameProcessor {
    sync_pattern: [u8; 4],
    frame_length: usize,
    parameter_map: HashMap<u16, ParameterDefinition>,
}

impl FrameProcessor {
    /// Process telemetry at high throughput (target: 100k frames/sec)
    pub fn process_batch(&self, frames: &[Frame]) -> Vec<TelemetryPoint> {
        frames.par_iter()  // Rayon parallel iterator
            .flat_map(|frame| self.decommutate(frame))
            .collect()
    }
    
    fn decommutate(&self, frame: &Frame) -> Vec<TelemetryPoint> {
        let header = SpacePacketHeader::parse(&mut frame.data.clone()).unwrap();
        
        self.parameter_map.get(&header.apid)
            .map(|params| self.extract_parameters(&frame.data, params))
            .unwrap_or_default()
    }
}
```

```rust
use wasm_bindgen::prelude::*;

/// SGP4 propagator compiled to WebAssembly for browser use
#[wasm_bindgen]
pub struct SGP4Propagator {
    satrec: sgp4::Satrec,
}

#[wasm_bindgen]
impl SGP4Propagator {
    /// Create from TLE lines
    #[wasm_bindgen(constructor)]
    pub fn new(line1: &str, line2: &str) -> Result<SGP4Propagator, JsValue> {
        let satrec = sgp4::Satrec::from_tle(line1, line2)
            .map_err(|e| JsValue::from_str(&e.to_string()))?;
        
        Ok(Self { satrec })
    }
    
    /// Propagate to time (minutes since epoch)
    #[wasm_bindgen]
    pub fn propagate(&mut self, tsince: f64) -> Result<Position, JsValue> {
        let result = sgp4::propagate(&mut self.satrec, tsince)
            .map_err(|e| JsValue::from_str(&e.to_string()))?;
        
        Ok(Position {
            x: result.position.x,
            y: result.position.y,
            z: result.position.z,
            vx: result.velocity.x,
            vy: result.velocity.y,
            vz: result.velocity.z,
        })
    }
    
    /// Propagate to array of times (batch processing)
    #[wasm_bindgen]
    pub fn propagate_batch(&mut self, times: &[f64]) -> Vec<f64> {
        times.iter()
            .flat_map(|&t| {
                if let Ok(result) = sgp4::propagate(&mut self.satrec, t) {
                    vec![
                        result.position.x,
                        result.position.y,
                        result.position.z,
                    ]
                } else {
                    vec![f64::NAN, f64::NAN, f64::NAN]
                }
            })
            .collect()
    }
}

#[wasm_bindgen]
pub struct Position {
    pub x: f64,
    pub y: f64,
    pub z: f64,
    pub vx: f64,
    pub vy: f64,
    pub vz: f64,
}
```

```julia
# julia/src/P9eCASA/src/collision_probability.jl

module CollisionProbability

using LinearAlgebra
using Distributions
using StaticArrays

export chan_2d_pc, monte_carlo_pc, alfano_3d_pc

"""
Chan's 2D collision probability method.

For short-term encounters where relative velocity is nearly constant.
"""
function chan_2d_pc(
    r_primary::SVector{3,Float64},
    r_secondary::SVector{3,Float64},
    v_rel::SVector{3,Float64},
    cov_primary::SMatrix{3,3,Float64},
    cov_secondary::SMatrix{3,3,Float64},
    combined_radius::Float64
)
    # Miss distance
    miss = r_secondary - r_primary
    miss_dist = norm(miss)
    
    # Combined covariance in encounter plane
    cov_combined = cov_primary + cov_secondary
    
    # Project to encounter plane (perpendicular to relative velocity)
    v_unit = v_rel / norm(v_rel)
    
    # Rotation to encounter frame
    R = _encounter_rotation(v_unit)
    
    # 2D covariance in encounter plane
    cov_2d = R[1:2, :] * cov_combined * R[1:2, :]'
    
    # Miss distance in encounter plane
    miss_2d = (R * miss)[1:2]
    
    # Compute 2D Gaussian integral over hard-body circle
    pc = _integrate_2d_gaussian(miss_2d, cov_2d, combined_radius)
    
    return pc
end

"""
Monte Carlo collision probability estimation.

Most accurate for complex scenarios.
"""
function monte_carlo_pc(
    state_primary::SVector{6,Float64},
    state_secondary::SVector{6,Float64},
    cov_primary::SMatrix{6,6,Float64},
    cov_secondary::SMatrix{6,6,Float64},
    combined_radius::Float64;
    n_samples::Int = 100_000
)
    # Sample primary states
    dist_primary = MvNormal(Vector(state_primary), Matrix(cov_primary))
    samples_primary = rand(dist_primary, n_samples)
    
    # Sample secondary states
    dist_secondary = MvNormal(Vector(state_secondary), Matrix(cov_secondary))
    samples_secondary = rand(dist_secondary, n_samples)
    
    # Count collisions (parallelized)
    collisions = Threads.Atomic{Int}(0)
    
    Threads.@threads for i in 1:n_samples
        r1 = @view samples_primary[1:3, i]
        r2 = @view samples_secondary[1:3, i]
        
        if norm(r1 - r2) < combined_radius
            Threads.atomic_add!(collisions, 1)
        end
    end
    
    return collisions[] / n_samples
end

end # module
```

```julia
module Simulation

using Dates
using DifferentialEquations
using StaticArrays

export MissionSimulator, SpacecraftModel, simulate!, MonteCarloRunner

"""
Mission Simulation Framework

Hybrid discrete-event + continuous-time simulation.
"""
mutable struct MissionSimulator
    # Simulation state
    current_time::DateTime
    end_time::DateTime
    time_step::Float64  # seconds
    
    # Models
    spacecraft::Vector{SpacecraftModel}
    ground_stations::Vector{GroundStation}
    environment::SpaceEnvironment
    
    # Event queue
    event_queue::PriorityQueue{SimEvent}
    
    # Results
    history::SimulationHistory
end

"""
Spacecraft model with subsystem simulation.
"""
struct SpacecraftModel
    # Identity
    id::String
    name::String
    
    # State
    orbit_state::MVector{6,Float64}  # Mutable for integration
    attitude::Quaternion
    
    # Subsystems
    power::PowerSubsystem
    thermal::ThermalSubsystem
    adcs::ADCSSubsystem
    comms::CommsSubsystem
end

"""
Run simulation to end time.
"""
function simulate!(sim::MissionSimulator)
    while sim.current_time < sim.end_time
        # Process events at current time
        while !isempty(sim.event_queue) && 
              peek(sim.event_queue).time <= sim.current_time
            event = dequeue!(sim.event_queue)
            _process_event!(sim, event)
        end
        
        # Advance continuous dynamics
        _advance_dynamics!(sim, sim.time_step)
        
        # Record history
        _record_state!(sim)
        
        # Advance time
        sim.current_time += Second(sim.time_step)
    end
    
    return sim.history
end

"""
Monte Carlo simulation runner for statistical analysis.
"""
struct MonteCarloRunner
    base_scenario::MissionSimulator
    n_runs::Int
    parameter_distributions::Dict{Symbol, Distribution}
end

function run_monte_carlo(runner::MonteCarloRunner)
    results = Vector{SimulationHistory}(undef, runner.n_runs)
    
    # Parallel Monte Carlo runs
    Threads.@threads for i in 1:runner.n_runs
        # Clone base scenario
        sim = deepcopy(runner.base_scenario)
        
        # Sample parameters
        _apply_parameter_samples!(sim, runner.parameter_distributions)
        
        # Run simulation
        results[i] = simulate!(sim)
    end
    
    return MonteCarloResults(results)
end
```

```typescript
    await init();
    wasmReady = true;

  return new SGP4Propagator(tle1, tle2);
}

export function propagateOrbit(
  propagator: SGP4Propagator,
  startTime: Date,
  endTime: Date,
  stepMinutes: number
): Float64Array {
  const epochTime = getEpochTime(propagator);
  const times: number[] = [];
  
  let current = startTime.getTime();
  const end = endTime.getTime();
  const step = stepMinutes * 60 * 1000;
  
  while (current <= end) {
    times.push((current - epochTime) / 60000); // minutes since epoch
    current += step;
  }
  
  // Batch propagation in WASM (fast)
  return propagator.propagate_batch(new Float64Array(times));
}
```

```typescript
    // Scene setup
    this.scene = new THREE.Scene();
    this.scene.background = new THREE.Color(0x000011);
    
    // Camera
    this.camera = new THREE.PerspectiveCamera(
      75,
      container.clientWidth / container.clientHeight,
      0.1,
      1000
    );
    this.camera.position.set(5, 3, 5);
    
    // Renderer
    this.renderer = new THREE.WebGLRenderer({ antialias: true });
    this.renderer.setSize(container.clientWidth, container.clientHeight);
    this.renderer.setPixelRatio(window.devicePixelRatio);
    container.appendChild(this.renderer.domElement);
    
    // Controls
    this.controls = new OrbitControls(this.camera, this.renderer.domElement);
    this.controls.enableDamping = true;
    
    // Lighting
    const ambientLight = new THREE.AmbientLight(0x404040);
    this.scene.add(ambientLight);
    
    const sunLight = new THREE.DirectionalLight(0xffffff, 1);
    sunLight.position.set(10, 10, 10);
    this.scene.add(sunLight);
    
    // Reference frame axes
    const axesHelper = new THREE.AxesHelper(3);
    this.scene.add(axesHelper);
    
    // Start render loop
    this.animate();
  }
  
  /**
   * Load spacecraft 3D model (GLTF)
   */
  async loadModel(url: string): Promise<void> {
    const loader = new GLTFLoader();
    
    return new Promise((resolve, reject) => {
      loader.load(
        url,
        (gltf) => {
          this.spacecraft = gltf.scene;
          this.scene.add(this.spacecraft);
          resolve();
        },
        undefined,
        reject
      );
    });
  }
  
  /**
   * Update spacecraft attitude (quaternion)
   */
  setAttitude(q: { w: number; x: number; y: number; z: number }) {
    this.attitude.set(q.x, q.y, q.z, q.w);
    if (this.spacecraft) {
      this.spacecraft.quaternion.copy(this.attitude);
    }
  }
  
  /**
   * Visualize solar panel orientation
   */
  showSolarPanels(panelAngles: number[]) {
    // Find solar panel meshes and rotate
    if (!this.spacecraft) return;
    
    this.spacecraft.traverse((child) => {
      if (child.name.startsWith('solar_panel_')) {
        const index = parseInt(child.name.split('_')[2]);
        if (index < panelAngles.length) {
          child.rotation.x = panelAngles[index] * Math.PI / 180;
        }
      }
    });
  }
  
  /**
   * Visualize antenna beam pattern
   */
  showAntennaPattern(
    antennaName: string,
    beamWidth: number,
    direction: THREE.Vector3
  ) {
    const geometry = new THREE.ConeGeometry(
      Math.tan(beamWidth * Math.PI / 180) * 5,
      5,
      32
    );
    const material = new THREE.MeshBasicMaterial({
      color: 0x00ff00,
      transparent: true,
      opacity: 0.3,
      wireframe: true,
    });
    
    const cone = new THREE.Mesh(geometry, material);
    cone.position.copy(direction.multiplyScalar(2.5));
    cone.lookAt(new THREE.Vector3(0, 0, 0));
    
    this.scene.add(cone);
  }
  
  /**
   * Show thermal heat map
   */
  showThermalMap(temperatures: Map<string, number>) {
    if (!this.spacecraft) return;
    
    this.spacecraft.traverse((child) => {
      if (child instanceof THREE.Mesh) {
        const temp = temperatures.get(child.name);
        if (temp !== undefined) {
          // Map temperature to color (blue = cold, red = hot)
          const normalized = (temp + 40) / 120; // -40°C to +80°C
          const color = new THREE.Color();
          color.setHSL(0.7 - normalized * 0.7, 1, 0.5);
          
          (child.material as THREE.MeshStandardMaterial).color = color;
        }
      }
    });
  }
  
  private animate() {
    requestAnimationFrame(() => this.animate());
    this.controls.update();
    this.renderer.render(this.scene, this.camera);
  }
  
  dispose() {
    this.renderer.dispose();
  }
}
```

```go
// go/internal/telemetry/rust/parser.go

package rust

/*
#cgo LDFLAGS: -L${SRCDIR}/../../../rust/target/release -lp9e_telemetry
#include <stdint.h>
#include <stdlib.h>

typedef struct {
    double timestamp;
    uint16_t parameter_id;
    double value;
    uint8_t quality;
} TelemetryPoint;

typedef struct {
    TelemetryPoint* points;
    size_t count;
    char* error;
} ParseResult;

extern ParseResult parse_ccsds_frame(const uint8_t* data, size_t len);
extern void free_parse_result(ParseResult* result);
*/
import "C"

import (
    "unsafe"
    
    "p9e.space/telemetry/domain"
)

// RustParser wraps Rust telemetry parser via CGO
type RustParser struct{}

// ParseFrame parses CCSDS frame using Rust (high performance)
func (p *RustParser) ParseFrame(data []byte) ([]domain.TelemetryPoint, error) {
    cData := (*C.uint8_t)(unsafe.Pointer(&data[0]))
    cLen := C.size_t(len(data))
    
    result := C.parse_ccsds_frame(cData, cLen)
    defer C.free_parse_result(&result)
    
    if result.error != nil {
        return nil, fmt.Errorf("rust parser error: %s", C.GoString(result.error))
    }
    
    points := make([]domain.TelemetryPoint, result.count)
    cPoints := unsafe.Slice(result.points, result.count)
    
    for i, cp := range cPoints {
        points[i] = domain.TelemetryPoint{
            Timestamp:   time.UnixMilli(int64(cp.timestamp * 1000)),
            ParameterID: uint16(cp.parameter_id),
            Value:       float64(cp.value),
            Quality:     domain.TelemetryQuality(cp.quality),
        }
    }
    
    return points, nil
}
```

### 10.3 Go ↔ Julia Integration

```go
// go/internal/satellite/julia/client.go

package julia

import (
    "context"
    
    pb "p9e.space/gen/julia/v1"
    "google.golang.org/grpc"
)

// Client wraps gRPC connection to Julia service
type Client struct {
    conn   *grpc.ClientConn
    astro  pb.AstrodynamicsServiceClient
    sim    pb.SimulationServiceClient
}

// NewClient creates Julia client
func NewClient(addr string) (*Client, error) {
    conn, err := grpc.Dial(addr, grpc.WithInsecure())
    if err != nil {
        return nil, err
    }
    
    return &Client{
        conn:  conn,
        astro: pb.NewAstrodynamicsServiceClient(conn),
        sim:   pb.NewSimulationServiceClient(conn),
    }, nil
}

// Propagate delegates orbit propagation to Julia
func (c *Client) Propagate(ctx context.Context, req *PropagateRequest) (*PropagateResponse, error) {
    pbReq := &pb.PropagateRequest{
        TleLine1:  req.TLE.Line1,
        TleLine2:  req.TLE.Line2,
        StartTime: timestamppb.New(req.StartTime),
        EndTime:   timestamppb.New(req.EndTime),
        StepSize:  req.StepSize.Seconds(),
        Method:    pb.PropagatorType(req.Method),
    }
    
    pbResp, err := c.astro.Propagate(ctx, pbReq)
    if err != nil {
        return nil, err
    }
    
    states := make([]OrbitState, len(pbResp.States))
    for i, s := range pbResp.States {
        states[i] = OrbitState{
            Time:     s.Time.AsTime(),
            Position: [3]float64{s.X, s.Y, s.Z},
            Velocity: [3]float64{s.Vx, s.Vy, s.Vz},
        }
    }
    
    return &PropagateResponse{States: states}, nil
}

// AssessConjunction delegates conjunction assessment to Julia
func (c *Client) AssessConjunction(ctx context.Context, req *ConjunctionRequest) (*ConjunctionAssessment, error) {
    pbReq := &pb.ConjunctionRequest{
        PrimaryTle:     req.PrimaryTLE,
        SecondaryTle:   req.SecondaryTLE,
        Tca:            timestamppb.New(req.TCA),
        CombinedRadius: req.CombinedRadius,
        MonteCarloN:    int32(req.MonteCarloN),
    }
    
    // Set covariances
    for _, cov := range req.PrimaryCov {
        pbReq.PrimaryCovariance = append(pbReq.PrimaryCovariance, cov...)
    }
    
    pbResp, err := c.astro.AssessConjunction(ctx, pbReq)
    if err != nil {
        return nil, err
    }
    
    return &ConjunctionAssessment{
        Pc:              pbResp.CollisionProbability,
        MissDistance:    pbResp.MissDistance,
        RelativeVelocity: pbResp.RelativeVelocity,
        Method:          pbResp.Method,
    }, nil
}
```

```yaml
      - uses: actions/checkout@v4

      - run: cd rust && cargo build --release
      - run: cd rust && cargo test
      - run: cd rust && cargo build --release --target wasm32-unknown-unknown -p p9e-wasm
      - uses: actions/upload-artifact@v4

          name: wasm-modules
```
