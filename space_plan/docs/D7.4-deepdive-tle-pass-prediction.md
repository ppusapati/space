## 1. Executive Summary

### 1.1 Purpose

The TLE & Pass Prediction subsystem provides the mathematical foundation for satellite tracking and pass scheduling. It enables:

- **TLE Parsing:** Decode Two-Line Element sets from various sources
- **Orbital Propagation:** Predict satellite positions using SGP4/SDP4
- **Pass Prediction:** Calculate visibility windows for ground stations
- **Tracking Data:** Generate azimuth, elevation, range, and Doppler
- **Scheduling Support:** Provide data for automated pass scheduling

### 1.2 Key Algorithms

| Algorithm | Purpose | Accuracy |
|-----------|---------|----------|
| **SGP4** | LEO propagation (period < 225 min) | ~1 km at epoch, degrades |
| **SDP4** | Deep space propagation (period ≥ 225 min) | ~1-10 km |
| **Pass Finder** | Visibility window calculation | < 1 second timing |
| **Doppler** | Frequency shift prediction | < 100 Hz at S-band |

### 1.3 Design Goals

| Goal | Target | Rationale |
|------|--------|-----------|
| **Prediction Speed** | 1000 passes/sec | Bulk scheduling |
| **Position Accuracy** | < 5 km (fresh TLE) | Antenna pointing |
| **Timing Accuracy** | < 1 second AOS/LOS | Pass scheduling |
| **Update Latency** | < 1 min TLE refresh | Operational accuracy |

### 2.3 TLE Parser Implementation

```go
// tle_parser.go

package orbital

import (
    "fmt"
    "strconv"
    "strings"
    "time"
)

// TLE represents a parsed Two-Line Element set
type TLE struct {
    // Identification
    Name           string
    CatalogNumber  int
    Classification string
    IntlDesignator string
    
    // Epoch
    Epoch          time.Time
    EpochYear      int
    EpochDay       float64
    
    // Mean motion derivatives
    MeanMotionDot  float64  // rev/day²
    MeanMotionDDot float64  // rev/day³
    
    // Drag
    BStar          float64  // 1/earth-radii
    
    // Orbital elements
    Inclination    float64  // radians
    RAAN           float64  // radians (Right Ascension of Ascending Node)
    Eccentricity   float64  // dimensionless
    ArgPerigee     float64  // radians
    MeanAnomaly    float64  // radians
    MeanMotion     float64  // rad/min
    
    // Derived
    RevNumber      int
    ElementSet     int
    
    // Metadata
    RawLine1       string
    RawLine2       string
    ParsedAt       time.Time
    Source         string
}

// ParseTLE parses a TLE from two or three lines
func ParseTLE(lines []string) (*TLE, error) {
    var name, line1, line2 string
    
    switch len(lines) {
    case 2:
        line1, line2 = lines[0], lines[1]
    case 3:
        name, line1, line2 = lines[0], lines[1], lines[2]
    default:
        return nil, fmt.Errorf("expected 2 or 3 lines, got %d", len(lines))
    }
    
    // Validate line lengths
    if len(line1) < 69 || len(line2) < 69 {
        return nil, fmt.Errorf("line too short")
    }
    
    // Validate line numbers
    if line1[0] != '1' || line2[0] != '2' {
        return nil, fmt.Errorf("invalid line numbers")
    }
    
    // Validate checksums
    if !validateChecksum(line1) {
        return nil, fmt.Errorf("line 1 checksum invalid")
    }
    if !validateChecksum(line2) {
        return nil, fmt.Errorf("line 2 checksum invalid")
    }
    
    tle := &TLE{
        Name:      strings.TrimSpace(name),
        RawLine1:  line1,
        RawLine2:  line2,
        ParsedAt:  time.Now().UTC(),
    }
    
    // Parse Line 1
    tle.CatalogNumber = parseInt(line1[2:7])
    tle.Classification = string(line1[7])
    tle.IntlDesignator = strings.TrimSpace(line1[9:17])
    
    // Parse epoch
    tle.EpochYear = parseInt(line1[18:20])
    tle.EpochDay = parseFloat(line1[20:32])
    tle.Epoch = epochToTime(tle.EpochYear, tle.EpochDay)
    
    // Parse mean motion derivatives
    tle.MeanMotionDot = parseFloat(line1[33:43])
    tle.MeanMotionDDot = parseScientific(line1[44:52])
    
    // Parse BSTAR drag term
    tle.BStar = parseScientific(line1[53:61])
    
    tle.ElementSet = parseInt(line1[64:68])
    
    // Parse Line 2
    catalogNumber2 := parseInt(line2[2:7])
    if catalogNumber2 != tle.CatalogNumber {
        return nil, fmt.Errorf("catalog number mismatch")
    }
    
    // Parse orbital elements (convert to radians)
    tle.Inclination = degToRad(parseFloat(line2[8:16]))
    tle.RAAN = degToRad(parseFloat(line2[17:25]))
    tle.Eccentricity = parseFloat("0." + line2[26:33])
    tle.ArgPerigee = degToRad(parseFloat(line2[34:42]))
    tle.MeanAnomaly = degToRad(parseFloat(line2[43:51]))
    
    // Mean motion: convert from rev/day to rad/min
    meanMotionRevDay := parseFloat(line2[52:63])
    tle.MeanMotion = meanMotionRevDay * (2.0 * math.Pi / 1440.0)
    
    tle.RevNumber = parseInt(line2[63:68])
    
    return tle, nil
}

// validateChecksum verifies TLE line checksum
func validateChecksum(line string) bool {
    if len(line) < 69 {
        return false
    }
    
    sum := 0
    for i := 0; i < 68; i++ {
        c := line[i]
        if c >= '0' && c <= '9' {
            sum += int(c - '0')
        } else if c == '-' {
            sum += 1
        }
    }
    
    checksum := int(line[68] - '0')
    return (sum % 10) == checksum
}

// parseScientific parses TLE scientific notation (e.g., "12345-4" = 0.12345e-4)
func parseScientific(s string) float64 {
    s = strings.TrimSpace(s)
    if len(s) == 0 {
        return 0
    }
    
    // Handle sign
    sign := 1.0
    if s[0] == '-' {
        sign = -1.0
        s = s[1:]
    } else if s[0] == '+' {
        s = s[1:]
    }
    
    // Find exponent marker
    expIdx := strings.IndexAny(s, "+-")
    if expIdx == -1 {
        // No exponent
        val, _ := strconv.ParseFloat("0."+s, 64)
        return sign * val
    }
    
    mantissa := "0." + s[:expIdx]
    exponent := s[expIdx:]
    
    val, _ := strconv.ParseFloat(mantissa+"e"+exponent, 64)
    return sign * val
}

// epochToTime converts TLE epoch to time.Time
func epochToTime(year2digit int, dayOfYear float64) time.Time {
    // Convert 2-digit year to 4-digit
    var year int
    if year2digit >= 57 {
        year = 1900 + year2digit
    } else {
        year = 2000 + year2digit
    }
    
    // Start of year
    jan1 := time.Date(year, 1, 1, 0, 0, 0, 0, time.UTC)
    
    // Add days (dayOfYear is 1-indexed)
    days := dayOfYear - 1.0
    duration := time.Duration(days * 24 * float64(time.Hour))
    
    return jan1.Add(duration)
}

// Age returns the age of this TLE
func (tle *TLE) Age() time.Duration {
    return time.Since(tle.Epoch)
}

// IsStale returns true if TLE is older than threshold
func (tle *TLE) IsStale(threshold time.Duration) bool {
    return tle.Age() > threshold
}

// OrbitalPeriod returns the orbital period in minutes
func (tle *TLE) OrbitalPeriod() float64 {
    return 2.0 * math.Pi / tle.MeanMotion
}

// SemiMajorAxis returns semi-major axis in km
func (tle *TLE) SemiMajorAxis() float64 {
    // a = (μ / n²)^(1/3)
    mu := 398600.4418 // km³/s² - Earth gravitational parameter
    n := tle.MeanMotion / 60.0 // rad/s
    return math.Pow(mu/(n*n), 1.0/3.0)
}

// Apogee returns apogee altitude in km
func (tle *TLE) Apogee() float64 {
    a := tle.SemiMajorAxis()
    return a*(1+tle.Eccentricity) - EarthRadius
}

// Perigee returns perigee altitude in km
func (tle *TLE) Perigee() float64 {
    a := tle.SemiMajorAxis()
    return a*(1-tle.Eccentricity) - EarthRadius
}
```

### 3.2 SGP4 Algorithm Flow

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                        SGP4 ALGORITHM FLOW                                  │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ STEP 1: INITIALIZATION (Once per TLE)                               │   │
│  │                                                                     │   │
│  │ • Recover original mean motion and semi-major axis                 │   │
│  │ • Determine near-earth vs deep-space (period < 225 min)            │   │
│  │ • Calculate constants: C1, C4, C5, D2, D3, D4, t2cof, etc.        │   │
│  │ • For deep-space: Initialize lunar/solar terms                     │   │
│  │                                                                     │   │
│  └────────────────────────────────┬────────────────────────────────────┘   │
│                                   │                                         │
│                                   ▼                                         │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ STEP 2: SECULAR EFFECTS (For each propagation time)                │   │
│  │                                                                     │   │
│  │ Update mean elements for secular drag and gravitation:             │   │
│  │                                                                     │   │
│  │ Ω = Ω₀ + Ω̇·Δt          (Right Ascension of Ascending Node)        │   │
│  │ ω = ω₀ + ω̇·Δt          (Argument of Perigee)                      │   │
│  │ M = M₀ + n₀·Δt + ...   (Mean Anomaly)                              │   │
│  │ e = e₀ + ė·Δt          (Eccentricity - drag effect)               │   │
│  │ a = a₀ + ȧ·Δt          (Semi-major axis - drag effect)            │   │
│  │                                                                     │   │
│  └────────────────────────────────┬────────────────────────────────────┘   │
│                                   │                                         │
│                                   ▼                                         │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ STEP 3: PERIODIC EFFECTS                                            │   │
│  │                                                                     │   │
│  │ Add long-period and short-period perturbations:                    │   │
│  │                                                                     │   │
│  │ • Long-period: Variations with orbital period                      │   │
│  │ • Short-period: Variations within one orbit                        │   │
│  │                                                                     │   │
│  │ For deep-space: Add lunar and solar perturbations                  │   │
│  │                                                                     │   │
│  └────────────────────────────────┬────────────────────────────────────┘   │
│                                   │                                         │
│                                   ▼                                         │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ STEP 4: KEPLER'S EQUATION                                           │   │
│  │                                                                     │   │
│  │ Solve for Eccentric Anomaly E:                                     │   │
│  │                                                                     │   │
│  │   M = E - e·sin(E)     (Iterative solution)                        │   │
│  │                                                                     │   │
│  │ Newton-Raphson iteration:                                          │   │
│  │   E_{n+1} = E_n - (E_n - e·sin(E_n) - M) / (1 - e·cos(E_n))       │   │
│  │                                                                     │   │
│  └────────────────────────────────┬────────────────────────────────────┘   │
│                                   │                                         │
│                                   ▼                                         │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ STEP 5: POSITION & VELOCITY IN ORBITAL PLANE                       │   │
│  │                                                                     │   │
│  │ True anomaly: ν = 2·atan(√((1+e)/(1-e))·tan(E/2))                  │   │
│  │ Distance: r = a·(1 - e·cos(E))                                     │   │
│  │                                                                     │   │
│  │ Position in orbital plane:                                         │   │
│  │   x = r·cos(ν)                                                     │   │
│  │   y = r·sin(ν)                                                     │   │
│  │                                                                     │   │
│  │ Velocity in orbital plane:                                         │   │
│  │   vx = -√(μ/p)·sin(ν)                                              │   │
│  │   vy = √(μ/p)·(e + cos(ν))                                         │   │
│  │                                                                     │   │
│  └────────────────────────────────┬────────────────────────────────────┘   │
│                                   │                                         │
│                                   ▼                                         │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │ STEP 6: ROTATION TO TEME COORDINATES                               │   │
│  │                                                                     │   │
│  │ Rotate from orbital plane to TEME using:                           │   │
│  │ • Argument of perigee (ω)                                          │   │
│  │ • Inclination (i)                                                  │   │
│  │ • Right Ascension of Ascending Node (Ω)                            │   │
│  │                                                                     │   │
│  │ R = R_z(-Ω) · R_x(-i) · R_z(-ω)                                    │   │
│  │                                                                     │   │
│  │ [x]      [x_orbital]                                                │   │
│  │ [y]  = R·[y_orbital]                                                │   │
│  │ [z]      [0        ]                                                │   │
│  │                                                                     │   │
│  └─────────────────────────────────────────────────────────────────────┘   │
│                                                                             │
│  OUTPUT: Position (x, y, z) in km, Velocity (vx, vy, vz) in km/s          │
│          Reference frame: TEME (True Equator, Mean Equinox)               │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 3.3 SGP4 Implementation

```go
// sgp4.go

package orbital

import (
    "math"
)

// Earth constants (WGS-84)
const (
    EarthRadius     = 6378.137          // km
    EarthFlattening = 1.0 / 298.257223563
    EarthMu         = 398600.4418       // km³/s²
    J2              = 1.082616e-3       // Second zonal harmonic
    J3              = -2.53881e-6       // Third zonal harmonic
    J4              = -1.65597e-6       // Fourth zonal harmonic
    XKE             = 0.0743669161      // sqrt(μ) in earth-radii³/min²
    XKMPER          = 6378.137          // Earth radius in km
    XPDOTP          = 1440.0 / (2.0 * math.Pi) // Minutes per radian
    MinutesPerDay   = 1440.0
)

// SGP4Propagator propagates TLE using SGP4/SDP4
type SGP4Propagator struct {
    tle       *TLE
    deepSpace bool
    
    // Initialization constants
    a0, n0    float64  // Recovered semi-major axis and mean motion
    c1, c4, c5 float64 // Drag coefficients
    d2, d3, d4 float64 // Higher-order drag terms
    t2cof, t3cof, t4cof, t5cof float64
    
    // J2-J4 related constants
    x1mth2, x3thm1, x7thm1 float64
    xlcof, aycof float64
    delmo, sinmo float64
    
    // Deep-space constants (if applicable)
    dsConstants *DeepSpaceConstants
}

// StateVector represents position and velocity
type StateVector struct {
    Position Vector3 // km in TEME
    Velocity Vector3 // km/s in TEME
    Time     time.Time
}

// Vector3 represents a 3D vector
type Vector3 struct {
    X, Y, Z float64
}

// NewSGP4Propagator initializes a propagator for a TLE
func NewSGP4Propagator(tle *TLE) (*SGP4Propagator, error) {
    p := &SGP4Propagator{
        tle: tle,
    }
    
    // Determine near-earth vs deep-space
    periodMinutes := tle.OrbitalPeriod()
    p.deepSpace = periodMinutes >= 225.0
    
    // Initialize SGP4 constants
    if err := p.initialize(); err != nil {
        return nil, err
    }
    
    return p, nil
}

// initialize computes the propagator constants
func (p *SGP4Propagator) initialize() error {
    // Unpack TLE elements
    n0 := p.tle.MeanMotion
    e0 := p.tle.Eccentricity
    i0 := p.tle.Inclination
    
    // Compute initial values
    cosio := math.Cos(i0)
    sinio := math.Sin(i0)
    theta2 := cosio * cosio
    
    p.x3thm1 = 3.0*theta2 - 1.0
    p.x1mth2 = 1.0 - theta2
    p.x7thm1 = 7.0*theta2 - 1.0
    
    // Recover semi-major axis from mean motion
    // Solve Kepler's third law with J2 perturbation
    a1 := math.Pow(XKE/n0, 2.0/3.0)
    
    betao2 := 1.0 - e0*e0
    betao := math.Sqrt(betao2)
    
    // First-order correction for J2
    delta1 := 1.5 * J2 * (3.0*theta2 - 1.0) / (betao2 * betao) / (a1 * a1)
    
    a0 := a1 * (1.0 - delta1*(1.0/3.0 + delta1*(1.0 + 134.0/81.0*delta1)))
    delta0 := 1.5 * J2 * (3.0*theta2 - 1.0) / (betao2 * betao) / (a0 * a0)
    
    p.n0 = n0 / (1.0 + delta0)
    p.a0 = a0 / (1.0 - delta0)
    
    // Compute drag and harmonic coefficients
    coef := p.tle.BStar / (1.0 + delta0)
    
    // ... (extensive constant calculations omitted for brevity)
    // Full implementation includes C1-C5, D2-D4, etc.
    
    if p.deepSpace {
        p.dsConstants = p.initializeDeepSpace()
    }
    
    return nil
}

// Propagate computes position and velocity at a given time
func (p *SGP4Propagator) Propagate(t time.Time) (*StateVector, error) {
    // Time since epoch in minutes
    tsince := t.Sub(p.tle.Epoch).Minutes()
    
    return p.PropagateMinutes(tsince)
}

// PropagateMinutes computes position and velocity at tsince minutes from epoch
func (p *SGP4Propagator) PropagateMinutes(tsince float64) (*StateVector, error) {
    // Secular effects of atmospheric drag and gravitation
    xmdf := p.tle.MeanAnomaly + p.n0*tsince
    omgadf := p.tle.ArgPerigee + p.computeOmegaDot()*tsince
    xnode := p.tle.RAAN + p.computeRAANDot()*tsince
    
    // For drag
    e := p.tle.Eccentricity
    a := p.a0
    
    if !p.deepSpace {
        // Near-earth drag effects
        delm := p.c1 * tsince * tsince
        delomg := p.d2 * tsince * tsince
        
        xmdf += delm
        omgadf += delomg
        
        // Secular decay
        a = p.a0 * math.Pow(1.0-p.c1*tsince, 2.0)
        e = p.tle.Eccentricity - p.tle.BStar*p.c4*tsince
    } else {
        // Deep-space perturbations
        xmdf, omgadf, xnode, e, a = p.deepSpaceSecular(tsince, xmdf, omgadf, xnode, e, a)
    }
    
    // Clamp eccentricity
    if e < 1e-6 {
        e = 1e-6
    }
    if e > 0.999 {
        e = 0.999
    }
    
    // Long-period periodics
    xlp := p.computeLongPeriodics(e, a, omgadf, xnode)
    
    // Solve Kepler's equation
    meanAnomaly := math.Mod(xmdf+xlp, 2.0*math.Pi)
    eccAnomaly := solveKepler(meanAnomaly, e)
    
    // Short-period periodics
    r, rdot, rfdot := p.computeShortPeriodics(e, a, eccAnomaly)
    
    // Orientation vectors
    sinuk := math.Sin(omgadf + xlp)
    cosuk := math.Cos(omgadf + xlp)
    sinik := math.Sin(p.tle.Inclination)
    cosik := math.Cos(p.tle.Inclination)
    sinnok := math.Sin(xnode)
    cosnok := math.Cos(xnode)
    
    // Unit vectors in orbital plane
    mx := -sinnok*cosik
    my := cosnok*cosik
    mz := sinik
    
    nx := cosnok
    ny := sinnok
    nz := 0.0
    
    ux := mx*sinuk + nx*cosuk
    uy := my*sinuk + ny*cosuk
    uz := mz * sinuk
    
    vx := mx*cosuk - nx*sinuk
    vy := my*cosuk - ny*sinuk
    vz := mz * cosuk
    
    // Position and velocity in TEME (km, km/s)
    pos := Vector3{
        X: r * ux * XKMPER,
        Y: r * uy * XKMPER,
        Z: r * uz * XKMPER,
    }
    
    vel := Vector3{
        X: (rdot*ux + rfdot*vx) * XKMPER / 60.0, // Convert from ER/min to km/s
        Y: (rdot*uy + rfdot*vy) * XKMPER / 60.0,
        Z: (rdot*uz + rfdot*vz) * XKMPER / 60.0,
    }
    
    return &StateVector{
        Position: pos,
        Velocity: vel,
        Time:     p.tle.Epoch.Add(time.Duration(tsince) * time.Minute),
    }, nil
}

// solveKepler solves Kepler's equation using Newton-Raphson
func solveKepler(M, e float64) float64 {
    // Initial guess
    E := M
    if e > 0.8 {
        E = math.Pi
    }
    
    // Newton-Raphson iteration
    for i := 0; i < 10; i++ {
        delta := (E - e*math.Sin(E) - M) / (1.0 - e*math.Cos(E))
        E -= delta
        
        if math.Abs(delta) < 1e-12 {
            break
        }
    }
    
    return E
}

// computeOmegaDot returns argument of perigee rate (rad/min)
func (p *SGP4Propagator) computeOmegaDot() float64 {
    e0 := p.tle.Eccentricity
    i0 := p.tle.Inclination
    
    cosio := math.Cos(i0)
    betao2 := 1.0 - e0*e0
    betao := math.Sqrt(betao2)
    
    // J2 effect on argument of perigee
    return -1.5 * J2 * p.n0 * (1.0 - 5.0*cosio*cosio) / (betao2 * betao * p.a0 * p.a0)
}

// computeRAANDot returns RAAN rate (rad/min)
func (p *SGP4Propagator) computeRAANDot() float64 {
    e0 := p.tle.Eccentricity
    i0 := p.tle.Inclination
    
    cosio := math.Cos(i0)
    betao2 := 1.0 - e0*e0
    betao := math.Sqrt(betao2)
    
    // J2 effect on RAAN
    return -1.5 * J2 * p.n0 * cosio / (betao2 * betao * p.a0 * p.a0)
}
```

```go
// pass_finder.go

package orbital

import (
    "math"
    "sort"
    "time"
)

// Pass represents a satellite pass over a ground station
type Pass struct {
    Satellite    string
    Station      string
    
    // Key times
    AOS          time.Time   // Acquisition of Signal
    LOS          time.Time   // Loss of Signal
    TCA          time.Time   // Time of Closest Approach (max elevation)
    
    // Angles at key times
    AOSAzimuth   float64     // degrees
    LOSAzimuth   float64     // degrees
    TCAAnzimuth  float64     // degrees
    MaxElevation float64     // degrees
    
    // Derived
    Duration     time.Duration
    
    // Optional detailed points
    Points       []PassPoint
}

// PassPoint represents a point in a pass trajectory
type PassPoint struct {
    Time      time.Time
    Azimuth   float64  // degrees
    Elevation float64  // degrees
    Range     float64  // km
    RangeRate float64  // km/s
    Doppler   float64  // Hz (for a given frequency)
}

// PassFinderConfig configures the pass finder
type PassFinderConfig struct {
    MinElevation  float64       // Minimum elevation threshold (degrees)
    CoarseStep    time.Duration // Coarse search step
    FineStep      time.Duration // Fine search step
    Precision     time.Duration // AOS/LOS precision
    PointInterval time.Duration // Interval for detailed points
}

// DefaultPassFinderConfig returns default configuration
func DefaultPassFinderConfig() PassFinderConfig {
    return PassFinderConfig{
        MinElevation:  5.0,                 // 5 degrees minimum
        CoarseStep:    60 * time.Second,    // 1 minute
        FineStep:      10 * time.Second,    // 10 seconds
        Precision:     100 * time.Millisecond, // 0.1 second
        PointInterval: 10 * time.Second,    // Point every 10 seconds
    }
}

// PassFinder finds satellite passes over ground stations
type PassFinder struct {
    propagator *SGP4Propagator
    station    *GroundStation
    config     PassFinderConfig
}

// GroundStation represents a ground station location
type GroundStation struct {
    ID        string
    Name      string
    Latitude  float64  // degrees
    Longitude float64  // degrees
    Altitude  float64  // meters above sea level
    
    // Precomputed
    ECEFPosition Vector3
}

// NewPassFinder creates a new pass finder
func NewPassFinder(propagator *SGP4Propagator, station *GroundStation, config PassFinderConfig) *PassFinder {
    return &PassFinder{
        propagator: propagator,
        station:    station,
        config:     config,
    }
}

// FindPasses finds all passes in the given time window
func (pf *PassFinder) FindPasses(start, end time.Time) ([]Pass, error) {
    passes := make([]Pass, 0)
    
    // Calculate appropriate step based on orbital period
    period := pf.propagator.tle.OrbitalPeriod()
    step := pf.config.CoarseStep
    if period < 100 { // Fast-moving LEO satellite
        step = 30 * time.Second
    }
    
    // Coarse search: find potential pass boundaries
    var inPass bool
    var passStart time.Time
    var prevEl float64
    
    for t := start; t.Before(end); t = t.Add(step) {
        state, err := pf.propagator.Propagate(t)
        if err != nil {
            continue
        }
        
        az, el, _ := pf.computeLookAngles(state)
        
        if !inPass && el > 0 && prevEl <= 0 {
            // Rising - potential AOS
            inPass = true
            passStart = t.Add(-step) // Search from before crossing
        } else if inPass && el <= 0 && prevEl > 0 {
            // Setting - potential LOS
            inPass = false
            
            // Refine this pass
            pass, err := pf.refinePas(passStart, t)
            if err == nil && pass.MaxElevation >= pf.config.MinElevation {
                passes = append(passes, *pass)
            }
        }
        
        prevEl = el
    }
    
    // Handle case where pass extends beyond end time
    if inPass {
        pass, err := pf.refinePass(passStart, end)
        if err == nil && pass.MaxElevation >= pf.config.MinElevation {
            passes = append(passes, *pass)
        }
    }
    
    // Sort by AOS
    sort.Slice(passes, func(i, j int) bool {
        return passes[i].AOS.Before(passes[j].AOS)
    })
    
    return passes, nil
}

// refinePass refines a rough pass to find exact AOS, LOS, and TCA
func (pf *PassFinder) refinePass(roughStart, roughEnd time.Time) (*Pass, error) {
    // Find exact AOS using binary search
    aos, err := pf.findHorizonCrossing(roughStart, roughStart.Add(2*pf.config.CoarseStep), true)
    if err != nil {
        return nil, err
    }
    
    // Find exact LOS using binary search
    los, err := pf.findHorizonCrossing(roughEnd.Add(-2*pf.config.CoarseStep), roughEnd, false)
    if err != nil {
        return nil, err
    }
    
    // Find TCA (maximum elevation) using golden section search
    tca, maxEl, err := pf.findMaxElevation(aos, los)
    if err != nil {
        return nil, err
    }
    
    // Get angles at key times
    aosState, _ := pf.propagator.Propagate(aos)
    losState, _ := pf.propagator.Propagate(los)
    tcaState, _ := pf.propagator.Propagate(tca)
    
    aosAz, _, _ := pf.computeLookAngles(aosState)
    losAz, _, _ := pf.computeLookAngles(losState)
    tcaAz, _, _ := pf.computeLookAngles(tcaState)
    
    pass := &Pass{
        Satellite:    pf.propagator.tle.Name,
        Station:      pf.station.ID,
        AOS:          aos,
        LOS:          los,
        TCA:          tca,
        AOSAzimuth:   radToDeg(aosAz),
        LOSAzimuth:   radToDeg(losAz),
        TCAAnzimuth:  radToDeg(tcaAz),
        MaxElevation: radToDeg(maxEl),
        Duration:     los.Sub(aos),
    }
    
    // Generate detailed points if configured
    if pf.config.PointInterval > 0 {
        pass.Points = pf.generatePassPoints(aos, los)
    }
    
    return pass, nil
}

// findHorizonCrossing uses binary search to find the horizon crossing time
func (pf *PassFinder) findHorizonCrossing(start, end time.Time, rising bool) (time.Time, error) {
    minElRad := degToRad(pf.config.MinElevation)
    
    for end.Sub(start) > pf.config.Precision {
        mid := start.Add(end.Sub(start) / 2)
        
        state, err := pf.propagator.Propagate(mid)
        if err != nil {
            return time.Time{}, err
        }
        
        _, el, _ := pf.computeLookAngles(state)
        
        if rising {
            if el < minElRad {
                start = mid
            } else {
                end = mid
            }
        } else {
            if el >= minElRad {
                start = mid
            } else {
                end = mid
            }
        }
    }
    
    return start.Add(end.Sub(start) / 2), nil
}

// findMaxElevation uses golden section search to find maximum elevation
func (pf *PassFinder) findMaxElevation(start, end time.Time) (time.Time, float64, error) {
    phi := (1.0 + math.Sqrt(5.0)) / 2.0 // Golden ratio
    
    a := start
    b := end
    
    // Golden section search
    for b.Sub(a) > pf.config.Precision {
        x1 := b.Add(-time.Duration(float64(b.Sub(a)) / phi))
        x2 := a.Add(time.Duration(float64(b.Sub(a)) / phi))
        
        state1, _ := pf.propagator.Propagate(x1)
        state2, _ := pf.propagator.Propagate(x2)
        
        _, el1, _ := pf.computeLookAngles(state1)
        _, el2, _ := pf.computeLookAngles(state2)
        
        if el1 > el2 {
            b = x2
        } else {
            a = x1
        }
    }
    
    tca := a.Add(b.Sub(a) / 2)
    state, err := pf.propagator.Propagate(tca)
    if err != nil {
        return time.Time{}, 0, err
    }
    
    _, maxEl, _ := pf.computeLookAngles(state)
    
    return tca, maxEl, nil
}

// computeLookAngles computes azimuth, elevation, and range from station to satellite
func (pf *PassFinder) computeLookAngles(state *StateVector) (az, el, rng float64) {
    // Convert satellite position from TEME to ECEF
    satECEF := temeToECEF(state.Position, state.Time)
    
    // Station position in ECEF
    stationECEF := pf.station.ECEFPosition
    
    // Vector from station to satellite
    dx := satECEF.X - stationECEF.X
    dy := satECEF.Y - stationECEF.Y
    dz := satECEF.Z - stationECEF.Z
    
    // Range
    rng = math.Sqrt(dx*dx + dy*dy + dz*dz)
    
    // Convert to topocentric (SEZ or ENU)
    lat := degToRad(pf.station.Latitude)
    lon := degToRad(pf.station.Longitude)
    
    sinLat := math.Sin(lat)
    cosLat := math.Cos(lat)
    sinLon := math.Sin(lon)
    cosLon := math.Cos(lon)
    
    // Rotation to ENU (East-North-Up)
    e := -sinLon*dx + cosLon*dy
    n := -sinLat*cosLon*dx - sinLat*sinLon*dy + cosLat*dz
    u := cosLat*cosLon*dx + cosLat*sinLon*dy + sinLat*dz
    
    // Azimuth (from North, clockwise)
    az = math.Atan2(e, n)
    if az < 0 {
        az += 2 * math.Pi
    }
    
    // Elevation
    el = math.Asin(u / rng)
    
    return az, el, rng
}

// generatePassPoints creates detailed tracking points for a pass
func (pf *PassFinder) generatePassPoints(aos, los time.Time) []PassPoint {
    points := make([]PassPoint, 0)
    
    freq := 437.0e6 // Default frequency for Doppler (can be parameterized)
    
    for t := aos; t.Before(los) || t.Equal(los); t = t.Add(pf.config.PointInterval) {
        state, err := pf.propagator.Propagate(t)
        if err != nil {
            continue
        }
        
        az, el, rng := pf.computeLookAngles(state)
        rangeRate := pf.computeRangeRate(state)
        doppler := pf.computeDoppler(rangeRate, freq)
        
        points = append(points, PassPoint{
            Time:      t,
            Azimuth:   radToDeg(az),
            Elevation: radToDeg(el),
            Range:     rng,
            RangeRate: rangeRate,
            Doppler:   doppler,
        })
    }
    
    // Ensure we have the exact LOS point
    if len(points) > 0 && !points[len(points)-1].Time.Equal(los) {
        state, _ := pf.propagator.Propagate(los)
        az, el, rng := pf.computeLookAngles(state)
        rangeRate := pf.computeRangeRate(state)
        doppler := pf.computeDoppler(rangeRate, freq)
        
        points = append(points, PassPoint{
            Time:      los,
            Azimuth:   radToDeg(az),
            Elevation: radToDeg(el),
            Range:     rng,
            RangeRate: rangeRate,
            Doppler:   doppler,
        })
    }
    
    return points
}
```

```go
package orbital

// GeodeticToECEF converts geodetic to ECEF coordinates
func GeodeticToECEF(lat, lon, alt float64) Vector3 {
    // Convert lat/lon to radians
    latRad := degToRad(lat)
    lonRad := degToRad(lon)
    
    // WGS-84 parameters
    a := 6378.137            // Equatorial radius (km)
    f := 1.0 / 298.257223563 // Flattening
    e2 := 2*f - f*f          // First eccentricity squared
    
    // Radius of curvature in prime vertical
    sinLat := math.Sin(latRad)
    cosLat := math.Cos(latRad)
    N := a / math.Sqrt(1-e2*sinLat*sinLat)
    
    // Altitude in km
    altKm := alt / 1000.0
    
    // ECEF position
    return Vector3{
        X: (N + altKm) * cosLat * math.Cos(lonRad),
        Y: (N + altKm) * cosLat * math.Sin(lonRad),
        Z: (N*(1-e2) + altKm) * sinLat,
    }
}

// ECEFToGeodetic converts ECEF to geodetic coordinates
func ECEFToGeodetic(pos Vector3) (lat, lon, alt float64) {
    // WGS-84 parameters
    a := 6378.137
    f := 1.0 / 298.257223563
    b := a * (1 - f)
    e2 := (a*a - b*b) / (a * a)
    ep2 := (a*a - b*b) / (b * b)
    
    // Longitude
    lon = math.Atan2(pos.Y, pos.X)
    
    // Distance from Z-axis
    p := math.Sqrt(pos.X*pos.X + pos.Y*pos.Y)
    
    // Iterative solution for latitude (Bowring's method)
    theta := math.Atan2(pos.Z*a, p*b)
    
    lat = math.Atan2(
        pos.Z+ep2*b*math.Pow(math.Sin(theta), 3),
        p-e2*a*math.Pow(math.Cos(theta), 3),
    )
    
    // Altitude
    sinLat := math.Sin(lat)
    cosLat := math.Cos(lat)
    N := a / math.Sqrt(1-e2*sinLat*sinLat)
    
    if math.Abs(cosLat) > 1e-10 {
        alt = p/cosLat - N
    } else {
        alt = math.Abs(pos.Z) - b
    }
    
    // Convert to degrees
    lat = radToDeg(lat)
    lon = radToDeg(lon)
    alt = alt * 1000 // km to meters
    
    return lat, lon, alt
}

// temeToECEF converts TEME to ECEF at a given time
func temeToECEF(teme Vector3, t time.Time) Vector3 {
    // Calculate Greenwich Mean Sidereal Time
    gmst := greenwichMeanSiderealTime(t)
    
    // Rotation matrix (TEME is approximately ECI)
    cosGMST := math.Cos(gmst)
    sinGMST := math.Sin(gmst)
    
    return Vector3{
        X: cosGMST*teme.X + sinGMST*teme.Y,
        Y: -sinGMST*teme.X + cosGMST*teme.Y,
        Z: teme.Z,
    }
}

// greenwichMeanSiderealTime computes GMST in radians
func greenwichMeanSiderealTime(t time.Time) float64 {
    // Julian date
    jd := julianDate(t)
    
    // Julian centuries from J2000.0
    T := (jd - 2451545.0) / 36525.0
    
    // GMST in seconds
    gmst := 67310.54841 +
        (876600.0*3600.0+8640184.812866)*T +
        0.093104*T*T -
        6.2e-6*T*T*T
    
    // Convert to radians
    gmst = math.Mod(gmst, 86400.0)
    if gmst < 0 {
        gmst += 86400.0
    }
    
    return gmst * (2.0 * math.Pi / 86400.0)
}

// julianDate computes Julian Date from time.Time
func julianDate(t time.Time) float64 {
    year := float64(t.Year())
    month := float64(t.Month())
    day := float64(t.Day())
    
    hour := float64(t.Hour())
    minute := float64(t.Minute())
    second := float64(t.Second()) + float64(t.Nanosecond())/1e9
    
    // Adjust for January/February
    if month <= 2 {
        year -= 1
        month += 12
    }
    
    // Julian date calculation
    A := math.Floor(year / 100)
    B := 2 - A + math.Floor(A/4)
    
    jd := math.Floor(365.25*(year+4716)) +
        math.Floor(30.6001*(month+1)) +
        day + B - 1524.5
    
    // Add time of day
    jd += (hour + minute/60.0 + second/3600.0) / 24.0
    
    return jd
}
```

```go
package orbital

import (
    "math"

// VisibilityMask represents terrain obstructions for a ground station
type VisibilityMask struct {
    StationID   string
    Points      []MaskPoint // Sorted by azimuth
    DefaultElev float64     // Default minimum elevation if no mask
}

// MaskPoint represents one point on the visibility mask
type MaskPoint struct {
    Azimuth   float64 // degrees, 0-360
    Elevation float64 // degrees, minimum elevation at this azimuth
}

// NewVisibilityMask creates a visibility mask
func NewVisibilityMask(points []MaskPoint, defaultElev float64) *VisibilityMask {
    // Sort by azimuth
    sort.Slice(points, func(i, j int) bool {
        return points[i].Azimuth < points[j].Azimuth
    })
    
    return &VisibilityMask{
        Points:      points,
        DefaultElev: defaultElev,
    }
}

// MinElevationAt returns the minimum visible elevation at a given azimuth
func (vm *VisibilityMask) MinElevationAt(azimuth float64) float64 {
    if len(vm.Points) == 0 {
        return vm.DefaultElev
    }
    
    // Normalize azimuth to 0-360
    for azimuth < 0 {
        azimuth += 360
    }
    for azimuth >= 360 {
        azimuth -= 360
    }
    
    // Find bracketing points
    var lower, upper *MaskPoint
    
    for i := range vm.Points {
        if vm.Points[i].Azimuth <= azimuth {
            lower = &vm.Points[i]
        }
        if vm.Points[i].Azimuth >= azimuth && upper == nil {
            upper = &vm.Points[i]
        }
    }
    
    // Handle wrap-around
    if lower == nil {
        lower = &vm.Points[len(vm.Points)-1]
    }
    if upper == nil {
        upper = &vm.Points[0]
    }
    
    // Interpolate
    if lower.Azimuth == upper.Azimuth {
        return lower.Elevation
    }
    
    // Handle wrap-around for interpolation
    azRange := upper.Azimuth - lower.Azimuth
    if azRange < 0 {
        azRange += 360
    }
    
    azDiff := azimuth - lower.Azimuth
    if azDiff < 0 {
        azDiff += 360
    }
    
    t := azDiff / azRange
    return lower.Elevation + t*(upper.Elevation-lower.Elevation)
}

// IsVisible returns true if the given az/el is above the mask
func (vm *VisibilityMask) IsVisible(azimuth, elevation float64) bool {
    minEl := vm.MinElevationAt(azimuth)
    return elevation >= minEl
}
```

### 6.2 Range Rate Calculation

```go
// doppler.go

package orbital

import (
    "math"
)

const SpeedOfLight = 299792.458 // km/s

// computeRangeRate calculates the rate of change of range
func (pf *PassFinder) computeRangeRate(state *StateVector) float64 {
    // Current satellite position and velocity in ECEF
    satPos := temeToECEF(state.Position, state.Time)
    satVel := temeVelocityToECEF(state.Velocity, state.Position, state.Time)
    
    // Station position (stationary in ECEF)
    stationPos := pf.station.ECEFPosition
    
    // Vector from station to satellite
    dx := satPos.X - stationPos.X
    dy := satPos.Y - stationPos.Y
    dz := satPos.Z - stationPos.Z
    
    // Range magnitude
    rng := math.Sqrt(dx*dx + dy*dy + dz*dz)
    
    // Range unit vector
    ux := dx / rng
    uy := dy / rng
    uz := dz / rng
    
    // Range rate = velocity dot range_unit_vector
    rangeRate := satVel.X*ux + satVel.Y*uy + satVel.Z*uz
    
    return rangeRate
}

// computeDoppler calculates Doppler shift for a given frequency
func (pf *PassFinder) computeDoppler(rangeRate, frequency float64) float64 {
    // Doppler shift: Δf = -f × (v_r / c)
    // Negative because approaching = positive Doppler (higher frequency)
    return -frequency * (rangeRate / SpeedOfLight)
}

// temeVelocityToECEF converts TEME velocity to ECEF
// Accounts for Earth's rotation
func temeVelocityToECEF(velTEME, posTEME Vector3, t time.Time) Vector3 {
    gmst := greenwichMeanSiderealTime(t)
    cosGMST := math.Cos(gmst)
    sinGMST := math.Sin(gmst)
    
    // Earth rotation rate (rad/s)
    omega := 7.292115e-5
    
    // ECEF velocity includes rotation effect
    return Vector3{
        X: cosGMST*velTEME.X + sinGMST*velTEME.Y + omega*(-sinGMST*posTEME.X+cosGMST*posTEME.Y),
        Y: -sinGMST*velTEME.X + cosGMST*velTEME.Y + omega*(-cosGMST*posTEME.X-sinGMST*posTEME.Y),
        Z: velTEME.Z,
    }
}

// DopplerProfile represents Doppler shift over a pass
type DopplerProfile struct {
    Points    []DopplerPoint
    MaxShift  float64
    MinShift  float64
    CrossTime time.Time // Time of zero Doppler
}

// DopplerPoint represents Doppler at a specific time
type DopplerPoint struct {
    Time      time.Time
    Doppler   float64 // Hz
    DopplerRate float64 // Hz/s
}

// GenerateDopplerProfile creates a detailed Doppler profile for a pass
func (pf *PassFinder) GenerateDopplerProfile(pass *Pass, frequency float64, interval time.Duration) *DopplerProfile {
    profile := &DopplerProfile{
        Points:   make([]DopplerPoint, 0),
        MaxShift: math.Inf(-1),
        MinShift: math.Inf(1),
    }
    
    var prevDoppler float64
    var prevTime time.Time
    
    for t := pass.AOS; t.Before(pass.LOS) || t.Equal(pass.LOS); t = t.Add(interval) {
        state, err := pf.propagator.Propagate(t)
        if err != nil {
            continue
        }
        
        rangeRate := pf.computeRangeRate(state)
        doppler := pf.computeDoppler(rangeRate, frequency)
        
        // Calculate Doppler rate
        var dopplerRate float64
        if !prevTime.IsZero() {
            dt := t.Sub(prevTime).Seconds()
            dopplerRate = (doppler - prevDoppler) / dt
        }
        
        profile.Points = append(profile.Points, DopplerPoint{
            Time:        t,
            Doppler:     doppler,
            DopplerRate: dopplerRate,
        })
        
        // Track extremes
        if doppler > profile.MaxShift {
            profile.MaxShift = doppler
        }
        if doppler < profile.MinShift {
            profile.MinShift = doppler
        }
        
        // Detect zero crossing
        if prevDoppler > 0 && doppler <= 0 {
            // Interpolate to find exact crossing time
            frac := prevDoppler / (prevDoppler - doppler)
            profile.CrossTime = prevTime.Add(time.Duration(frac * float64(interval)))
        }
        
        prevDoppler = doppler
        prevTime = t
    }
    
    return profile
}
```

```go
// GetCompensatedFrequency returns the frequency to tune to for a given time
func (dc *DopplerCompensator) GetCompensatedFrequency(t time.Time) (float64, error) {
    pf := NewPassFinder(dc.propagator, dc.station, DefaultPassFinderConfig())
    
    state, err := dc.propagator.Propagate(t)
    if err != nil {
        return dc.frequency, err
    }
    
    rangeRate := pf.computeRangeRate(state)
    doppler := pf.computeDoppler(rangeRate, dc.frequency)
    
    // Compensated frequency = base frequency - Doppler shift
    // (To receive f_transmitted, tune to f_transmitted + Doppler)
    return dc.frequency + doppler, nil
}

// GenerateFrequencyTable creates a lookup table for Doppler compensation
func (dc *DopplerCompensator) GenerateFrequencyTable(start, end time.Time, interval time.Duration) []FrequencyPoint {
    points := make([]FrequencyPoint, 0)
    
    for t := start; t.Before(end); t = t.Add(interval) {
        freq, err := dc.GetCompensatedFrequency(t)
        if err != nil {
            continue
        }
        
        points = append(points, FrequencyPoint{
            Time:      t,
            Frequency: freq,
        })
    }
    
    return points
}
```

```go
// PredictPasses predicts passes for a satellite over a station
func (ps *PredictionService) PredictPasses(
    ctx context.Context,
    req *PredictPassesRequest,
) (*PredictPassesResponse, error) {
    
    // Validate request
    if err := validateRequest(req); err != nil {
        return nil, err
    }
    
    // Check cache first
    cacheKey := ps.buildCacheKey(req)
    if cached, ok := ps.passCache.Get(cacheKey); ok {
        return cached.(*PredictPassesResponse), nil
    }
    
    // Get TLE for satellite
    tle, err := ps.tleManager.GetTLE(ctx, req.SatelliteID)
    if err != nil {
        return nil, fmt.Errorf("failed to get TLE: %w", err)
    }
    
    // Warn if TLE is stale
    if tle.IsStale(ps.config.RefreshTLEAge) {
        // Log warning but continue
        log.Warn("Using stale TLE", 
            "satellite", req.SatelliteID,
            "age", tle.Age())
    }
    
    // Get station
    station, err := ps.stationManager.GetStation(ctx, req.StationID)
    if err != nil {
        return nil, fmt.Errorf("failed to get station: %w", err)
    }
    
    // Get or create propagator
    propagator, err := ps.getOrCreatePropagator(tle)
    if err != nil {
        return nil, fmt.Errorf("failed to create propagator: %w", err)
    }
    
    // Configure pass finder
    minEl := req.MinElevation
    if minEl == 0 {
        minEl = ps.config.DefaultMinElevation
    }
    
    config := PassFinderConfig{
        MinElevation:  minEl,
        CoarseStep:    60 * time.Second,
        FineStep:      10 * time.Second,
        Precision:     100 * time.Millisecond,
        PointInterval: time.Duration(req.PointIntervalSec) * time.Second,
    }
    
    finder := NewPassFinder(propagator, station, config)
    
    // Find passes
    start := req.StartTime.AsTime()
    end := req.EndTime.AsTime()
    
    passes, err := finder.FindPasses(start, end)
    if err != nil {
        return nil, fmt.Errorf("failed to find passes: %w", err)
    }
    
    // Build response
    response := &PredictPassesResponse{
        Passes:      make([]*PassInfo, len(passes)),
        TLEEpoch:    timestamppb.New(tle.Epoch),
        TLEAge:      durationpb.New(tle.Age()),
        ComputedAt:  timestamppb.Now(),
    }
    
    for i, pass := range passes {
        response.Passes[i] = passToProto(&pass)
    }
    
    // Cache response
    ps.passCache.Set(cacheKey, response, ps.config.CachePassDuration)
    
    return response, nil
}

    if err != nil {
        return nil, err
    }
    
    // Propagate to requested time
    t := req.Time.AsTime()
    state, err := propagator.Propagate(t)
    if err != nil {
        return nil, err
    }
    
    // Compute look angles
    finder := NewPassFinder(propagator, station, DefaultPassFinderConfig())
    az, el, rng := finder.computeLookAngles(state)
    rangeRate := finder.computeRangeRate(state)
    doppler := finder.computeDoppler(rangeRate, req.Frequency)
    
    // Convert satellite position to geodetic
    satECEF := temeToECEF(state.Position, t)
    satLat, satLon, satAlt := ECEFToGeodetic(satECEF)
    
    return &TrackingData{
        Time:            req.Time,
        Azimuth:         radToDeg(az),
        Elevation:       radToDeg(el),
        Range:           rng,
        RangeRate:       rangeRate,
        Doppler:         doppler,
        SatelliteLat:    satLat,
        SatelliteLon:    satLon,
        SatelliteAlt:    satAlt,
        IsVisible:       el > 0,
    }, nil
}

// StreamTrackingData streams real-time tracking updates
func (ps *PredictionService) StreamTrackingData(
    req *StreamTrackingRequest,
    stream TrackingService_StreamTrackingDataServer,
) error {
    
    interval := time.Duration(req.IntervalMs) * time.Millisecond
    if interval < 100*time.Millisecond {
        interval = 100 * time.Millisecond
    }
    
    ticker := time.NewTicker(interval)
    defer ticker.Stop()
    
    for {
        select {
        case <-stream.Context().Done():
            return nil
            
        case t := <-ticker.C:
            data, err := ps.GetTrackingData(stream.Context(), &GetTrackingDataRequest{
                SatelliteID: req.SatelliteID,
                StationID:   req.StationID,
                Time:        timestamppb.New(t),
                Frequency:   req.Frequency,
            })
            
            if err != nil {
                continue
            }
            
            if err := stream.Send(data); err != nil {
                return err
            }
        }
    }
}
```

```go
// accuracy.go

package orbital

import (
    "math"
    "time"
)

// AccuracyEstimate provides estimated prediction accuracy
type AccuracyEstimate struct {
    PositionError     float64       // km
    TimingError       time.Duration // AOS/LOS uncertainty
    ElevationError    float64       // degrees
    DopplerError      float64       // Hz
    ConfidenceLevel   float64       // 0-1
}

// EstimateAccuracy estimates prediction accuracy based on TLE age and orbit
func EstimateAccuracy(tle *TLE, predictionTime time.Time) *AccuracyEstimate {
    age := predictionTime.Sub(tle.Epoch)
    ageHours := age.Hours()
    
    // Base error grows with time
    // Position error model: ~0.5 km at epoch, growing exponentially
    basePosError := 0.5 // km at epoch
    
    // Growth rate depends on altitude and drag
    perigee := tle.Perigee()
    growthRate := 0.0
    
    if perigee < 400 {
        growthRate = 0.05 // High drag LEO
    } else if perigee < 800 {
        growthRate = 0.02 // Medium LEO
    } else {
        growthRate = 0.01 // Higher orbits
    }
    
    posError := basePosError * math.Exp(growthRate*ageHours)
    
    // Timing error from position error
    // Satellite velocity ~7.5 km/s for LEO
    velocity := 7.5 // km/s approximate
    timingError := time.Duration(posError/velocity*1000) * time.Millisecond
    
    // Elevation error (very rough estimate)
    elevError := posError / 500.0 // degrees
    
    // Doppler error (depends on frequency)
    dopplerError := posError * 0.1 // Hz per km of position error (rough)
    
    // Confidence decreases with age
    confidence := 1.0 / (1.0 + ageHours/24.0)
    
    return &AccuracyEstimate{
        PositionError:   posError,
        TimingError:     timingError,
        ElevationError:  elevError,
        DopplerError:    dopplerError,
        ConfidenceLevel: confidence,
    }
}
```

```go
package orbital

import (

// BatchPassFinder finds passes for multiple satellite/station pairs
type BatchPassFinder struct {
    workers    int
    tleCache   *TLECache
    stationDB  *StationDB
}

// BatchRequest represents a batch prediction request
type BatchRequest struct {
    SatelliteIDs []string
    StationIDs   []string
    StartTime    time.Time
    EndTime      time.Time
    MinElevation float64
}

// BatchResult contains results for one satellite/station pair
type BatchResult struct {
    SatelliteID string
    StationID   string
    Passes      []Pass
    Error       error
}

// FindPassesBatch finds passes for multiple pairs in parallel
func (bf *BatchPassFinder) FindPassesBatch(req *BatchRequest) []BatchResult {
    // Create work items
    type workItem struct {
        satID     string
        stationID string
    }
    
    items := make([]workItem, 0)
    for _, satID := range req.SatelliteIDs {
        for _, stationID := range req.StationIDs {
            items = append(items, workItem{satID, stationID})
        }
    }
    
    // Create result channel
    results := make(chan BatchResult, len(items))
    
    // Create work channel
    work := make(chan workItem, len(items))
    for _, item := range items {
        work <- item
    }
    close(work)
    
    // Start workers
    var wg sync.WaitGroup
    for i := 0; i < bf.workers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            
            for item := range work {
                result := BatchResult{
                    SatelliteID: item.satID,
                    StationID:   item.stationID,
                }
                
                // Get TLE and station
                tle, err := bf.tleCache.Get(item.satID)
                if err != nil {
                    result.Error = err
                    results <- result
                    continue
                }
                
                station, err := bf.stationDB.Get(item.stationID)
                if err != nil {
                    result.Error = err
                    results <- result
                    continue
                }
                
                // Find passes
                prop, _ := NewSGP4Propagator(tle)
                finder := NewPassFinder(prop, station, PassFinderConfig{
                    MinElevation: req.MinElevation,
                    CoarseStep:   60 * time.Second,
                })
                
                passes, err := finder.FindPasses(req.StartTime, req.EndTime)
                if err != nil {
                    result.Error = err
                } else {
                    result.Passes = passes
                }
                
                results <- result
            }
        }()
    }
    
    // Wait for completion
    go func() {
        wg.Wait()
        close(results)
    }()
    
    // Collect results
    allResults := make([]BatchResult, 0, len(items))
    for result := range results {
        allResults = append(allResults, result)
    }
    
    return allResults
}

// PropagatorPool manages a pool of reusable propagators
type PropagatorPool struct {
    propagators sync.Map // satelliteID -> *poolEntry
}

type poolEntry struct {
    propagator *SGP4Propagator
    tleEpoch   time.Time
    mu         sync.Mutex
}

// Get retrieves or creates a propagator for a satellite
func (pp *PropagatorPool) Get(tle *TLE) (*SGP4Propagator, error) {
    key := tle.CatalogNumber
    
    // Check if we have a valid propagator
    if entry, ok := pp.propagators.Load(key); ok {
        pe := entry.(*poolEntry)
        pe.mu.Lock()
        defer pe.mu.Unlock()
        
        // Check if TLE is still current
        if pe.tleEpoch.Equal(tle.Epoch) {
            return pe.propagator, nil
        }
        
        // TLE changed, create new propagator
        prop, err := NewSGP4Propagator(tle)
        if err != nil {
            return nil, err
        }
        pe.propagator = prop
        pe.tleEpoch = tle.Epoch
        return prop, nil
    }
    
    // Create new propagator
    prop, err := NewSGP4Propagator(tle)
    if err != nil {
        return nil, err
    }
    
    pp.propagators.Store(key, &poolEntry{
        propagator: prop,
        tleEpoch:   tle.Epoch,
    })
    
    return prop, nil
}
```

```go
// caching.go

package orbital

import (
    "crypto/sha256"
    "encoding/hex"
    "time"
    
    "github.com/redis/go-redis/v9"
)

// PassCache caches predicted passes
type PassCache struct {
    redis      *redis.Client
    ttl        time.Duration
    keyPrefix  string
}

// CacheKey generates a unique key for a prediction request
func (pc *PassCache) CacheKey(satID, stationID string, start, end time.Time, minEl float64) string {
    // Round times to minute for cache efficiency
    startRounded := start.Truncate(time.Minute)
    endRounded := end.Truncate(time.Minute)
    
    data := fmt.Sprintf("%s:%s:%d:%d:%.1f",
        satID, stationID,
        startRounded.Unix(), endRounded.Unix(),
        minEl)
    
    hash := sha256.Sum256([]byte(data))
    return pc.keyPrefix + hex.EncodeToString(hash[:8])
}

// Get retrieves cached passes
func (pc *PassCache) Get(ctx context.Context, key string) ([]Pass, bool) {
    data, err := pc.redis.Get(ctx, key).Bytes()
    if err != nil {
        return nil, false
    }
    
    var passes []Pass
    if err := json.Unmarshal(data, &passes); err != nil {
        return nil, false
    }
    
    return passes, true
}

// Set stores passes in cache
func (pc *PassCache) Set(ctx context.Context, key string, passes []Pass) error {
    data, err := json.Marshal(passes)
    if err != nil {
        return err
    }
    
    return pc.redis.Set(ctx, key, data, pc.ttl).Err()
}

// InvalidateForSatellite invalidates all passes for a satellite
// (e.g., when TLE is updated)
func (pc *PassCache) InvalidateForSatellite(ctx context.Context, satID string) error {
    pattern := pc.keyPrefix + satID + ":*"
    
    var cursor uint64
    for {
        keys, cursor, err := pc.redis.Scan(ctx, cursor, pattern, 100).Result()
        if err != nil {
            return err
        }
        
        if len(keys) > 0 {
            pc.redis.Del(ctx, keys...)
        }
        
        if cursor == 0 {
            break
        }
    }
    
    return nil
}
```

```go
// TLEManager handles TLE acquisition and caching
type TLEManager struct {
    sources       []TLESource
    cache         *TLECache
    refreshPeriod time.Duration
    maxAge        time.Duration
    
    mu            sync.RWMutex
    lastRefresh   map[int]time.Time // NORAD ID -> last refresh time
}

// TLESource interface for TLE providers
type TLESource interface {
    Name() string
    Fetch(ctx context.Context, noradID int) (*TLE, error)
    FetchBulk(ctx context.Context, noradIDs []int) ([]*TLE, error)
    FetchCategory(ctx context.Context, category string) ([]*TLE, error)
}

// SpaceTrackSource fetches TLEs from Space-Track.org
type SpaceTrackSource struct {
    username    string
    password    string
    baseURL     string
    client      *http.Client
    rateLimiter *RateLimiter
}

// Fetch retrieves TLE for a single satellite
func (st *SpaceTrackSource) Fetch(ctx context.Context, noradID int) (*TLE, error) {
    // Rate limiting
    if err := st.rateLimiter.Wait(ctx); err != nil {
        return nil, err
    }
    
    // Build query URL
    url := fmt.Sprintf("%s/basicspacedata/query/class/gp/NORAD_CAT_ID/%d/orderby/EPOCH%%20desc/limit/1/format/tle",
        st.baseURL, noradID)
    
    // Make authenticated request
    req, err := http.NewRequestWithContext(ctx, "GET", url, nil)
    if err != nil {
        return nil, err
    }
    
    // Add authentication cookie (obtained during login)
    // ...
    
    resp, err := st.client.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()
    
    // Parse TLE response
    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return nil, err
    }
    
    lines := strings.Split(strings.TrimSpace(string(body)), "\n")
    if len(lines) < 2 {
        return nil, fmt.Errorf("invalid TLE response")
    }
    
    return ParseTLE(lines)
}

// GetTLE retrieves TLE for a satellite, using cache when possible
func (tm *TLEManager) GetTLE(ctx context.Context, satelliteID string) (*TLE, error) {
    // Check cache first
    if tle, ok := tm.cache.Get(satelliteID); ok {
        if !tle.IsStale(tm.maxAge) {
            return tle, nil
        }
    }
    
    // Parse satellite ID (could be NORAD ID or name)
    noradID, err := parseNoradID(satelliteID)
    if err != nil {
        return nil, fmt.Errorf("invalid satellite ID: %w", err)
    }
    
    // Try each source in order
    for _, source := range tm.sources {
        tle, err := source.Fetch(ctx, noradID)
        if err != nil {
            log.Warn("TLE source failed", 
                "source", source.Name(), 
                "norad_id", noradID,
                "error", err)
            continue
        }
        
        // Cache and return
        tle.Source = source.Name()
        tm.cache.Set(satelliteID, tle)
        
        tm.mu.Lock()
        tm.lastRefresh[noradID] = time.Now()
        tm.mu.Unlock()
        
        return tle, nil
    }
    
    // Fall back to cached (even if stale)
    if tle, ok := tm.cache.Get(satelliteID); ok {
        log.Warn("Using stale TLE", "satellite", satelliteID, "age", tle.Age())
        return tle, nil
    }
    
    return nil, fmt.Errorf("TLE not available for %s", satelliteID)
}

// RefreshAll refreshes TLEs for all tracked satellites
func (tm *TLEManager) RefreshAll(ctx context.Context) error {
    // Get list of tracked satellites
    satellites := tm.cache.ListSatellites()
    
    // Group by source for bulk fetching
    noradIDs := make([]int, 0, len(satellites))
    for _, sat := range satellites {
        if id, err := parseNoradID(sat); err == nil {
            noradIDs = append(noradIDs, id)
        }
    }
    
    // Bulk fetch from primary source
    if len(tm.sources) > 0 {
        tles, err := tm.sources[0].FetchBulk(ctx, noradIDs)
        if err != nil {
            return err
        }
        
        for _, tle := range tles {
            tm.cache.Set(fmt.Sprintf("%d", tle.CatalogNumber), tle)
        }
    }
    
    return nil
}

// TLECache provides in-memory and Redis-backed TLE caching
type TLECache struct {
    memory sync.Map
    redis  *redis.Client
    ttl    time.Duration
}

// Set caches a TLE
func (tc *TLECache) Set(satelliteID string, tle *TLE) {
    // Memory cache
    tc.memory.Store(satelliteID, tle)
    
    // Redis cache
    if tc.redis != nil {
        data, _ := json.Marshal(tle)
        tc.redis.Set(context.Background(), "tle:"+satelliteID, data, tc.ttl)
    }
}

// Get retrieves a cached TLE
func (tc *TLECache) Get(satelliteID string) (*TLE, bool) {
    // Check memory first
    if val, ok := tc.memory.Load(satelliteID); ok {
        return val.(*TLE), true
    }
    
    // Check Redis
    if tc.redis != nil {
        data, err := tc.redis.Get(context.Background(), "tle:"+satelliteID).Bytes()
        if err == nil {
            var tle TLE
            if json.Unmarshal(data, &tle) == nil {
                tc.memory.Store(satelliteID, &tle)
                return &tle, true
            }
        }
    }
    
    return nil, false
}
```

```go
    gt := &GroundTrack{
        Points:   make([]GroundTrackPoint, 0),
        Duration: duration,
    }
    
    end := start.Add(duration)
    
    for t := start; t.Before(end); t = t.Add(interval) {

        if err != nil {
            continue
        }

        satECEF := temeToECEF(state.Position, t)

        gt.Points = append(gt.Points, GroundTrackPoint{
            Time:       t,
            Latitude:   lat,
            Longitude:  lon,
            Altitude:   alt / 1000.0, // Convert to km
            InSunlight: inSunlight,
        })
    }
    
    return gt
}

    }
    
    return map[string]interface{}{
        "type": "Feature",
        "geometry": map[string]interface{}{
            "type":        "LineString",
            "coordinates": coordinates,
        },
        "properties": map[string]interface{}{
```

```go
// polar_plot.go

package orbital

import (
    "math"
)

// PolarPlot represents a pass visualization in polar coordinates
type PolarPlot struct {
    Points []PolarPoint
    Pass   *Pass
}

// PolarPoint represents a point on the polar plot
type PolarPoint struct {
    Time   time.Time
    Theta  float64 // Azimuth in radians
    R      float64 // 90 - Elevation (so zenith is at center)
}

// GeneratePolarPlot creates data for a polar plot visualization
func GeneratePolarPlot(pass *Pass) *PolarPlot {
    plot := &PolarPlot{
        Points: make([]PolarPoint, len(pass.Points)),
        Pass:   pass,
    }
    
    for i, pt := range pass.Points {
        plot.Points[i] = PolarPoint{
            Time:  pt.Time,
            Theta: degToRad(pt.Azimuth),
            R:     90 - pt.Elevation, // Invert so zenith is at center
        }
    }
    
    return plot
}

// ToSVGPath converts the polar plot to an SVG path string
func (pp *PolarPlot) ToSVGPath(radius float64) string {
    if len(pp.Points) == 0 {
        return ""
    }
    
    var path strings.Builder
    
    for i, pt := range pp.Points {
        // Convert polar to Cartesian
        // Note: SVG coordinates have Y increasing downward
        scale := radius / 90.0
        x := scale * pt.R * math.Sin(pt.Theta)
        y := -scale * pt.R * math.Cos(pt.Theta) // Negative for SVG coordinate system
        
        if i == 0 {
            path.WriteString(fmt.Sprintf("M %.2f %.2f", x, y))
        } else {
            path.WriteString(fmt.Sprintf(" L %.2f %.2f", x, y))
        }
    }
    
    return path.String()
}

// GenerateSVG creates a complete SVG polar plot
func (pp *PolarPlot) GenerateSVG(width, height int) string {
    cx := float64(width) / 2
    cy := float64(height) / 2
    radius := math.Min(cx, cy) * 0.9
    
    var svg strings.Builder
    
    // SVG header
    svg.WriteString(fmt.Sprintf(`<svg width="%d" height="%d" xmlns="http://www.w3.org/2000/svg">`, width, height))
    
    // Background
    svg.WriteString(`<rect width="100%" height="100%" fill="#1a1a2e"/>`)
    
    // Grid circles (elevation rings)
    for el := 0; el <= 90; el += 30 {
        r := radius * float64(90-el) / 90.0
        svg.WriteString(fmt.Sprintf(`<circle cx="%.1f" cy="%.1f" r="%.1f" fill="none" stroke="#333" stroke-width="1"/>`, cx, cy, r))
        svg.WriteString(fmt.Sprintf(`<text x="%.1f" y="%.1f" fill="#666" font-size="10">%d°</text>`, cx+5, cy-r, el))
    }
    
    // Cardinal directions
    directions := []struct {
        angle float64
        label string
    }{
        {0, "N"}, {90, "E"}, {180, "S"}, {270, "W"},
    }
    
    for _, dir := range directions {
        rad := degToRad(dir.angle)
        x1 := cx + radius*0.05*math.Sin(rad)
        y1 := cy - radius*0.05*math.Cos(rad)
        x2 := cx + radius*math.Sin(rad)
        y2 := cy - radius*math.Cos(rad)
        
        svg.WriteString(fmt.Sprintf(`<line x1="%.1f" y1="%.1f" x2="%.1f" y2="%.1f" stroke="#333" stroke-width="1"/>`, x1, y1, x2, y2))
        
        tx := cx + (radius+15)*math.Sin(rad)
        ty := cy - (radius+15)*math.Cos(rad)
        svg.WriteString(fmt.Sprintf(`<text x="%.1f" y="%.1f" fill="#999" font-size="14" text-anchor="middle" dominant-baseline="middle">%s</text>`, tx, ty, dir.label))
    }
    
    // Pass path
    path := pp.ToSVGPath(radius)
    svg.WriteString(fmt.Sprintf(`<g transform="translate(%.1f, %.1f)">`, cx, cy))
    svg.WriteString(fmt.Sprintf(`<path d="%s" fill="none" stroke="#00ff00" stroke-width="2"/>`, path))
    
    // Mark AOS, TCA, LOS
    if len(pp.Points) > 0 {
        // AOS (first point)
        aos := pp.Points[0]
        x := radius / 90.0 * aos.R * math.Sin(aos.Theta)
        y := -radius / 90.0 * aos.R * math.Cos(aos.Theta)
        svg.WriteString(fmt.Sprintf(`<circle cx="%.1f" cy="%.1f" r="5" fill="#00ff00"/>`, x, y))
        
        // LOS (last point)
        los := pp.Points[len(pp.Points)-1]
        x = radius / 90.0 * los.R * math.Sin(los.Theta)
        y = -radius / 90.0 * los.R * math.Cos(los.Theta)
        svg.WriteString(fmt.Sprintf(`<circle cx="%.1f" cy="%.1f" r="5" fill="#ff0000"/>`, x, y))
    }
    
    svg.WriteString(`</g>`)
    svg.WriteString(`</svg>`)
    
    return svg.String()
}
```
